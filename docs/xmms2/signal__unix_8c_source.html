40 sigwaiter (gpointer data)

43  xmms_object_cmd_arg_t arg;

74 void
75 xmms_signal_block (void)
76 {
77  sigset_t signals;
78 
79  sigemptyset(&signals);
80 
81  sigaddset (&signals, SIGHUP);
82  sigaddset (&signals, SIGTERM);
83  sigaddset (&signals, SIGINT);
84 
85  pthread_sigmask (SIG_BLOCK, &signals, &osignals);
86 
87  /* Thanks to bug #8533731 in CoreServices on Mac OS X, calling
88  * FindComponent/AudioComponentNext in the CoreAudio output
89  * plugin will cause SIGPIPE to be unblocked. To solve this
90  * we have to fend off SIGPIPE here instead of via sigmask.
91  * Doesn't affect the behavior on other platforms.
92  */
93  signal (SIGPIPE, SIG_IGN);
94 }
