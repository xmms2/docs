80 {

83 
84  sync = xmms_object_new (xmms_coll_sync_t, xmms_coll_sync_destroy);
85 
86  sync->uuid = g_strdup (uuid);
87 
88  g_cond_init (&sync->cond);
89  g_mutex_init (&sync->mutex);
90 
91  xmms_object_ref (dag);
92  sync->dag = dag;
93 
94  xmms_object_ref (playlist);
95  sync->playlist = playlist;
96 
97  path = XMMS_BUILD_PATH ("collections", "${uuid}.db");
98  sync->config = xmms_config_property_register ("collection.directory", path,
99  xmms_coll_sync_schedule_sync, sync);
100  g_free (path);
101 
102  /* Connection coll_sync_cb to some signals */
103  xmms_object_connect (XMMS_OBJECT (dag),
104  XMMS_IPC_SIGNAL_COLLECTION_CHANGED,
105  xmms_coll_sync_schedule_sync, sync);
106 
107  /* FIXME: These signals should trigger COLLECTION_CHANGED */
108  xmms_object_connect (XMMS_OBJECT (playlist),
109  XMMS_IPC_SIGNAL_PLAYLIST_CHANGED,
110  xmms_coll_sync_schedule_sync, sync);
111 
112  xmms_object_connect (XMMS_OBJECT (playlist),
113  XMMS_IPC_SIGNAL_PLAYLIST_CURRENT_POS,
114  xmms_coll_sync_schedule_sync, sync);
115 
116  xmms_object_connect (XMMS_OBJECT (playlist),
117  XMMS_IPC_SIGNAL_PLAYLIST_LOADED,
118  xmms_coll_sync_schedule_sync, sync);
119 
120  xmms_coll_sync_register_ipc_commands (XMMS_OBJECT (sync));
121 
122  /* First restore synchronously to make sure we are in a sensible state. */
123  xmms_coll_sync_restore (sync, FALSE);
124 
125  xmms_coll_sync_start (sync);
126

266 {
267  if (g_file_test (path, G_FILE_TEST_IS_REGULAR))
268  return TRUE;
269 
270  if (g_file_test (path, G_FILE_TEST_IS_DIR)) {
271  /* TODO: This branch can be removed after release as it will only
272  * be used for installations that were running development
273  * builds between 0.8 and 0.9
274  */
275  gint exit_status;
276  gchar *cmdline;
277 
278  exit_status = 0;
279 
280  cmdline = g_strdup_printf ("_xmms2-migrate-collections-v0 %s", path);
281 
282  if (!g_spawn_command_line_sync (cmdline, NULL, NULL, &exit_status, NULL) || exit_status) {
283  xmms_log_fatal ("Could not run \"%s\", try to run it manually", cmdline);
284  }
285 
286  /* If migration for some reason failed, just move the failing
287  * directory and leave room for a fresh one.
288  */
289  if (g_file_test (path, G_FILE_TEST_IS_DIR))
290  return xmms_coll_sync_move_to_legacy_path (path, error);
291  } else {
292  gchar *dirname;
293  gint res;
294 
295  dirname = g_path_get_dirname (path);
296  res = g_mkdir_with_parents (dirname, 0755);
297  g_free (dirname);

493 /**
494  * Wait until no collections have changed for 10 seconds, then sync.
495  * @internal
496  */
497 static gpointer
498 xmms_coll_sync_loop (gpointer udata)
499 {
500  xmms_coll_sync_t *sync = (xmms_coll_sync_t *) udata;
501 
502  g_mutex_lock (&sync->mutex);
503 
504  while (sync->state != XMMS_COLL_SYNC_STATE_SHUTDOWN) {
505  if (sync->state == XMMS_COLL_SYNC_STATE_IDLE) {
506  g_cond_wait (&sync->cond, &sync->mutex);
507  }
508 
509  /* Wait until no requests have been filed for 10 seconds. */
510  while (sync->state == XMMS_COLL_SYNC_STATE_DELAYED) {
511  gint64 end_time;
512 
513  sync->state = XMMS_COLL_SYNC_STATE_IDLE;
514 
515  end_time = g_get_monotonic_time () + XMMS_COLL_SYNC_DELAY;
516 
517  g_cond_wait_until (&sync->cond, &sync->mutex, end_time);
518  }
519 
520  if (sync->state != XMMS_COLL_SYNC_STATE_SHUTDOWN) {
521  /* The dag might be locked when calling schedule_sync, so we need to
522  * unlock to avoid deadlocks. We also reset state to IDLE to ack that
523  * the request to sync is being taken care of.
524  */
525  sync->state = XMMS_COLL_SYNC_STATE_IDLE;
526 
527  g_mutex_unlock (&sync->mutex);
528  xmms_coll_sync_save (sync);
529  g_mutex_lock (&sync->mutex);
530  }
531  }
532 
533  g_mutex_unlock (&sync->mutex);
534 
535  xmms_coll_sync_save (sync);
536 
537  return NULL;
538 }
