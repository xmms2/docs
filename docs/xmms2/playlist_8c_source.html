155 xmms_playlist_update_queue (xmms_playlist_t *playlist, const gchar *plname,
156  xmmsv_t *coll)
157 {
158  gint history, currpos;
159 
160  XMMS_DBG ("PLAYLIST: update queue.");
161 
162  if (!xmms_collection_get_int_attr (coll, "history", &history)) {
163  history = 0;
164  }
165 
166  currpos = xmms_playlist_coll_get_currpos (coll);
167  while (currpos > history) {
168  /* Removing entries is fast enough to be processed at once. */
169  xmms_playlist_remove_unlocked (playlist, plname, coll, 0, NULL);
170  currpos = xmms_playlist_coll_get_currpos (coll);
171  }

174 static void
175 xmms_playlist_update_partyshuffle (xmms_playlist_t *playlist,
176  const gchar *plname, xmmsv_t *coll)
177 {
178  gint history, upcoming, currpos, size;
179  xmmsv_t *src;
180 
181  XMMS_DBG ("PLAYLIST: Update partyshuffle.");
182 
183  if (!xmms_collection_get_int_attr (coll, "history", &history)) {
184  history = 0;
185  }
186 
187  if (!xmms_collection_get_int_attr (coll, "upcoming", &upcoming)) {
188  upcoming = XMMS_DEFAULT_PARTYSHUFFLE_UPCOMING;
189  }
190 
191  currpos = xmms_playlist_coll_get_currpos (coll);
192  while (currpos > history) {
193  /* Removing entries is fast enough to be processed at once. */
194  xmms_playlist_remove_unlocked (playlist, plname, coll, 0, NULL);
195  currpos = xmms_playlist_coll_get_currpos (coll);
196  }
197 
198  g_return_if_fail(xmmsv_list_get (xmmsv_coll_operands_get (coll), 0, &src));
199 
200  /* Since getting random media can be slow on huge medialibs, we refill only
201  * one entry at a time. This let other threads a chance to get the lock on
202  * the playlist object as soon as possible. */
203  size = xmms_playlist_coll_get_size (coll);
204  if (size < currpos + 1 + upcoming) {
205  xmms_medialib_entry_t randentry;
206  randentry = xmms_collection_get_random_media (playlist->colldag, src);
207  if (randentry > 0) {
208  xmms_playlist_add_entry_unlocked (playlist, plname, coll, randentry, NULL);
209  }
210  }
211 }

334 static gboolean
335 xmms_playlist_advance_do (xmms_playlist_t *playlist)
336 {
337  gint size, currpos;
338  gboolean ret = TRUE;
339  xmmsv_t *plcoll;
340  const gchar *jumplist;
341  xmms_error_t err;
342  xmms_playlist_t *buffer = playlist;
343  gint newpos;
344 
345  xmms_error_reset (&err);
346 
347  plcoll = xmms_playlist_get_coll (playlist, XMMS_ACTIVE_PLAYLIST, NULL);
348  if (plcoll == NULL) {
349  ret = FALSE;
350  } else if ((size = xmms_playlist_coll_get_size (plcoll)) == 0) {
351  if (xmmsv_coll_attribute_get_string (plcoll, "jumplist", &jumplist)) {
352  xmms_playlist_client_load (buffer, jumplist, &err);
353  if (xmms_error_isok (&err)) {
354  ret = xmms_playlist_advance_do (playlist);
355  } else {
356  ret = FALSE;
357  }
358  } else {
359  ret = FALSE;
360  }
361  } else if (!playlist->repeat_one) {
362  currpos = xmms_playlist_coll_get_currpos (plcoll);
363  currpos++;
364 
365  if (currpos == size && !playlist->repeat_all &&
366  xmmsv_coll_attribute_get_string (plcoll, "jumplist", &jumplist)) {
367 
368  xmms_collection_set_int_attr (plcoll, "position", -1);
369  XMMS_PLAYLIST_CURRPOS_MSG (-1, XMMS_ACTIVE_PLAYLIST);
370 
371  xmms_playlist_client_load (buffer, jumplist, &err);
372  if (xmms_error_isok (&err)) {
373  ret = xmms_playlist_advance_do (playlist);
374  } else {
375  ret = FALSE;
376  }
377  } else {
378  newpos = currpos%size;
379  xmms_collection_set_int_attr (plcoll, "position", newpos);
380  XMMS_PLAYLIST_CURRPOS_MSG (newpos, XMMS_ACTIVE_PLAYLIST);
381  ret = (currpos != size) || playlist->repeat_all;
382  }
383  }
384 
385  return ret;
386 }
