161 static int _compare_rows (const s4_resultrow_t **row1, const s4_resultrow_t **row2, s4_order_t *order)
162 {
163  int i, j, ret = 0;
164  for (i = 0; !ret && i < order->size; i++) {
165  s4_order_entry_t *entry = &order->columns[i];
166  if (entry->type == ORDER_TYPE_COLUMN) {
167  const s4_val_t *val1 = NULL;
168  const s4_val_t *val2 = NULL;
169 
170  for (j = 0; j < entry->size; j++) {
171  int col = entry->columns[j];
172  if ((*row1)->cols[col] != NULL) {
173  val1 = s4_result_get_val ((*row1)->cols[col]);
174  break;
175  }
176  }
177 
178  for (j = 0; j < entry->size; j++) {
179  int col = entry->columns[j];
180  if ((*row2)->cols[col] != NULL) {
181  val2 = s4_result_get_val ((*row2)->cols[col]);
182  break;
183  }
184  }
185 
186  if (val1 == NULL || val2 == NULL) {
187  if (val1 == NULL)
188  ret--;
189  if (val2 == NULL)
190  ret++;
191  } else {
192  ret = s4_val_cmp (val1, val2, entry->collation);
193  }
194 
195  if (entry->direction == S4_ORDER_DESCENDING)
196  ret = -ret;
197  } else {
198  ret = g_rand_int_range (entry->random, ORDER_RAND_MIN, ORDER_RAND_MAX);
199  }
200  }
201 
202  return (ret == 0) ? (row1 - row2) : ret;
203 }

315 /**
316  * Unreferences a resultrow. If the refcount hits 0 the row will be freed
317  * @param row The row to unrefperence
318  */
319 void s4_resultrow_unref (s4_resultrow_t *row)
320 {
321  if (row->refs <= 0) {
322  S4_ERROR ("s4_resultrow_unref: ref_count <= 0");
323  return;
324  }
325  row->refs--;
326 
327  if (row->refs <= 0) {
328  int i;
329  for (i = 0; i < row->col_count; i++) {
330  if (row->cols[i] != NULL) {
331  s4_result_t *prev,*res = row->cols[i];
332  while (res != NULL) {
333  prev = res;
334  res = (s4_result_t*)s4_result_next (res);
335  s4_result_free (prev);
336  }
337  }
338  }
339  free (row);
340  }
341 }
