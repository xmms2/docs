23 /**
24  * Helper function to build a list #xmmsv_t containing the
25  * strings from the input array.
26  *
27  * @param array An array of C strings. Must be NULL-terminated if num
28  * is -1.
29  * @param num The optional number of elements to read from the array. Set to
30  * -1 if the array is NULL-terminated.
31  * @return An #xmmsv_t containing the list of strings. Must be
32  * unreffed manually when done.
33  */
34 xmmsv_t *
35 xmmsv_make_stringlist (char *array[], int num)
36 {
37  xmmsv_t *list, *elem;
38  int i;
39 
40  list = xmmsv_new_list ();
41  if (array) {
42  for (i = 0; (num >= 0 && i < num) || array[i]; i++) {
43  elem = xmmsv_new_string (array[i]);
44  xmmsv_list_append (list, elem);
45  xmmsv_unref (elem);
46  }
47  }
48 
49  return list;
50 }

52 xmmsv_t *
53 xmmsv_build_dict_va (const char *firstkey, va_list ap)
54 {
55  const char *key;
56  xmmsv_t *val, *res;
57 
58  res = xmmsv_new_dict ();
59  if (!res)
60  return NULL;
61 
62  key = firstkey;
63  do {
64  val = va_arg (ap, xmmsv_t *);
65 
66  if (!xmmsv_dict_set (res, key, val)) {
67  xmmsv_unref (res);
68  res = NULL;
69  break;
70  }
71  xmmsv_unref (val);
72  key = va_arg (ap, const char *);
73  } while (key);
74 
75  return res;
76 }

153 /**
154  * Creates a metadata fetch specification.
155  *
156  * @param fields A list of fields to fetch, or NULL to fetch everything
157  * @param get A list of what to get ("id", "key", "value", "source")
158  * @param aggregate The aggregation function to use
159  * @param sourcepref A list of sources, first one has the highest priority
160  * @return A metadata fetch specification
161  */
162 xmmsv_t *xmmsv_build_metadata (xmmsv_t *fields, xmmsv_t *get, const char *aggregate, xmmsv_t *sourcepref)
163 {
164  xmmsv_t *res = xmmsv_new_dict ();
165  if (res == NULL)
166  return NULL;
167 
168  xmmsv_dict_set_string (res, "type", "metadata");
169 
170  if (fields != NULL) {
171  if (xmmsv_get_type (fields) == XMMSV_TYPE_STRING) {
172  xmmsv_t *list = xmmsv_new_list ();
173  xmmsv_list_append (list, fields);
174  xmmsv_unref (fields);
175  fields = list;
176  }
177  xmmsv_dict_set (res, "fields", fields);
178  xmmsv_unref (fields);
179  }
180  if (get != NULL) {
181  if (xmmsv_get_type (get) == XMMSV_TYPE_STRING) {
182  xmmsv_t *list = xmmsv_new_list ();
183  xmmsv_list_append (list, get);
184  xmmsv_unref (get);
185  get = list;
186  }
187  xmmsv_dict_set (res, "get", get);
188  xmmsv_unref (get);
189  }
190  if (sourcepref != NULL) {
191  xmmsv_dict_set (res, "source-preference", sourcepref);
192  xmmsv_unref (sourcepref);
193  }
194  if (aggregate != NULL) {
195  xmmsv_dict_set_string (res, "aggregate", aggregate);
196  }
197 
198  return res;
