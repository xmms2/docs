22 /**
23  * Send a client-to-client message.
24  * @param c The connection to the server.
25  * @param reply_policy Whether to expect no reply, a single reply or multiple
26  * replies for this message.
27  * @param payload The contents of the message.
28  */
29 xmmsc_result_t *
30 xmmsc_c2c_send (xmmsc_connection_t *c, int dest,
31  xmms_c2c_reply_policy_t reply_policy, xmmsv_t *payload)
32 {
33  uint32_t cookie;
34  xmmsc_result_t *res;
35 
36  x_check_conn (c, NULL);
37  x_api_error_if (!dest, "with 0 as dest.", NULL);
38  x_api_error_if (!payload, "with NULL payload.", NULL);
39 
40  cookie = xmmsc_send_cmd_cookie (c, XMMS_IPC_OBJECT_COURIER,
41  XMMS_IPC_COMMAND_COURIER_SEND_MESSAGE,
42  XMMSV_LIST_ENTRY_INT (dest),
43  XMMSV_LIST_ENTRY_INT (reply_policy),
44  XMMSV_LIST_ENTRY (xmmsv_ref (payload)),
45  XMMSV_LIST_END);
46 
47  /* Multi-replies behave like broadcasts */
48  if (reply_policy == XMMS_C2C_REPLY_POLICY_MULTI_REPLY) {
49  res = xmmsc_result_new (c, XMMSC_RESULT_CLASS_BROADCAST, cookie);
50  } else {
51  res = xmmsc_result_new (c, XMMSC_RESULT_CLASS_DEFAULT, cookie);
52  }
53 
54  /* If this message does not expect a reply, the client
55  * will get a reply from the server when the message has been sent,
56  * so the result is not a c2c result.
57  */
58  if (res && reply_policy != XMMS_C2C_REPLY_POLICY_NO_REPLY) {
59  xmmsc_result_c2c_set (res);
60  }
61 
62  return res;
63 }
