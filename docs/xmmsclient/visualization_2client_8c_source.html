66 {

69  x_check_conn (c, 0);
70 
71  c->visc++;
72  c->visv = realloc (c->visv, sizeof (xmmsc_visualization_t*) * c->visc);
73  if (!c->visv) {
74  x_oom ();
75  c->visc = 0;
76  }
77  if (c->visc > 0) {
78  int vv = c->visc-1;
79  if (!(c->visv[vv] = x_new0 (xmmsc_visualization_t, 1))) {
80  x_oom ();
81  } else {
82  c->visv[vv]->idx = vv;
83  c->visv[vv]->state = VIS_NEW;
84  res = xmmsc_send_msg_no_arg (c, XMMS_IPC_OBJECT_VISUALIZATION,
85  XMMS_IPC_COMMAND_VISUALIZATION_REGISTER);
86  if (res) {
87  xmmsc_result_visc_set (res, c->visv[vv]);
88  }
89  }
90  }
91  return res;
92 }

119 xmmsc_result_t *
120 xmmsc_visualization_start (xmmsc_connection_t *c, int vv)
121 {
122  xmmsc_result_t *res = NULL;
123  xmmsc_visualization_t *v;
124 
125  x_check_conn (c, 0);
126  v = get_dataset (c, vv);
127  x_api_error_if (!v, "with unregistered/unconnected visualization dataset", 0);
128 
129  switch (v->state) {
130  case VIS_WORKING:
131  case VIS_ERRORED:
132  break;
133  case VIS_NEW:
134 #ifdef HAVE_SEMTIMEDOP
135  /* first try unixshm */
136  v->type = VIS_UNIXSHM;
137  res = setup_shm_prepare (c, vv);
138  if (res) {
139  v->state = VIS_TRYING_UNIXSHM;
140  } else {
141  v->state = VIS_TO_TRY_UDP;
142  }
143  break;
144 #endif
145  case VIS_TO_TRY_UDP:
146  v->type = VIS_UDP;
147  res = setup_udp_prepare (c, vv);
148  if (res) {
149  v->state = VIS_TRYING_UDP;
150  } else {
151  v->state = VIS_ERRORED;
152  }
153  break;
154  default:
155  v->state = VIS_ERRORED;
156  x_api_warning ("out of sequence");
157  break;
158  }
159 
160  return res;
161 }

163 void
164 xmmsc_visualization_start_handle (xmmsc_connection_t *c, xmmsc_result_t *res)
165 {
166  xmmsc_visualization_t *v;
167  bool ret;
168 
169  v = xmmsc_result_visc_get (res);
170  if (!v) {
171  x_api_error_if (1, "non vis result?", );
172  }
173 
174  switch (v->state) {
175  case VIS_WORKING:
176  case VIS_ERRORED:
177  break;
178  case VIS_TRYING_UNIXSHM:
179  ret = setup_shm_handle (res);
180  if (!ret) {
181  c->error = strdup ("Server doesn't support or couldn't attach shared memory!");
182  v->state = VIS_TO_TRY_UDP;
183  } else {
184  v->state = VIS_WORKING;
185  }
186  break;
187  case VIS_TRYING_UDP:
188  ret = setup_udp_handle (res);
189  if (!ret) {
190  c->error = strdup ("Server doesn't support or couldn't setup UDP!");
191  v->state = VIS_ERRORED;
192  v->type = VIS_NONE;
193 
194  xmmsc_send_cmd (c, XMMS_IPC_OBJECT_VISUALIZATION,
195  XMMS_IPC_COMMAND_VISUALIZATION_SHUTDOWN,
196  XMMSV_LIST_ENTRY_INT (v->id),
197  XMMSV_LIST_END);
198  } else {
199  v->state = VIS_WORKING;
200  }
201  break;
202  default:
203  v->state = VIS_ERRORED;
204  x_api_warning ("out of sequence");
205  break;
206  }
207 }

316 int
317 check_drawtime (double ts, int drawtime)
318 {
319  struct timeval time;
320  double diff;
321 
322  if (drawtime <= 0)
323  return 0;
324 
325  gettimeofday (&time, NULL);
326  diff = ts - tv2ts (&time);
327  if (diff < 0) {
328  return 1;
329  }
330 
331  /* xmms_sleep_ms has a granularity of 10 ms.
332  To not sleep too long, we sleep 10 ms less than intended */
333  diff -= (drawtime + 10) * 0.001;
334  if (diff < 0) {
335  diff = 0;
336  }
337 
338  xmms_sleep_ms ((int) (diff * 1000));
339 
340  return 0;
341 }
