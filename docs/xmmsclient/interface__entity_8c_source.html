124  break;
125

348 /**
349  * Check whether the arguments passed to a method call are correct.
350  * @param mdesc The description of the method.
351  * @param positional_args A list of positional arguments to be checked.
352  * @param named_args A dict of named arguments to be checked.
353  *
354  * @return true if the arguments are valid and false otherwise.
355  * In either case, positional_args and named_args may have been modified.
356  */
357 static bool
358 xmmsc_sc_method_desc_check_args (xmmsc_sc_method_desc_t *mdesc,
359  xmmsv_t *positional_args, xmmsv_t *named_args)
360 {
361  const char *name;
362  xmmsv_t *arg, *value;
363  xmmsv_type_t expected_type;
364  xmmsv_list_iter_t *itsig, *itrecv;
365 
366  /* Check positional arguments.
367  * Traverse the list of positional arguments in the signature
368  * and the list of args we received, checking if types match.
369  */
370 
371  xmmsv_get_list_iter (mdesc->positional_args, &itsig);
372  xmmsv_get_list_iter (positional_args, &itrecv);
373 
374  while (xmmsv_list_iter_valid (itsig) && xmmsv_list_iter_valid (itrecv)) {
375  xmmsv_list_iter_entry (itsig, &arg);
376  xmmsv_list_iter_entry (itrecv, &value);
377 
378  expected_type = xmmsv_sc_argument_get_type (arg);
379  if (expected_type == XMMSV_TYPE_NONE ||
380  expected_type != xmmsv_get_type (value)) {
381 
382  /* Wrong type! */
383  return false;
384  }
385 
386  xmmsv_list_iter_next (itsig);
387  xmmsv_list_iter_next (itrecv);
388  }
389 
390  if (xmmsv_list_iter_valid (itrecv) && !mdesc->va_positional) {
391  /* Too many positional arguments! */
392  return false;
393  }
394 
395  while (xmmsv_list_iter_valid (itsig)) {
396  /* Less positional arguments received than declared.
397  * See if we can fill up the rest with the default values.
398  */
399 
400  xmmsv_list_iter_entry (itsig, &arg);
401  value = xmmsv_sc_argument_get_default_value (arg);
402 
403  if (!value) {
404  /* Missing positional arguments! */
405  return false;
406  }
407 
408  xmmsv_list_append (positional_args, value);
409  xmmsv_list_iter_next (itsig);
410  }
411 
412  /* Check named arguments.
413  * Traverse the list of named arguments in the signature
414  * and the dict of named arguments we received, checking names
415  * and types.
416  */
417 
418  xmmsv_get_list_iter (mdesc->named_args, &itsig);
419 
420  while (xmmsv_list_iter_valid (itsig)) {
421  xmmsv_list_iter_entry (itsig, &arg);
422  name = xmmsv_sc_argument_get_name (arg);
423 
424  if (!xmmsv_dict_get (named_args, name, &value)) {
425  /* Missing named argument.
426  * Try to use the default value or fail if
427  * there is none.
428  */
429 
430  value = xmmsv_sc_argument_get_default_value (arg);
431  if (!value) {
432  return false;
433  }
434 
435  xmmsv_dict_set (named_args, name, value);
436  }
437 
438  expected_type = xmmsv_sc_argument_get_type (arg);
439  if (expected_type == XMMSV_TYPE_NONE ||
440  expected_type != xmmsv_get_type (value)) {
441 
442  /* Wrong type! */
443  return false;
444  }
445 
446  xmmsv_list_iter_next (itsig);
447  }
448 
449  if (!mdesc->va_named &&
450  xmmsv_list_get_size (mdesc->named_args) <
451  xmmsv_dict_get_size (named_args)) {
452 
453  /* Too many named arguments! */
454  return false;
455  }
456 
457  return true;
458 }

609 /**
610  * Introspect into a namespace interface entity.
611  *
612  * The namespace description is a dictionary of the form:\n
613  * "name" : the namespace's name.\n
614  * "docstring" : the namespace's docstring.\n
615  * "constants" : a dictionary of constants.\n
616  * "namespaces" : a list of names of subnamespaces.\n
617  * "methods" : a list of descriptions of the methods in the namespace.\n
618  * "broadcasts" : a list of descriptions of the broadcasts in the namespace.\n
619  *
620  * @param nms The namespace interface entity.
621  * @return A dictionary describing the namespace.
622  */
623 xmmsv_t *
624 xmmsc_sc_interface_entity_namespace_introspect (xmmsc_sc_interface_entity_t *nms)
625 {
626  x_list_t *node;
627  xmmsc_sc_namespace_t *namespace;
628  xmmsc_sc_interface_entity_t *child;
629  xmmsv_t *methods, *subnamespaces, *broadcasts, *desc;
630 
631  methods = xmmsv_new_list ();
632  subnamespaces = xmmsv_new_list ();
633  broadcasts = xmmsv_new_list ();
634 
635  namespace = nms->entity.namespace;
636 
637  /* Traverse the list of children */
638  node = namespace->children;
639  while (node) {
640  child = node->data;
641 
642  /* Introspect each child and add it to the corresponding list */
643  switch (xmmsc_sc_interface_entity_get_type (child)) {
644  case XMMSC_SC_INTERFACE_ENTITY_TYPE_NAMESPACE:
645  xmmsv_list_append (subnamespaces, child->name);
646  break;
647

673  xmmsv_ref (namespace->constants)),
674  XMMSV_DICT_ENTRY ("namespaces", subnamespaces),

779 xmmsc_sc_namespace_lookup_child (xmmsc_sc_namespace_t *nms, xmmsv_t *name)
780 {

801 /**
802  * Resolve a path into an interface entity.
803  * @param root The namespace from which to start.
804  * @param path A list of strings for each entity along the path.
805  * @param parent A return location to store the parent namespace, or NULL.
806  * @return The interface entity, or NULL if not found.
807  */
808 xmmsc_sc_interface_entity_t *
809 xmmsc_sc_namespace_resolve_path (xmmsc_sc_namespace_t *root,
810  xmmsv_t *path,
811  xmmsc_sc_namespace_t **parent)
812 {
813  xmmsv_t *name;
814  xmmsv_list_iter_t *it;
815  xmmsc_sc_interface_entity_t *ifent = NULL;
816  xmmsc_sc_namespace_t *last_ns = NULL;
817 
818  x_return_null_if_fail (root);
819  x_return_null_if_fail (path);
820 
821  xmmsv_get_list_iter (path, &it);
822  while (xmmsv_list_iter_valid (it)) {
823  last_ns = root;
824 
825  xmmsv_list_iter_entry (it, &name);
826  xmmsv_list_iter_next (it);
827 
828  ifent = xmmsc_sc_namespace_lookup_child (root, name);
829  if (!ifent) {
830  return NULL;
831  }
832 
833  if (ifent->type == XMMSC_SC_INTERFACE_ENTITY_TYPE_NAMESPACE) {
834  root = ifent->entity.namespace;
835  } else {
836  break;
837  }
838  }
839 
840  if (xmmsv_list_iter_valid (it)) {
841  /* Path too large */
842  last_ns = NULL;
843  ifent = NULL;
844  }
845 
846  if (parent) {
847  *parent = last_ns;
848  }
849 
850  return ifent;
851 }
