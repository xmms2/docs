26 /**
27  * Call a method in another client.
28  * @param c The connection structure.
29  * @param dest The destination client's id.
30  * @param method A non-empty list of strings forming the path to the method.
31  * @param pargs A list of positional arguments. May be NULL.
32  * @param nargs A dict of named arguments. May be NULL.
33  */
34 xmmsc_result_t *
35 xmmsc_sc_call (xmmsc_connection_t *c, int dest, xmmsv_t *method,
36  xmmsv_t *pargs, xmmsv_t *nargs)
37 {
38  xmmsc_result_t *res;
39  xmmsv_t *msg, *call;
40 
41  x_check_conn (c, NULL);
42  x_api_error_if (!dest, "with 0 as destination client.", NULL);
43  x_api_error_if (!method, "with NULL method path.", NULL);
44  x_api_error_if (!xmmsv_list_get_size (method), "with empty method.", NULL);
45  x_api_error_if (pargs && xmmsv_get_type (pargs) != XMMSV_TYPE_LIST,
46  "with a non-list of positional arguments.", NULL);
47  x_api_error_if (nargs && xmmsv_get_type (nargs) != XMMSV_TYPE_DICT,
48  "with a non-dict of named arguments.", NULL);
49  x_api_error_if (!xmmsv_list_restrict_type (method, XMMSV_TYPE_STRING),
50  "with non-string in method path", NULL);
51 
52  /* Normalize NULL to empty values. */
53  pargs = pargs ? xmmsv_ref (pargs) : xmmsv_new_list ();
54  nargs = nargs ? xmmsv_ref (nargs) : xmmsv_new_dict ();
55 
56  /* Pack the call information in a dict.
57  * Let the dict steal a reference to pargs and nargs since either
58  * we created new values or we referenced the old ones above.
59  */
60  call = xmmsv_build_dict (XMMSV_DICT_ENTRY (XMMSC_SC_CALL_METHOD_KEY,
61  xmmsv_ref (method)),
62  XMMSV_DICT_ENTRY (XMMSC_SC_CALL_PARGS_KEY,
63  pargs),
64  XMMSV_DICT_ENTRY (XMMSC_SC_CALL_NARGS_KEY,
65  nargs),
66  XMMSV_DICT_END);
67 
68  /* And send that dict as the argument to the destination CALL command */
69  msg = xmmsv_build_dict (XMMSV_DICT_ENTRY_INT (XMMSC_SC_CMD_KEY,
70  XMMSC_SC_CALL),
71  XMMSV_DICT_ENTRY (XMMSC_SC_ARGS_KEY, call),
72  XMMSV_DICT_END);
73 
74  res = xmmsc_c2c_send (c, dest, XMMS_C2C_REPLY_POLICY_SINGLE_REPLY, msg);
75  xmmsv_unref (msg);
76 
77  return res;
78 }

117 /**
118  * Emit a broadcast.
119  * @param c The connection structure.
120  * @param broadcast A non-empty list of strings specifying the broadcast
121  * path.
122  * @param value The value to emit. Must not be NULL.
123  */
124 bool
125 xmmsc_sc_broadcast_emit (xmmsc_connection_t *c, xmmsv_t *broadcast,
126  xmmsv_t *value)
127 {
128  int id;
129  xmmsv_t *idlist, *v;
130  xmmsv_list_iter_t *it;
131  xmmsc_sc_interface_entity_t *entity;
132 
133  x_check_conn (c, false);
134  x_api_error_if (!broadcast, "with NULL broadcast path.", false);
135  x_api_error_if (!xmmsv_list_get_size (broadcast),
136  "with empty broadcast.", false);
137  x_api_error_if (!xmmsv_list_restrict_type (broadcast, XMMSV_TYPE_STRING),
138  "with non-string in broadcast path", false);
139  x_api_error_if (!value, "with NULL value", false);
140 
141  entity = xmmsc_sc_locate_interface_entity (c, broadcast);
142  x_api_error_if (!entity ||
143  xmmsc_sc_interface_entity_get_type (entity) !=
144  XMMSC_SC_INTERFACE_ENTITY_TYPE_BROADCAST,
145  "with a path that doesn't resolve to a broadcast",
146  false);
147 
148  /* Go through the list of msgids and reply to each sending 'value'.
149  * Don't expect counter-replies.
150  */
151  idlist = xmmsc_sc_interface_entity_broadcast_get_ids (entity);
152  xmmsv_get_list_iter (idlist, &it);
153  while (xmmsv_list_iter_valid (it)) {
154  if (xmmsv_list_iter_entry (it, &v) && xmmsv_get_int (v, &id)) {
155  xmmsc_result_t *res;
156 
157  res = xmmsc_c2c_reply (c, id, XMMS_C2C_REPLY_POLICY_NO_REPLY, value);
158  xmmsc_result_unref (res);
159  }
160 
161  xmmsv_list_iter_next (it);
162  }
163 
164  return true;
165 }
