50 static int
51 on_message_received (xmmsv_t *c2c_msg, void *userdata)
52 {
53  int cmd, msgid;
54  xmmsc_connection_t *c;
55  xmmsv_t *payload, *sc_args, *reply;
56 
57  c = (xmmsc_connection_t *) userdata;
58 
59  /* Parse message payload and message id so we can reply */
60  msgid = xmmsv_c2c_message_get_id (c2c_msg);
61  if (msgid <= 0) {
62  return true;
63  }
64 
65  payload = xmmsv_c2c_message_get_payload (c2c_msg);
66  if (!payload) {
67  reply = xmmsv_new_error ("failed to get payload");
68  goto send_reply;
69  }
70 
71  /* Parse the command and arguments out of the payload */
72  if (!xmmsv_dict_entry_get_int32 (payload, XMMSC_SC_CMD_KEY, &cmd)) {
73  reply = xmmsv_new_error ("failed to parse command");
74  goto send_reply;
75  }
76 
77  if (!xmmsv_dict_get (payload, XMMSC_SC_ARGS_KEY, &sc_args)) {
78  reply = xmmsv_new_error ("failed to parse arguments for command");
79  goto send_reply;
80  }
81 
82  /* And dispatch the command */
83  switch (cmd) {
84  case XMMSC_SC_CALL:
85  reply = command_call (c, sc_args);
86  break;
87 
88  case XMMSC_SC_BROADCAST_SUBSCRIBE:
89  reply = command_broadcast_subscribe (c, sc_args, msgid);
90  break;
91 
92  case XMMSC_SC_INTROSPECT:
93  reply = command_introspect (c, sc_args);
94  break;
95 
96  default:
97  reply = xmmsv_new_error ("unrecognized command");
98  break;
99  }
100 
101 send_reply:
102  if (reply) {
103  xmmsc_result_t *res;
104  res = xmmsc_c2c_reply (c,
105  msgid,
106  XMMS_C2C_REPLY_POLICY_NO_REPLY,
107  reply);
108 
109  xmmsc_result_unref (res);
110  xmmsv_unref (reply);
111  }
112 
113  return true;
114 }

116 /**
117  * Implements XMMSC_SC_CALL.
118  * This command expects as argument a dictionary mapping:
119  * XMMSC_SC_CALL_METHOD_KEY to a method path;
120  * XMMSC_SC_CALL_PARGS_KEY to a list of positional parameters;
121  * and XMMSC_SC_CALL_NARGS_KEY to a dictionary of named parameters.
122  */
123 static xmmsv_t *
124 command_call (xmmsc_connection_t *c, xmmsv_t *call)
125 {
126  xmmsc_sc_interface_entity_t *method;
127  xmmsv_t *method_path, *pargs, *nargs;
128 
129  if (!xmmsv_dict_get (call, XMMSC_SC_CALL_METHOD_KEY, &method_path)) {
130  return xmmsv_new_error ("failed to get method path");
131  }
132 
133  method = xmmsc_sc_locate_interface_entity (c, method_path);
134  if (!method) {
135  return xmmsv_new_error ("failed to resolve method path");
136  }
137 
138  if (xmmsc_sc_interface_entity_get_type (method) !=
139  XMMSC_SC_INTERFACE_ENTITY_TYPE_METHOD) {
140 
141  return xmmsv_new_error ("specified path is not a method");
142  }
143 
144  if (!xmmsv_dict_get (call, XMMSC_SC_CALL_PARGS_KEY, &pargs)) {
145  return xmmsv_new_error ("failed to get positional args");
146  }
147 
148  if (!xmmsv_dict_get (call, XMMSC_SC_CALL_NARGS_KEY, &nargs)) {
149  return xmmsv_new_error ("failed to get named args");
150  }
151 
152  return xmmsc_sc_interface_entity_method_call (method, pargs, nargs);
153 }

173 /**
174  * Implements XMMSC_SC_INTROSPECT.
175  * This command expects as its argument a dictionary mapping:
176  * XMMSC_SC_INTROSPECT_PATH_KEY to a path to an interface entity;
177  * XMMSC_SC_INTROSPECT_TYPE_KEY optionally, to the expected type of the
178  * interface entity;
179  * XMMSC_SC_INTROSPECT_KEYFILTER_KEY optionally, to a keyfilter.
180  *
181  * A keyfilter specifies a list of string keys whose values will be
182  * followed in sequence in the interface entity's description dictionary
183  * (and nested dictionaries) to yield the value to return.
184  *
185  * For instance, in a dict argument where:
186  * XMMSC_SC_INTROSPECT_PATH_KEY : ["some", "namespace", "path"]
187  * XMMSC_SC_INTROSPECT_TYPE_KEY : XMMSC_SC_INTERFACE_ENTITY_TYPE_NAMESPACE
188  * XMMSC_SC_INTROSPECT_KEYFILTER_KEY : ["constants", "some_constant_key"]
189  * the return value will be the value of the constant referenced by the
190  * key "some_constant_key" in the namespace.
191  *
192  * Likewise, when:
193  * XMMSC_SC_INTROSPECT_PATH_KEY : ["some", "method", "path"]
194  * XMMSC_SC_INTROSPECT_TYPE_KEY : XMMSC_SC_INTERFACE_ENTITY_METHOD
195  * XMMSC_SC_INTROSPECT_KEYFILTER_KEY : ["docstring"]
196  * the return value for the command will be the method's docstring,
197  * rather than its full description.
198  */
199 static xmmsv_t *
200 command_introspect (xmmsc_connection_t *c, xmmsv_t *args)
201 {
202  int type;
203  const char *key;
204  xmmsv_list_iter_t *it;
205  xmmsv_t *path, *ret, *keyfilter, *v;
206  xmmsc_sc_interface_entity_t *ifent;
207 
208  if (!xmmsv_dict_get (args, XMMSC_SC_INTROSPECT_PATH_KEY, &path)) {
209  return xmmsv_new_error ("failed to get path from arguments");
210  }
211 
212  ifent = xmmsc_sc_locate_interface_entity (c, path);
213  if (!ifent) {
214  return xmmsv_new_error ("failed to resolve path");
215  }
216 
217  if (xmmsv_dict_entry_get_int (args, XMMSC_SC_INTROSPECT_TYPE_KEY, &type) &&
218  xmmsc_sc_interface_entity_get_type (ifent) != type) {
219 
220  return xmmsv_new_error ("unexpected type");
221  }
222 
223  ret = introspect_interface_entity (ifent);
224 
225  /* Follow trail of keys specified in keyfilter */
226  if (xmmsv_dict_get (args, XMMSC_SC_INTROSPECT_KEYFILTER_KEY, &keyfilter)) {
227  xmmsv_get_list_iter (keyfilter, &it);
228 
229  while (xmmsv_list_iter_valid (it)) {
230  xmmsv_list_iter_entry_string (it, &key);
231  xmmsv_dict_get (ret, key, &v);
232 
233  /* Get a reference to the value and unref the
234  * parent dict since we won't need it anymore.
235  */
236  xmmsv_ref (v);
237  xmmsv_unref (ret);
238  ret = v;
239 
240  xmmsv_list_iter_next (it);
241  }
242  }
243 
244  return ret;
245 }

247 /**
248  * Introspect an interface entity regardless of type.
249  */
250 static xmmsv_t *
251 introspect_interface_entity (xmmsc_sc_interface_entity_t *ifent)
252 {
253  switch (xmmsc_sc_interface_entity_get_type (ifent)) {
254  case XMMSC_SC_INTERFACE_ENTITY_TYPE_METHOD:
255  return xmmsc_sc_interface_entity_method_introspect (ifent);
256 
257  case XMMSC_SC_INTERFACE_ENTITY_TYPE_NAMESPACE:
258  return xmmsc_sc_interface_entity_namespace_introspect (ifent);
259 
260  case XMMSC_SC_INTERFACE_ENTITY_TYPE_BROADCAST:
261  return xmmsc_sc_interface_entity_broadcast_introspect (ifent);
262  }
263  return NULL;
264 }
