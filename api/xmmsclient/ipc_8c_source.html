61 int
62 xmmsc_ipc_io_in_callback (xmmsc_ipc_t *ipc)
63 {
64  bool disco = false;
65 
66  x_return_val_if_fail (ipc, false);
67  x_return_val_if_fail (!ipc->disconnect, false);
68 
69  while (!disco) {
70  if (!ipc->read_msg)
71  ipc->read_msg = xmms_ipc_msg_alloc ();
72 
73  if (xmms_ipc_msg_read_transport (ipc->read_msg, ipc->transport, &disco)) {
74  xmms_ipc_msg_t *msg = ipc->read_msg;
75 
76  /* must unset read_msg here,
77  because exec_msg can cause reentrancy */
78  ipc->read_msg = NULL;
79 
80  xmmsc_ipc_exec_msg (ipc, msg);
81 
82  } else {
83 
84  break;
85  }
86  }
87 
88  if (disco)
89  xmmsc_ipc_disconnect (ipc);
90 
91  return !disco;
92 }

102 int
103 xmmsc_ipc_io_out_callback (xmmsc_ipc_t *ipc)
104 {
105  bool disco = false;
106 
107  x_return_val_if_fail (ipc, false);
108  x_return_val_if_fail (!ipc->disconnect, false);
109 
110  while (!x_queue_is_empty (ipc->out_msg)) {
111  xmms_ipc_msg_t *msg = x_queue_peek_head (ipc->out_msg);
112  if (xmms_ipc_msg_write_transport (msg, ipc->transport, &disco)) {
113  x_queue_pop_head (ipc->out_msg);
114  xmms_ipc_msg_destroy (msg);
115  } else {
116  break;
117  }
118  }
119 
120  if (disco) {
121  xmmsc_ipc_disconnect (ipc);
122  } else {
123  if (ipc->need_out_callback)
124  ipc->need_out_callback (xmmsc_ipc_io_out (ipc),
125  ipc->need_out_data);
126  }
127 
128  return !disco;
129 }

217 xmmsc_ipc_result_lookup (xmmsc_ipc_t *ipc, uint32_t cookie)
218 {
219  xmmsc_result_t *res = NULL;
220  x_list_t *n;
221 
222  x_return_val_if_fail (ipc, NULL);
223 
224  xmmsc_ipc_lock (ipc);
225 
226  for (n = ipc->results_list; n; n = x_list_next (n)) {
227  xmmsc_result_t *tmp = n->data;
228 
229  if (cookie == xmmsc_result_cookie_get (tmp)) {
230  res = tmp;
231  break;
232  }
233  }
234 
235  xmmsc_ipc_unlock (ipc);
236 
237  return res;

270 void
271 xmmsc_ipc_wait_for_event (xmmsc_ipc_t *ipc, unsigned int timeout)
272 {
273  fd_set rfdset;
274  fd_set wfdset;
275  struct timeval tmout;
276  xmms_socket_t fd;
277 
278  x_return_if_fail (ipc);
279  x_return_if_fail (!ipc->disconnect);
280 
281  tmout.tv_sec = timeout;
282  tmout.tv_usec = 0;
283 
284  fd = xmms_ipc_transport_fd_get (ipc->transport);
285 
286  FD_ZERO (&rfdset);
287  FD_SET (fd, &rfdset);
288 
289  FD_ZERO (&wfdset);
290  if (xmmsc_ipc_io_out (ipc)) {
291  FD_SET (fd, &wfdset);
292  }
293 
294  if (select (fd + 1, &rfdset, &wfdset, NULL, &tmout) == SOCKET_ERROR) {
295  return;
296  }
297 
298  if (FD_ISSET (fd, &rfdset)) {
299  if (!xmmsc_ipc_io_in_callback (ipc)) {
300  return;
301  }
302  }
303  if (FD_ISSET (fd, &wfdset)) {
304  xmmsc_ipc_io_out_callback (ipc);
305  }
306 }

325 xmmsc_ipc_destroy (xmmsc_ipc_t *ipc)
326 {
327  x_list_t *n;
328 
329  if (!ipc)
330  return;
331 
332  for (n = ipc->results_list; n; n = x_list_next (n)) {
333  xmmsc_result_t *tmp = n->data;
334  xmmsc_result_clear_weakrefs (tmp);
335  }
336 
337  x_list_free (ipc->results_list);
338  if (ipc->transport) {
339  xmms_ipc_transport_destroy (ipc->transport);
340  }
341 
342  if (ipc->out_msg) {
343  x_queue_free (ipc->out_msg);
344  }
345 
346  if (ipc->read_msg) {
347  xmms_ipc_msg_destroy (ipc->read_msg);
348  }
349 
350  if (ipc->error) {
351  free (ipc->error);
352  }
353 
354  if (ipc->disconnect_data && ipc->disconnect_data_free_func) {
355  xmmsc_user_data_free_func_t f = ipc->disconnect_data_free_func;
356  f (ipc->disconnect_data);
357  }
358 
359  if (ipc->need_out_data && ipc->need_out_data_free_func) {
360  xmmsc_user_data_free_func_t f = ipc->need_out_data_free_func;
361  f (ipc->need_out_data);
362  }
363 
364  free (ipc);
365 }
