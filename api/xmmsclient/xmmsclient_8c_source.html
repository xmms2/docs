50 /**
51  * @defgroup XMMSClient
52  * @brief This functions are used to connect a client software
53  * to the XMMS2 daemon.
54  *
55  * For proper integration with a client you need to hook the XMMSIPC
56  * to your clients mainloop. XMMS2 ships with a couple of default
57  * mainloop integrations but can easily be extended to fit your own
58  * application.
59  *
60  * There are three kinds of messages that will be involved in communication
61  * with the XMMS2 server.
62  * - Commands: Sent by the client to the server with arguments. Commands will
63  * generate a reply to the client.
64  * - Broadcasts: Sent by the server to the client if requested. Requesting a
65  * broadcast is done by calling one of the xmmsc_broadcast functions.
66  * - Signals: Like broadcasts but they are throttled, the client has to request
67  * the next signal when the callback is called. #xmmsc_result_restart is used to
68  * "restart" the next signal.
69  *
70  * Each client command will return a #xmmsc_result_t which holds the command id
71  * that will be used to map the result back to the right caller. The #xmmsc_result_t
72  * is used to get the result from the server.
73  *
74  * @{
75  */

136 /**
137  * Connects to the XMMS server.
138  * If ipcpath is NULL, it will try to open the default path.
139  *
140  * @param c The connection to the server. This must be initialized
141  * with #xmmsc_init first.
142  * @param ipcpath The IPC path, it's broken down like this: &lt;protocol&gt;://&lt;path&gt;[:&lt;port&gt;].
143  * If ipcpath is %NULL it will default to "unix:///tmp/xmms-ipc-<username>"
144  * - Protocol could be "tcp" or "unix"
145  * - Path is either the UNIX socket, or the ipnumber of the server.
146  * - Port is only used when the protocol tcp.
147  *
148  * @returns TRUE on success and FALSE if some problem
149  * occured. call xmmsc_get_last_error to find out.
150  *
151  * @sa xmmsc_get_last_error
152  */
153 
154 int
155 xmmsc_connect (xmmsc_connection_t *c, const char *ipcpath)
156 {
157  xmmsc_ipc_t *ipc;
158  xmmsc_result_t *result;
159  xmmsv_t *value;
160 
161  const char *buf;
162 
163  x_api_error_if (!c, "with a NULL connection", false);
164 
165  if (!ipcpath) {
166  if (!xmms_default_ipcpath_get (c->path, sizeof (c->path))) {
167  return false;
168  }
169  } else {
170  snprintf (c->path, sizeof (c->path), "%s", ipcpath);
171  }
172 
173  ipc = xmmsc_ipc_init ();
174 
175  if (!xmmsc_ipc_connect (ipc, c->path)) {
176  c->error = strdup ("xmms2d is not running.");
177  xmmsc_ipc_destroy (ipc);
178  return false;
179  }
180 
181  c->ipc = ipc;
182  result = xmmsc_send_hello (c);
183  xmmsc_result_wait (result);
184  value = xmmsc_result_get_value (result);
185  if (xmmsv_is_error (value)) {
186  xmmsv_get_error (value, &buf);
187  c->error = strdup (buf);
188  xmmsc_result_unref (result);
189  return false;
190  } else {
191  xmmsv_get_int64 (value, &c->id);
192  }
193  xmmsc_result_unref (result);
194  return true;
195 }
