505 int
506 _xmmsc_medialib_verify_url (const char *url)
507 {
508  int i;
509 
510  for (i = 0; url[i]; i++) {
511  if (!(GOODCHAR (url[i]) || url[i] == '+' || url[i] == '%' || url[i] == '?' || url[i] == '=' || url[i] == '&'))
512  return 0;
513  }
514  return 1;
515 }

517 char *
518 _xmmsc_medialib_encode_url_old (const char *url, int narg, const char **args)
519 {
520  static const char hex[16] = "0123456789abcdef";
521  int i = 0, j = 0, extra = 0;
522  char *res;
523 
524  x_api_error_if (!url, "with a NULL url", NULL);
525 
526  for (i = 0; i < narg; i++) {
527  extra += strlen (args[i]) + 2;
528  }
529 
530  res = malloc (strlen (url) * 3 + 1 + extra);
531  if (!res)
532  return NULL;
533 
534  for (i = 0; url[i]; i++) {
535  unsigned char chr = url[i];
536  if (GOODCHAR (chr)) {
537  res[j++] = chr;
538  } else if (chr == ' ') {
539  res[j++] = '+';
540  } else {
541  res[j++] = '%';
542  res[j++] = hex[((chr & 0xf0) >> 4)];
543  res[j++] = hex[(chr & 0x0f)];
544  }
545  }
546 
547  for (i = 0; i < narg; i++) {
548  int l;
549  l = strlen (args[i]);
550  res[j] = (i == 0) ? '?' : '&';
551  j++;
552  memcpy (&res[j], args[i], l);
553  j += l;
554  }
555 
556  res[j] = '\0';
557 
558  return res;
559 }
