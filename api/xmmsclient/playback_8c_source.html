93 /**
94  * Seek to a position given in milliseconds in the current playback.
95  *
96  * @param c The connection structure.
97  * @param milliseconds
98  * @param whence Specifies how the absolute position in milliseconds is
99  * determined. If whence is XMMS_PLAYBACK_SEEK_SET, @milliseconds is treated
100  * as an absolute value. If whence is XMMS_PLAYBACK_SEEK_CUR, the new
101  * position is computed by adding @milliseconds to the current position.
102  */
103

106  xmms_playback_seek_mode_t whence)
107 {
108  x_check_conn (c, NULL);
109 
110  return xmmsc_send_cmd (c, XMMS_IPC_OBJECT_PLAYBACK, XMMS_IPC_COMMAND_PLAYBACK_SEEK_MS,
111  XMMSV_LIST_ENTRY_INT (milliseconds),
112  XMMSV_LIST_ENTRY_INT (whence),
113  XMMSV_LIST_END);
