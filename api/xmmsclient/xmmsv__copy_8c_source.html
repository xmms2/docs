27 /**
28  * Return a new value object which is a deep copy of the input value
29  *
30  * @param val #xmmsv_t to copy.
31  * @return 1 the address to the new copy of the value.
32  */
33 xmmsv_t *
34 xmmsv_copy (xmmsv_t *val)
35 {
36  xmmsv_t *cur_val = NULL;
37  xmmsv_type_t type;
38  int64_t i;
39  const char *s;
40  float f;
41 
42  x_return_val_if_fail (val, 0);
43  type = xmmsv_get_type (val);
44  switch (type) {
45  case XMMSV_TYPE_DICT:
46  cur_val = duplicate_dict_value (val);
47  break;
48  case XMMSV_TYPE_LIST:
49  cur_val = duplicate_list_value (val);
50  break;
51  case XMMSV_TYPE_INT64:
52  xmmsv_get_int (val, &i);
53  cur_val = xmmsv_new_int (i);
54  break;
55  case XMMSV_TYPE_FLOAT:
56  xmmsv_get_float (val, &f);
57  cur_val = xmmsv_new_float (f);
58  break;
59  case XMMSV_TYPE_STRING:
60  xmmsv_get_string (val, &s);
61  cur_val = xmmsv_new_string (s);
62  break;
63  case XMMSV_TYPE_ERROR:
64  xmmsv_get_error (val, &s);
65  cur_val = xmmsv_new_error (s);
66  break;
67  case XMMSV_TYPE_COLL:
68  cur_val = duplicate_coll_value (val);
69  break;
70  case XMMSV_TYPE_BIN:
71  cur_val = xmmsv_new_bin (val->value.bin.data, val->value.bin.len);
72  break;
73  case XMMSV_TYPE_BITBUFFER:
74  cur_val = xmmsv_new_bitbuffer ();
75  xmmsv_bitbuffer_put_data (cur_val, val->value.bit.buf, val->value.bit.len / 8);
76  xmmsv_bitbuffer_goto (cur_val, xmmsv_bitbuffer_pos (val));
77  break;
78  default:
79  cur_val = xmmsv_new_none ();
80  break;
81  }
82  assert (cur_val);
83  return cur_val;
84 }

86 xmmsv_t *
87 duplicate_dict_value (xmmsv_t *val)
88 {
89  xmmsv_t *dup_val;
90  xmmsv_dict_iter_t *it;
91  const char *key;
92  xmmsv_t *v;
93  xmmsv_t *new_elem;
94 
95  x_return_val_if_fail (xmmsv_get_dict_iter (val, &it), NULL);
96  dup_val = xmmsv_new_dict ();
97  while (xmmsv_dict_iter_pair (it, &key, &v)) {
98  new_elem = xmmsv_copy (v);
99  xmmsv_dict_set (dup_val, key, new_elem);
100  xmmsv_unref (new_elem);
101  xmmsv_dict_iter_next (it);
102  }
103 
104  xmmsv_dict_iter_explicit_destroy (it);
105 
106  return dup_val;
107 }
