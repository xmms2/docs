21 /**
22  * Get number of bytes used for one sample length worth of music.
23  *
24  * That is, (size of one sample) * channels.
25  */
26 gint
27 xmms_sample_frame_size_get (const xmms_stream_type_t *st)
28 {
29  gint format, channels;
30  format = xmms_stream_type_get_int (st, XMMS_STREAM_TYPE_FMT_FORMAT);
31  channels = xmms_stream_type_get_int (st, XMMS_STREAM_TYPE_FMT_CHANNELS);
32  return xmms_sample_size_get (format) * channels;
33 }

64 /**
65  * Convert from bytes to samples for this format. This is exact.
66  * On error 0 is returned and error is set.
67  */
68 gint64
69 xmms_sample_bytes_to_samples (const xmms_stream_type_t *st, gint64 bytes,
70  xmms_error_t *error)
71 {
72  gint fs = xmms_sample_frame_size_get (st);
73  if (bytes % fs != 0) {
74  xmms_error_set (error, XMMS_ERROR_INVAL,
75  "xmms_sample_bytes_to_samples with non-integral number of samples!");
76  return 0;
77  }
78  return bytes / fs;
79 }
