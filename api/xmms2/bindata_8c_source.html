110 gchar *
111 xmms_bindata_calculate_md5 (const guchar *data, gsize size, gchar ret[33])
112 {
113  md5_state_t state;
114  md5_byte_t digest[16];
115  int di;
116  static gchar hex[] = {
117  '0', '1', '2', '3', '4', '5', '6', '7',
118  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f'
119  };
120 
121  md5_init (&state);
122  md5_append (&state, (const md5_byte_t *)data, size);
123  md5_finish (&state, digest);
124 
125  for (di = 0; di < 16; ++di) {
126  ret[di * 2] = hex[digest[di] >> 4];
127  ret[di * 2 + 1] = hex[digest[di] & 0x0f];
128  }
129  ret[32] = 0;
130  return ret;
131 }

147 static gboolean
148 _xmms_bindata_add (xmms_bindata_t *bindata, const guchar *data, gsize len, gchar hash[33], xmms_error_t *err)
149 {
150  const guchar *ptr;
151  gsize left;
152  gchar *path;
153  FILE *fp;
154 
155  xmms_bindata_calculate_md5 (data, len, hash);
156 
157  path = xmms_bindata_build_path (bindata, hash);
158 
159  if (g_file_test (path, G_FILE_TEST_IS_REGULAR)) {
160  XMMS_DBG ("file %s is already in bindata dir", hash);
161  g_free (path);
162  return TRUE;
163  }
164 
165  XMMS_DBG ("Creating %s", path);
166  fp = fopen (path, "wb");
167  if (!fp) {
168  xmms_log_error ("Couldn't create %s", path);
169  xmms_error_set (err, XMMS_ERROR_GENERIC, "Couldn't create file on server!");
170  g_free (path);
171  return FALSE;
172  }
173 
174  /* write the data to the file */
175  ptr = data;
176  left = len;
177 
178  while (left > 0) {
179  size_t w;
180 
181  w = fwrite (ptr, 1, left, fp);
182  if (!w && ferror (fp)) {
183  fclose (fp);
184  unlink (path);
185 
186  xmms_log_error ("Couldn't write data");
187  xmms_error_set (err, XMMS_ERROR_GENERIC,
188  "Couldn't write data!");
189  g_free (path);
190  return FALSE;
191  }
192 
193  left -= w;
194  ptr += w;
195  }
196 
197  fclose (fp);
198  g_free (path);
199 
200  return TRUE;
201 }

232

234  while (!feof (fp)) {
235  gchar buf[1024];
236  gint l;
237 
238  l = fread (buf, 1, 1024, fp);
239  if (ferror (fp)) {
240  g_string_free (str, TRUE);
241  xmms_log_error ("Error reading bindata '%s'", hash);
242  xmms_error_set (err, XMMS_ERROR_GENERIC, "Error reading file");
243  fclose (fp);
244  return NULL;
245  }
246  g_string_append_len (str, buf, l);
247  }
248 
249  fclose (fp);
250 
251  res = xmmsv_new_bin ((unsigned char *)str->str, str->len);
252 
253  g_string_free (str, TRUE);
254

437 static void
438 md5_process (md5_state_t *pms, const md5_byte_t *data /*[64]*/)
439 {
440  md5_word_t
441  a = pms->abcd[0], b = pms->abcd[1],
442  c = pms->abcd[2], d = pms->abcd[3];
443  md5_word_t t;
444 #if BYTE_ORDER > 0
445  /* Define storage only for big-endian CPUs. */
446  md5_word_t X[16];
447 #else
448  /* Define storage for little-endian or both types of CPUs. */
449  md5_word_t xbuf[16];
450  const md5_word_t *X;
451 #endif
452 
453  {
454 #if BYTE_ORDER == 0
455  /*
456  * Determine dynamically whether this is a big-endian or
457  * little-endian machine, since we can use a more efficient
458  * algorithm on the latter.
459  */
460  static const int w = 1;
461 
462  if (*((const md5_byte_t *)&w)) /* dynamic little-endian */
463 #endif
464 #if BYTE_ORDER <= 0 /* little-endian */
465  {
466  /*
467  * On little-endian machines, we can process properly aligned
468  * data without copying it.
469  */
470  if (!((data - (const md5_byte_t *)0) & 3)) {
471  /* data are properly aligned */
472  X = (const md5_word_t *)data;
473  } else {
474  /* not aligned */
475  memcpy (xbuf, data, 64);
476  X = xbuf;
477  }
478  }
479 #endif
480 #if BYTE_ORDER == 0
481  else /* dynamic big-endian */
482 #endif
483 #if BYTE_ORDER >= 0 /* big-endian */
484  {
485  /*
486  * On big-endian machines, we must arrange the bytes in the
487  * right order.
488  */
489  const md5_byte_t *xp = data;
490  int i;
491 
492 # if BYTE_ORDER == 0
493  X = xbuf;/* (dynamic only) */
494 # else
495 # define xbuf X /* (static only) */
496 # endif
497  for (i = 0; i < 16; ++i, xp += 4)
498  xbuf[i] = xp[0] + (xp[1] << 8) + (xp[2] << 16) + (xp[3] << 24);
499  }
500 #endif
501  }
502 
503 #define ROTATE_LEFT(x, n) (((x) << (n)) | ((x) >> (32 - (n))))
504 
505  /* Round 1. */
506  /* Let [abcd k s i] denote the operation
507  a = b + ((a + F(b,c,d) + X[k] + T[i]) <<< s). */
508 #define F(x, y, z) (((x) & (y)) | (~(x) & (z)))
509 #define SET(a, b, c, d, k, s, Ti)\
510  t = a + F (b,c,d) + X[k] + Ti;\
511  a = ROTATE_LEFT (t, s) + b
512  /* Do the following 16 operations. */
513  SET (a, b, c, d, 0, 7, T1);
514  SET (d, a, b, c, 1, 12, T2);
515  SET (c, d, a, b, 2, 17, T3);
516  SET (b, c, d, a, 3, 22, T4);
517  SET (a, b, c, d, 4, 7, T5);
518  SET (d, a, b, c, 5, 12, T6);
519  SET (c, d, a, b, 6, 17, T7);
520  SET (b, c, d, a, 7, 22, T8);
521  SET (a, b, c, d, 8, 7, T9);
522  SET (d, a, b, c, 9, 12, T10);
523  SET (c, d, a, b, 10, 17, T11);
524  SET (b, c, d, a, 11, 22, T12);
525  SET (a, b, c, d, 12, 7, T13);
526  SET (d, a, b, c, 13, 12, T14);
527  SET (c, d, a, b, 14, 17, T15);
528  SET (b, c, d, a, 15, 22, T16);
529 #undef SET
530 
531  /* Round 2. */
532  /* Let [abcd k s i] denote the operation
533  a = b + ((a + G(b,c,d) + X[k] + T[i]) <<< s). */
534 #define G(x, y, z) (((x) & (z)) | ((y) & ~(z)))
535 #define SET(a, b, c, d, k, s, Ti)\
536  t = a + G (b,c,d) + X[k] + Ti;\
537  a = ROTATE_LEFT (t, s) + b
538  /* Do the following 16 operations. */
539  SET (a, b, c, d, 1, 5, T17);
540  SET (d, a, b, c, 6, 9, T18);
541  SET (c, d, a, b, 11, 14, T19);
542  SET (b, c, d, a, 0, 20, T20);
543  SET (a, b, c, d, 5, 5, T21);
544  SET (d, a, b, c, 10, 9, T22);
545  SET (c, d, a, b, 15, 14, T23);
546  SET (b, c, d, a, 4, 20, T24);
547  SET (a, b, c, d, 9, 5, T25);

630 {
631  const md5_byte_t *p = data;
632  int left = nbytes;
633  int offset = (pms->count[0] >> 3) & 63;
634  md5_word_t nbits = (md5_word_t)(nbytes << 3);
635 
636  if (nbytes <= 0)
637  return;
638 
639  /* Update the message length. */
640  pms->count[1] += nbytes >> 29;
641  pms->count[0] += nbits;
642  if (pms->count[0] < nbits)
643  pms->count[1]++;
644 
645  /* Process an initial partial block. */
646  if (offset) {
647  int copy = (offset + nbytes > 64 ? 64 - offset : nbytes);
648 
649  memcpy (pms->buf + offset, p, copy);
650  if (offset + copy < 64)
651  return;
652  p += copy;
653  left -= copy;
654  md5_process (pms, pms->buf);
655  }
656 
657  /* Process full blocks. */
658  for (; left >= 64; p += 64, left -= 64)
659  md5_process (pms, p);
660 
661  /* Process a final partial block. */
662  if (left)
663  memcpy (pms->buf, p, left);
664 }
