94 
95  /* perhaps rework this later */
96  if (len > XMMSC_VISUALIZATION_WINDOW_SIZE * chan * sizeof (short)) {
97  len = XMMSC_VISUALIZATION_WINDOW_SIZE * chan * sizeof (short);

99 
100  read = xmms_xform_read (xform, buf, len, error);
101  if (read > 0) {
