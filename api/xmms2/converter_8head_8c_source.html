76 xmms_sample_converter_t *
77 xmms_sample_converter_init (xmms_stream_type_t *from, xmms_stream_type_t *to)
78 {
79  xmms_sample_converter_t *conv = xmms_object_new (xmms_sample_converter_t, xmms_sample_converter_destroy);
80  gint fformat, fsamplerate, fchannels;
81  gint tformat, tsamplerate, tchannels;
82 
83  fformat = xmms_stream_type_get_int (from, XMMS_STREAM_TYPE_FMT_FORMAT);
84  fsamplerate = xmms_stream_type_get_int (from, XMMS_STREAM_TYPE_FMT_SAMPLERATE);
85  fchannels = xmms_stream_type_get_int (from, XMMS_STREAM_TYPE_FMT_CHANNELS);
86  tformat = xmms_stream_type_get_int (to, XMMS_STREAM_TYPE_FMT_FORMAT);
87  tsamplerate = xmms_stream_type_get_int (to, XMMS_STREAM_TYPE_FMT_SAMPLERATE);
88  tchannels = xmms_stream_type_get_int (to, XMMS_STREAM_TYPE_FMT_CHANNELS);
89 
90  g_return_val_if_fail (tformat != -1, NULL);
91  g_return_val_if_fail (tchannels != -1, NULL);
92  g_return_val_if_fail (tsamplerate != -1, NULL);
93 
94  conv->from = from;
95  conv->to = to;
96 
97  conv->resample = fsamplerate != tsamplerate;
98 
99  conv->func = xmms_sample_conv_get (fchannels, fformat,
100  tchannels, tformat,
101  conv->resample);
102 
103  if (!conv->func) {
104  xmms_object_unref (conv);
105  xmms_log_error ("Unable to convert from %s/%d/%d to %s/%d/%d.",
106  xmms_sample_name_get (fformat), fsamplerate, fchannels,
107  xmms_sample_name_get (tformat), tsamplerate, tchannels);
108  return NULL;
109  }
110 
111  if (conv->resample)
112  recalculate_resampler (conv, fsamplerate, tsamplerate);
113 
114  return conv;
115 }

174 recalculate_resampler (xmms_sample_converter_t *conv, guint from, guint to)
175 {
176  guint a,b;
177 
178  /* calculate ratio */
179  if (from > to){
180  a = from;
181  b = to;
182  } else {
183  b = to;
184  a = from;
185  }
186 
187  while (b != 0) { /* good 'ol euclid is helpful as usual */
188  guint t = a % b;
189  a = b;
190  b = t;
191  }
192 
193  XMMS_DBG ("Resampling ratio: %d:%d",
194  from / a, to / a);
195 
196  conv->interpolator_ratio = to/a;
197  conv->decimator_ratio = from/a;

212  * do the actual converstion between two audio formats.

214 void
215 xmms_sample_convert (xmms_sample_converter_t *conv, xmms_sample_t *in, guint len, xmms_sample_t **out, guint *outlen)
216 {
217  int inusiz, outusiz;
218  int olen;
219  guint res;
220 
221  inusiz = xmms_sample_frame_size_get (conv->from);
222 
223  g_return_if_fail (len % inusiz == 0);
224 
225  if (conv->same) {
226  *outlen = len;
227  *out = in;
228  return;
229  }
230 
231  len /= inusiz;
232 
233  outusiz = xmms_sample_frame_size_get (conv->to);
234 
235  if (conv->resample) {
236  olen = (len * conv->interpolator_ratio / conv->decimator_ratio) * outusiz + outusiz;
237  } else {
238  olen = len * outusiz;
239  }
240  if (olen > conv->bufsiz) {
241  void *t;
242  t = g_realloc (conv->buf, olen);
243  g_assert (t); /* XXX */
244  conv->buf = t;
245  conv->bufsiz = olen;
246  }
247 
248  res = conv->func (conv, in, len, conv->buf);
249 
250  *outlen = res * outusiz;
251  *out = conv->buf;
252 
253 }

256 xmms_sample_convert_scale (xmms_sample_converter_t *conv, gint64 samples)

258  /* this isn't 100% accurate, we should take care
