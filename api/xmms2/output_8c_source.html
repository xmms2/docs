221 static void
222 update_playtime (xmms_output_t *output, int advance)
223 {
224  guint buffersize = 0;
225 
226  g_mutex_lock (&output->playtime_mutex);
227  output->played += advance;
228  g_mutex_unlock (&output->playtime_mutex);
229 
230  buffersize = xmms_output_plugin_method_latency_get (output->plugin, output);
231 
232  if (output->played < buffersize) {
233  buffersize = output->played;
234  }
235 
236  g_mutex_lock (&output->playtime_mutex);
237 
238  if (output->format) {
239  guint ms = xmms_sample_bytes_to_ms (output->format,
240  output->played - buffersize);
241  if ((ms / 100) != (output->played_time / 100)) {
242  xmms_object_emit (XMMS_OBJECT (output),
243  XMMS_IPC_SIGNAL_PLAYBACK_PLAYTIME,
244  xmmsv_new_int (ms));
245  }
246  output->played_time = ms;
247 
248  }
249 
250  g_mutex_unlock (&output->playtime_mutex);
251

282 static gboolean
283 song_changed (void *data)
284 {
285  /* executes in the output thread; NOT the filler thread */
286  xmms_output_song_changed_arg_t *arg = (xmms_output_song_changed_arg_t *)data;
287  xmms_medialib_entry_t entry;
288  xmms_stream_type_t *type;
289 
290  entry = xmms_xform_entry_get (arg->chain);
291 
292  XMMS_DBG ("Running hotspot! Song changed!! %d", entry);
293 
294  arg->output->played = 0;
295  arg->output->current_entry = entry;
296 
297  type = xmms_xform_outtype_get (arg->chain);
298 
299  if (!xmms_output_format_set (arg->output, type)) {
300  gint fmt, rate, chn;
301 
302  fmt = xmms_stream_type_get_int (type, XMMS_STREAM_TYPE_FMT_FORMAT);
303  rate = xmms_stream_type_get_int (type, XMMS_STREAM_TYPE_FMT_SAMPLERATE);
304  chn = xmms_stream_type_get_int (type, XMMS_STREAM_TYPE_FMT_CHANNELS);
305 
306  XMMS_DBG ("Couldn't set format %s/%d/%d, stopping filler..",
307  xmms_sample_name_get (fmt), rate, chn);
308 
309  xmms_output_filler_state_nolock (arg->output, FILLER_STOP);
310  xmms_ringbuf_set_eos (arg->output->filler_buffer, TRUE);
311  return FALSE;
312  }
313 
314  if (arg->flush)
315  xmms_output_flush (arg->output);
316 
317  xmms_object_emit (XMMS_OBJECT (arg->output),
318  XMMS_IPC_SIGNAL_PLAYBACK_CURRENT_ID,
319  xmmsv_new_int (entry));
320 
321  return TRUE;
322 }

368 static void *
369 xmms_output_filler (void *arg)
370 {
371  xmms_output_t *output = (xmms_output_t *)arg;
372  xmms_xform_t *chain = NULL;
373  gboolean last_was_kill = FALSE;
374  char buf[4096];
375  xmms_error_t err;
376  gint ret;
377 
378  xmms_error_reset (&err);
379 
380  g_mutex_lock (&output->filler_mutex);
381  while (output->filler_state != FILLER_QUIT) {
382  if (output->filler_state == FILLER_STOP) {
383  if (chain) {
384  xmms_object_unref (chain);
385  chain = NULL;
386  }
387  xmms_ringbuf_set_eos (output->filler_buffer, TRUE);
388  g_cond_wait (&output->filler_state_cond, &output->filler_mutex);
389  last_was_kill = FALSE;
390  continue;
391  }
392  if (output->filler_state == FILLER_KILL) {
393  if (chain) {
394  xmms_object_unref (chain);
395  chain = NULL;
396  output->filler_state = FILLER_RUN;
397  last_was_kill = TRUE;
398  } else {
399  output->filler_state = FILLER_STOP;
400  }
401  continue;
402  }
403  if (output->filler_state == FILLER_SEEK) {
404  if (!chain) {
405  XMMS_DBG ("Seek without chain, ignoring..");
406  output->filler_state = FILLER_STOP;
407  continue;
408  }
409 
410  ret = xmms_xform_this_seek (chain, output->filler_seek, XMMS_XFORM_SEEK_SET, &err);
411  if (ret == -1) {
412  XMMS_DBG ("Seeking failed: %s", xmms_error_message_get (&err));
413  } else {
414  XMMS_DBG ("Seek ok! %d", ret);
415 
416  output->filler_skip = output->filler_seek - ret;
417  if (output->filler_skip < 0) {
418  XMMS_DBG ("Seeked %d samples too far! Updating position...",
419  -output->filler_skip);
420 
421  output->filler_skip = 0;
422  output->filler_seek = ret;
423  }
424 
425  xmms_ringbuf_clear (output->filler_buffer);
426  xmms_ringbuf_hotspot_set (output->filler_buffer, seek_done, NULL, output);
427  }
428  output->filler_state = FILLER_RUN;
429  }
430 
431  if (!chain) {
432  xmms_medialib_entry_t entry;
433  xmms_output_song_changed_arg_t *hsarg;
434 
435  g_mutex_unlock (&output->filler_mutex);
436 
437  entry = xmms_playlist_current_entry (output->playlist);
438  if (!entry) {
439  XMMS_DBG ("No entry from playlist!");
440  output->filler_state = FILLER_STOP;
441  g_mutex_lock (&output->filler_mutex);
442  continue;
443  }
444 
445  chain = xmms_xform_chain_setup (output->medialib, entry, output->format_list, FALSE);
446  if (!chain) {
447  xmms_medialib_session_t *session;
448 
449  do {
450  session = xmms_medialib_session_begin (output->medialib);
451  if (xmms_medialib_entry_property_get_int (session, entry, XMMS_MEDIALIB_ENTRY_PROPERTY_STATUS) == XMMS_MEDIALIB_ENTRY_STATUS_NEW) {
452  xmms_medialib_entry_remove (session, entry);
453  } else {
454  xmms_medialib_entry_status_set (session, entry, XMMS_MEDIALIB_ENTRY_STATUS_NOT_AVAILABLE);
455  }
456  } while (!xmms_medialib_session_commit (session));
457 
458  if (!xmms_playlist_advance (output->playlist)) {
459  XMMS_DBG ("End of playlist");
460  output->filler_state = FILLER_STOP;
461  }
462  g_mutex_lock (&output->filler_mutex);
463  continue;
464  }
465 
466  hsarg = g_new0 (xmms_output_song_changed_arg_t, 1);
467  hsarg->output = output;
468  hsarg->chain = chain;
469  hsarg->flush = last_was_kill;
470  xmms_object_ref (chain);
471 
472  last_was_kill = FALSE;
473 
474  g_mutex_lock (&output->filler_mutex);
475  xmms_ringbuf_hotspot_set (output->filler_buffer, song_changed, song_changed_arg_free, hsarg);
476  }
477 
478  xmms_ringbuf_wait_free (output->filler_buffer, sizeof (buf), &output->filler_mutex);
479 
480  if (output->filler_state != FILLER_RUN) {
481  XMMS_DBG ("State changed while waiting...");
482  continue;
483  }
484  g_mutex_unlock (&output->filler_mutex);
485 
486  ret = xmms_xform_this_read (chain, buf, sizeof (buf), &err);
487 
488  g_mutex_lock (&output->filler_mutex);
489 
490  if (ret > 0) {
491  gint skip = MIN (ret, output->toskip);
492 
493  output->toskip -= skip;
494  if (ret > skip) {
495  xmms_ringbuf_write_wait (output->filler_buffer,
496  buf + skip,
497  ret - skip,
498  &output->filler_mutex);
499  }
500  } else {
501  if (ret == -1) {
502  /* print error */
503  xmms_error_reset (&err);
504  }
505  xmms_object_unref (chain);
506  chain = NULL;
507  if (!xmms_playlist_advance (output->playlist)) {
508  XMMS_DBG ("End of playlist");
509  output->filler_state = FILLER_STOP;
510  }
511  }
512 
513  }
514 
515  if (chain)
516  xmms_object_unref (chain);
517 
518  g_mutex_unlock (&output->filler_mutex);
519 
520  return NULL;
521 }

523 gint
524 xmms_output_read (xmms_output_t *output, char *buffer, gint len)
525 {
526  gint ret;
527  xmms_error_t err;
528 
529  xmms_error_reset (&err);
530 
531  g_return_val_if_fail (output, -1);
532  g_return_val_if_fail (buffer, -1);
533 
534  g_mutex_lock (&output->filler_mutex);
535  xmms_ringbuf_wait_used (output->filler_buffer, len, &output->filler_mutex);
536  ret = xmms_ringbuf_read (output->filler_buffer, buffer, len);
537  if (ret == 0 && xmms_ringbuf_iseos (output->filler_buffer)) {
538  xmms_output_status_set (output, XMMS_PLAYBACK_STATUS_STOP);
539  g_mutex_unlock (&output->filler_mutex);
540  return -1;
541  }
542  g_mutex_unlock (&output->filler_mutex);
543 
544  update_playtime (output, ret);
545 
546  if (ret < len) {
547  XMMS_DBG ("Underrun %d of %d (%d)", ret, len, xmms_sample_frame_size_get (output->format));
548 
549  if ((ret % xmms_sample_frame_size_get (output->format)) != 0) {
550  xmms_log_error ("***********************************");
551  xmms_log_error ("* Read non-multiple of sample size,");
552  xmms_log_error ("* you probably hear noise now :)");
553  xmms_log_error ("***********************************");
554  }
555  output->buffer_underruns++;
556  }
557 
558  output->bytes_written += ret;
559 
560  return ret;
561 }
