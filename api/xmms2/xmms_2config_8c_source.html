362 /**
363  * @internal Get the current parser state for the given element name
364  * @param[in] name Element name to match to a state
365  * @return Parser state matching element name
366  */
367 static xmms_configparser_state_t
368 get_current_state (const gchar *name)
369 {
370  static struct {
371  const gchar *name;
372  xmms_configparser_state_t state;
373  } *ptr, lookup[] = {
374  {"xmms", XMMS_CONFIG_STATE_START},
375  {"section", XMMS_CONFIG_STATE_SECTION},
376  {"property", XMMS_CONFIG_STATE_PROPERTY},
377  {NULL, XMMS_CONFIG_STATE_INVALID}
378  };
379 
380  for (ptr = lookup; ptr && ptr->name; ptr++) {
381  if (!strcmp (ptr->name, name)) {
382  return ptr->state;
383  }
384  }
385 
386  return XMMS_CONFIG_STATE_INVALID;
387 }

390  * @internal Look for the value associated with an attribute name, given lists
391  * of attribute names and attribute values.
392  * @param[in] names List of attribute names
393  * @param[in] values List of attribute values matching up to names
394  * @param[in] needle Attribute name to look for
395  * @return The attribute value, or NULL if not found
396  */
397 static const gchar *
398 lookup_attribute (const gchar **names, const gchar **values,
399  const gchar *needle)
400 {
401  const gchar **n, **v;
402 
403  for (n = names, v = values; *n && *v; n++, v++) {

407  }

412 /**
413  * @internal Parse start tag in config file. This function is called whenever
414  * a start tag is encountered by the GMarkupParser from #xmms_config_init
415  * @param ctx The parser context.
416  * @param name The name of the element encountered
417  * @param attr_name List of attribute names in tag
418  * @param attr_data List of attribute data in tag
419  * @param userdata User data - In this case, the global config
420  * @param error GError to be filled in if an error is encountered
421  */
422 static void
423 xmms_config_parse_start (GMarkupParseContext *ctx,
424  const gchar *name,
425  const gchar **attr_name,
426  const gchar **attr_data,
427  gpointer userdata,
428  GError **error)
429 {
430  xmms_config_t *config = userdata;
431  xmms_configparser_state_t state;
432  const gchar *attr;
433 
434  state = get_current_state (name);
435  g_queue_push_head (config->states, GINT_TO_POINTER (state));
436 
437  switch (state) {
438  case XMMS_CONFIG_STATE_INVALID:
439  *error = g_error_new (G_MARKUP_ERROR,
440  G_MARKUP_ERROR_UNKNOWN_ELEMENT,
441  "Unknown element '%s'", name);
442  return;
443  case XMMS_CONFIG_STATE_START:
444  /* check config version here */
445  attr = lookup_attribute (attr_name, attr_data, "version");
446  if (attr) {
447  if (strcmp (attr, "0.02") == 0) {
448  config->version = 2;
449  } else {
450  config->version = atoi (attr);
451  }
452  }
453  return;
454  default:
455  break;
456  }
457 
458  attr = lookup_attribute (attr_name, attr_data, "name");
459  if (!attr) {
460  *error = g_error_new (G_MARKUP_ERROR,
461  G_MARKUP_ERROR_INVALID_CONTENT,
462  "Attribute 'name' missing");
463  return;
464  }
465 
466  switch (state) {
467  case XMMS_CONFIG_STATE_SECTION:
468  g_queue_push_head (config->sections, g_strdup (attr));
469 
470  break;
471  case XMMS_CONFIG_STATE_PROPERTY:
472  g_free (config->value_name);
473  config->value_name = g_strdup (attr);
474 
475  break;
476  default:
477  break;
478  }
479 }

515 /**
516  * @internal Parse text in config file. This function is called whenever
517  * text (anything between start and end tags) is encountered by the
518  * GMarkupParser from #xmms_config_init
519  * @param ctx The parser context.
520  * @param text The text
521  * @param text_len Length of the text
522  * @param userdata User data - In this case, the global config
523  * @param error GError to be filled in if an error is encountered
524  */
525 static void
526 xmms_config_parse_text (GMarkupParseContext *ctx,
527  const gchar *text,
528  gsize text_len,
529  gpointer userdata,
530  GError **error)
531 {
532  xmms_config_t *config = userdata;
533  xmms_configparser_state_t state;
534  xmms_config_property_t *prop;
535  GList *l;
536  gchar key[256] = "";
537  gsize siz = sizeof (key);
538 
539  state = GPOINTER_TO_INT (g_queue_peek_head (config->states));
540 
541  if (state != XMMS_CONFIG_STATE_PROPERTY)
542  return;
543 
544  /* assemble the config key, based on the traversed sections */
545  for (l = config->sections->tail; l; l = l->prev) {
546  g_strlcat (key, l->data, siz);
547  g_strlcat (key, ".", siz);
548  }
549 
550  g_strlcat (key, config->value_name, siz);
551 
552  prop = xmms_config_property_new (g_strdup (key));
553  xmms_config_property_set_data (prop, (gchar *) text);
554 
555  g_tree_replace (config->properties, (gchar *) prop->name, prop);
556 }

678 static void
679 load_config (xmms_config_t *config, const gchar *filename)
680 {
681  GMarkupParser pars;
682  GMarkupParseContext *ctx;
683  int fd = -1;
684  gboolean parserr = FALSE, eof = FALSE;
685 
686  memset (&pars, 0, sizeof (pars));
687 
688  pars.start_element = xmms_config_parse_start;
689  pars.end_element = xmms_config_parse_end;
690  pars.text = xmms_config_parse_text;
691 
692  if (g_strcmp0 (filename, "memory://") == 0) {
693  /* no persistent storage */
694  return;
695  }
696 
697  if (g_file_test (filename, G_FILE_TEST_EXISTS)) {
698  fd = open (filename, O_RDONLY);
699  }
700 
701  if (fd < 0) {
702  xmms_log_info ("No configfile specified, using default values.");
703  return;
704  }
705 
706  config->is_parsing = TRUE;
707  config->states = g_queue_new ();
708  config->sections = g_queue_new ();
709 
710  ctx = g_markup_parse_context_new (&pars, 0, config, NULL);
711 
712  while ((!eof) && (!parserr)) {
713  GError *error = NULL;
714  gchar buffer[1024];
715  gint ret;
716 
717  ret = read (fd, buffer, 1024);
718  if (ret < 1) {
719  g_markup_parse_context_end_parse (ctx, &error);
720  if (error) {
721  xmms_log_error ("Cannot parse config file: %s",
722  error->message);
723  g_error_free (error);
724  error = NULL;
725  parserr = TRUE;
726  }
727  eof = TRUE;
728  }
729

731  if (error) {

733  error->message);
734  g_error_free (error);
735  error = NULL;
736  parserr = TRUE;
737  }

739  * above managed to parse the <xmms> element during the first
740  * iteration of this loop */
741  if (XMMS_CONFIG_VERSION > config->version) {
742  clear_config (config);
743  break;
744  }
745  }
746 
747  close (fd);
748  g_markup_parse_context_free (ctx);
749 
750  while (!g_queue_is_empty (config->sections)) {
751  g_free (g_queue_pop_head (config->sections));
752  }
753 
754  g_queue_free (config->states);
755  g_queue_free (config->sections);
756 
757  config->is_parsing = FALSE;
758

760  xmms_log_info ("The config file could not be parsed, reverting to default configuration..");
761  clear_config (config);
762  }

801 static gboolean
802 dump_tree (gchar *current_key, xmms_config_property_t *prop,
803  dump_tree_data_t *data)
804 {
805  gchar *prop_name, section[256];
806  gchar *dot = NULL, *current_last_dot, *start = current_key;
807 
808  prop_name = strrchr (current_key, '.');
809 
810  /* check whether we need to open a new section.
811  * this is always the case if data->prev_key == NULL.
812  * but if the sections of the last key and the current key differ,
813  * we also need to do that.
814  */
815  if (data->prev_key) {
816  gchar *c = current_key, *o = data->prev_key;
817  gsize dots = 0;
818 
819  /* position c and o at the respective ends of the common
820  * prefixes of the previous and the current key.
821  */
822  while (*c && *o && *c == *o) {
823  c++;
824  o++;
825 
826  if (*c == '.')
827  start = c + 1;
828  };
829 
830  /* from this position on, count the number of dots in the
831  * previous key (= number of dots that are present in the
832  * previous key, but no the current key).
833  */
834  while (*o) {
835  if (*o == '.')
836  dots++;
837 
838  o++;
839  };
840 
841  /* we'll close the previous key's sections now, so we don't
842  * have to worry about it next time this function is called.
843  */
844  if (dots)
845  data->prev_key = NULL;
846 
847  while (dots--) {
848  /* decrease indent level */
849  data->indent[--data->indent_level] = '\0';
850 
851  fprintf (data->fp, "%s</section>\n", data->indent);
852  }
853  }
854 
855  /* open section tags */
856  dot = strchr (start, '.');
857  current_last_dot = start - 1;
858 
859  while (dot) {
860  strncpy (section, current_last_dot + 1, dot - current_last_dot + 1);
861  section[dot - current_last_dot - 1] = 0;
862 
863  fprintf (data->fp, "%s<section name=\"%s\">\n",
864  data->indent, section);
865 
866  /* increase indent level */
867  g_assert (data->indent_level < 127);
868  data->indent[data->indent_level] = '\t';
869  data->indent[++data->indent_level] = '\0';
870 
871  current_last_dot = dot;
872  dot = strchr (dot + 1, '.');
873  };
874 
875  data->prev_key = current_key;
876 
877  fprintf (data->fp, "%s<property name=\"%s\">%s</property>\n",
878  data->indent, prop_name + 1,
879  xmms_config_property_get_string (prop));
880 
881  return FALSE; /* keep going */
882 }

884 /**
885  * @internal Save the global configuration to disk.
886  * @param file Absolute path to configfile. This will be overwritten.
887  * @return TRUE on success.
888  */
889 gboolean
890 xmms_config_save (void)
891 {
892  FILE *fp = NULL;
893  dump_tree_data_t data;
894 
895  g_return_val_if_fail (global_config, FALSE);
896 
897  if (g_strcmp0 (global_config->filename, "memory://") == 0) {
898  return FALSE;
899  }
900 
901  /* don't try to save config while it's being read */
902  if (global_config->is_parsing)
903  return FALSE;
904 
905  if (!(fp = fopen (global_config->filename, "w"))) {
906  xmms_log_error ("Couldn't open %s for writing.",
907  global_config->filename);
908  return FALSE;
909  }
910 
911  fprintf (fp, "<?xml version=\"1.0\"?>\n<xmms version=\"%i\">\n",
912  XMMS_CONFIG_VERSION);
913 
914  data.fp = fp;
915  data.state = XMMS_CONFIG_STATE_START;
916  data.prev_key = NULL;
917 
918  strcpy (data.indent, "\t");
919  data.indent_level = 1;
920 
921  g_tree_foreach (global_config->properties,
922  (GTraverseFunc) dump_tree, &data);
923 
924  /* close the remaining section tags. the final indent level
925  * was started with the opening xmms tag, so the loop condition
926  * is '> 1' here rather than '> 0'.
927  */
928  while (data.indent_level > 1) {
929  /* decrease indent level */
930  data.indent[--data.indent_level] = '\0';
931 
932  fprintf (fp, "%s</section>\n", data.indent);
933  }
934 
935  fprintf (fp, "</xmms>\n");
936  fclose (fp);
937 
938  return TRUE;
939 }
