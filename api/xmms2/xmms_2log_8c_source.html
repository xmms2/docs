62 static void
63 xmms_log_handler (const gchar *log_domain, GLogLevelFlags log_level, const gchar *message, gpointer user_data)
64 {
65  gchar logts_buf[256];
66  time_t tv = 0;
67  struct tm st;
68  const char *level = "??";
69  gint verbosity = GPOINTER_TO_INT (user_data);
70 
71  tv = time (NULL);
72 
73  if (log_level & G_LOG_LEVEL_CRITICAL) {
74  level = " FAIL";
75  } else if (log_level & G_LOG_LEVEL_ERROR) {
76  level = "FATAL";
77  } else if (log_level & G_LOG_LEVEL_WARNING) {
78  level = "ERROR";
79  } else if (log_level & G_LOG_LEVEL_MESSAGE) {
80  level = " INFO";
81  if (verbosity < 1)
82  return;
83  } else if (log_level & G_LOG_LEVEL_DEBUG) {
84  level = "DEBUG";
85  if (verbosity < 2)
86  return;
87  }
88 
89  if (!logts_format ||
90  !xmms_localtime (&tv, &st) ||
91  !strftime (logts_buf, sizeof(logts_buf), logts_format, &st)) {
92  logts_buf[0] = '\0';
93  }
94 
95  if (log_domain && log_domain[0]) {
96  printf ("%s%s in %s: %s\n", logts_buf, level, log_domain, message);
97  } else {
98  printf ("%s%s: %s\n", logts_buf, level, message);
99  }
100 
101  fflush (stdout);
102 
103  if (log_level & G_LOG_LEVEL_ERROR) {
104  exit (EXIT_FAILURE);
105  }
106 }
