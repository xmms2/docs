121 xmms_ipc_register_signal (xmms_ipc_client_t *client,

123 {
124  xmmsv_t *arg;
125  gint32 signalid;
126  int r;
127 
128  if (!arguments || !xmmsv_list_get (arguments, 0, &arg)) {

132 
133  r = xmmsv_get_int32 (arg, &signalid);

183 static void
184 process_msg (xmms_ipc_client_t *client, xmms_ipc_msg_t *msg)
185 {
186  xmms_object_t *object;
187  xmms_object_cmd_arg_t arg;
188  xmms_ipc_msg_t *retmsg;
189  xmmsv_t *error, *arguments;
190  uint32_t objid, cmdid;
191 
192  g_return_if_fail (msg);
193 
194  objid = xmms_ipc_msg_get_object (msg);
195  cmdid = xmms_ipc_msg_get_cmd (msg);
196 
197  if (!xmms_ipc_msg_get_value (msg, &arguments)) {
198  xmms_log_error ("Cannot read command arguments. "
199  "Ignoring command.");
200 
201  return;
202  }
203 
204  if (objid == XMMS_IPC_OBJECT_SIGNAL) {
205  if (cmdid == XMMS_IPC_COMMAND_SIGNAL) {
206  xmms_ipc_register_signal (client, msg, arguments);
207  } else if (cmdid == XMMS_IPC_COMMAND_BROADCAST) {
208  xmms_ipc_register_broadcast (client, msg, arguments);
209  } else {
210  xmms_log_error ("Bad command id (%d) for signal object", cmdid);
211  }
212 
213  goto out;
214  }
215 
216  if (objid >= XMMS_IPC_OBJECT_END) {
217  xmms_log_error ("Bad object id (%d)", objid);
218  goto out;
219  }
220 
221  g_mutex_lock (&ipc_object_pool_lock);
222  object = ipc_object_pool->objects[objid];
223  g_mutex_unlock (&ipc_object_pool_lock);
224  if (!object) {
225  xmms_log_error ("Object %d was not found!", objid);
226  goto out;
227  }
228 
229  if (!g_tree_lookup (object->cmds, GUINT_TO_POINTER (cmdid))) {
230  xmms_log_error ("No such cmd %d on object %d", cmdid, objid);
231  goto out;
232  }
233 
234  xmms_object_cmd_arg_init (&arg);
235  arg.args = arguments;
236  arg.client = client->id;
237  arg.cookie = xmms_ipc_msg_get_cookie (msg);
238 
239  xmms_object_cmd_call (object, cmdid, &arg);
240  if (xmms_error_isok (&arg.error)) {
241  if (!arg.retval) {
242  /* Skip reply if method is a noreply and didn't fail */
243  goto out;
244  }
245 
246  retmsg = xmms_ipc_msg_new (objid, XMMS_IPC_COMMAND_REPLY);
247  xmms_ipc_handle_cmd_value (retmsg, arg.retval);
248  } else {
249  /* FIXME: or we could omit setting the command to _CMD_ERROR
250  * and let the client check whether the value it got is an
251  * error xmmsv_t. If so, don't forget to
252  * update the client-side of IPC too. */
253  retmsg = xmms_ipc_msg_new (objid, XMMS_IPC_COMMAND_ERROR);
254 
255  error = xmmsv_new_error (xmms_error_message_get (&arg.error));
256  xmms_ipc_msg_put_value (retmsg, error);
257  xmmsv_unref (error);
258  }
259 
260  if (arg.retval)
261  xmmsv_unref (arg.retval);
262 
263  xmms_ipc_msg_set_cookie (retmsg, xmms_ipc_msg_get_cookie (msg));
264  g_mutex_lock (&client->lock);
265  xmms_ipc_client_msg_write (client, retmsg);
266  g_mutex_unlock (&client->lock);
267 
268 out:
269  if (arguments) {
270  xmmsv_unref (arguments);
271  }
272 }

275 static gboolean
276 xmms_ipc_client_read_cb (GIOChannel *iochan,
277  GIOCondition cond,
278  gpointer data)
279 {
280  xmms_ipc_client_t *client = data;
281  bool disconnect = FALSE;
282 
283  g_return_val_if_fail (client, FALSE);
284 
285  if (cond & G_IO_IN) {
286  while (TRUE) {
287  if (!client->read_msg) {
288  client->read_msg = xmms_ipc_msg_alloc ();
289  }
290 
291  if (xmms_ipc_msg_read_transport (client->read_msg, client->transport, &disconnect)) {
292  xmms_ipc_msg_t *msg = client->read_msg;
293  client->read_msg = NULL;
294  process_msg (client, msg);
295  xmms_ipc_msg_destroy (msg);
296  } else {
297  break;
298  }
299  }
300  }
301 
302  if (disconnect || (cond & G_IO_HUP)) {
303  if (client->read_msg) {
304  xmms_ipc_msg_destroy (client->read_msg);
305  client->read_msg = NULL;
306  }
307  XMMS_DBG ("disconnect was true!");
308  g_main_loop_quit (client->ml);
309  return FALSE;
310  }
311 
312  if (cond & G_IO_ERR) {
313  xmms_log_error ("Client got error, maybe connection died?");
314  g_main_loop_quit (client->ml);
315  return FALSE;
316  }
317 
318  return TRUE;
319 }

321 static gboolean
322 xmms_ipc_client_write_cb (GIOChannel *iochan,
323  GIOCondition cond,
324  gpointer data)
325 {
326  xmms_ipc_client_t *client = data;
327  bool disconnect = FALSE;
328 
329  g_return_val_if_fail (client, FALSE);
330 
331  while (TRUE) {
332  xmms_ipc_msg_t *msg;
333 
334  g_mutex_lock (&client->lock);
335  msg = g_queue_peek_head (client->out_msg);
336  g_mutex_unlock (&client->lock);
337 
338  if (!msg)
339  break;
340 
341  if (!xmms_ipc_msg_write_transport (msg,
342  client->transport,
343  &disconnect)) {
344  if (disconnect) {
345  break;
346  } else {
347  /* try sending again later */
348  return TRUE;
349  }
350  }
351 
352  g_mutex_lock (&client->lock);
353  g_queue_pop_head (client->out_msg);
354  g_mutex_unlock (&client->lock);
355 
356  xmms_ipc_msg_destroy (msg);
357  }
358 
359  return FALSE;
360 }

419  g_mutex_init (&client->lock);
420  client->id = next_client_id++;

425 static void
426 xmms_ipc_client_destroy (xmms_ipc_client_t *client)
427 {
428  guint i;
429 
430  XMMS_DBG ("Destroying client!");
431 
432  if (client->ipc) {
433  g_mutex_lock (&client->ipc->mutex_lock);
434  client->ipc->clients = g_list_remove (client->ipc->clients, client);
435  g_mutex_unlock (&client->ipc->mutex_lock);
436  }
437 
438  g_main_loop_unref (client->ml);
439  g_io_channel_unref (client->iochan);
440 
441  xmms_ipc_transport_destroy (client->transport);
442 
443  g_mutex_lock (&client->lock);
444  while (!g_queue_is_empty (client->out_msg)) {
445  xmms_ipc_msg_t *msg = g_queue_pop_head (client->out_msg);
446  xmms_ipc_msg_destroy (msg);
447  }
448 
449  g_queue_free (client->out_msg);
450 
451  for (i = 0; i < XMMS_IPC_SIGNAL_END; i++) {
452  g_list_free (client->broadcasts[i]);
453  }
454 
455  g_mutex_unlock (&client->lock);
456  g_mutex_clear (&client->lock);
457  g_free (client);
458 }

529  * Look up a client based on its id.

532 xmms_ipc_lookup_client (gint32 id)
533 {
534  GList *c, *s;
535  xmms_ipc_t *ipc;
536  xmms_ipc_client_t *cli;
537 
538  g_mutex_lock (&ipc_servers_lock);
539  for (s = ipc_servers; s && s->data; s = g_list_next (s)) {
540  ipc = s->data;
541 
542  g_mutex_lock (&ipc->mutex_lock);
543  for (c = ipc->clients; c; c = g_list_next (c)) {
544  cli = c->data;
545 
546  if (cli->id == id) {
547  g_mutex_unlock (&ipc->mutex_lock);
548  g_mutex_unlock (&ipc_servers_lock);
549  return cli;
550  }
551  }
552  g_mutex_unlock (&ipc->mutex_lock);
553  }
554  g_mutex_unlock (&ipc_servers_lock);
555 
556  return NULL;

560  * Put a message in the queue awaiting to be sent to the client.
561  * Should hold client->lock.
562  */
563 static gboolean
564 xmms_ipc_client_msg_write (xmms_ipc_client_t *client, xmms_ipc_msg_t *msg)
565 {
566  gboolean queue_empty;
567 
568  g_return_val_if_fail (client, FALSE);
569  g_return_val_if_fail (msg, FALSE);
570 
571  queue_empty = g_queue_is_empty (client->out_msg);
572  g_queue_push_tail (client->out_msg, msg);
573 
574  /* If there's no write in progress, add a new callback */
575  if (queue_empty) {
576  GMainContext *context = g_main_loop_get_context (client->ml);
577  GSource *source = g_io_create_watch (client->iochan, G_IO_OUT);
578 
579  g_source_set_callback (source,
580  (GSourceFunc) xmms_ipc_client_write_cb,
581  (gpointer) client,
582  NULL);
583  g_source_attach (source, context);
584  g_source_unref (source);
585 
586  g_main_context_wakeup (context);
587  }
588 
589  return TRUE;
590 }

616 static gboolean
617 xmms_ipc_source_accept (GIOChannel *chan, GIOCondition cond, gpointer data)
618 {
619  xmms_ipc_t *ipc = (xmms_ipc_t *) data;
620  xmms_ipc_transport_t *transport;
621  xmms_ipc_client_t *client;
622  GThread * client_thread;
623 
624  if (!(cond & G_IO_IN)) {
625  xmms_log_error ("IPC listener got error/hup");
626  return FALSE;
627  }
628 
629  XMMS_DBG ("Client connected");
630  transport = xmms_ipc_server_accept (ipc->transport);
631  if (!transport) {
632  xmms_log_error ("accept returned null!");
633  return TRUE;
634  }
635 
636  client = xmms_ipc_client_new (ipc, transport);
637  if (!client) {
638  xmms_ipc_transport_destroy (transport);
639  return TRUE;
640  }
641 
642  g_mutex_lock (&ipc->mutex_lock);
643  ipc->clients = g_list_append (ipc->clients, client);
644  g_mutex_unlock (&ipc->mutex_lock);
645 
646  /* Now that the client has been registered in the ipc->clients list
647  * we may safely start its thread.
648  */
649  client_thread = g_thread_new ("x2 client", xmms_ipc_client_thread, client);
650  /* let the thread free it's resources once it is finished */
651  g_thread_unref (client_thread);
652 
653  return TRUE;
654 }

945 /**
946  * Start the server
947  */
948 gboolean
949 xmms_ipc_setup_server (const gchar *path)
950 {
951  xmms_ipc_transport_t *transport;
952  xmms_ipc_t *ipc;
953  gchar **split;
954  gint i = 0, num_init = 0;
955  g_return_val_if_fail (path, FALSE);
956 
957  split = g_strsplit (path, ";", 0);
958 
959  for (i = 0; split && split[i]; i++) {
960  ipc = g_new0 (xmms_ipc_t, 1);
961  if (!ipc) {
962  XMMS_DBG ("No IPC server initialized.");
963  continue;
964  }
965 
966  transport = xmms_ipc_server_init (split[i]);
967  if (!transport) {
968  g_free (ipc);
969  xmms_log_error ("Couldn't setup IPC listening on '%s'.", split[i]);
970  continue;
971  }
972 
973 
974  g_mutex_init (&ipc->mutex_lock);
975  ipc->transport = transport;
976  ipc->signals = ipc_object_pool->signals;
977  ipc->broadcasts = ipc_object_pool->broadcasts;
978  ipc->objects = ipc_object_pool->objects;
979 
980  xmms_ipc_setup_server_internaly (ipc);
981  xmms_log_info ("IPC listening on '%s'.", split[i]);
982 
983  g_mutex_lock (&ipc_servers_lock);
984  ipc_servers = g_list_prepend (ipc_servers, ipc);
985  g_mutex_unlock (&ipc_servers_lock);
986 
987  num_init++;
988  }
989 
990  g_strfreev (split);
991 
992 
993  /* If there is less than one socket, there is sth. wrong. */
994  if (num_init < 1)
995  return FALSE;
996 
997  XMMS_DBG ("IPC setup done.");
998  return TRUE;
999 }
