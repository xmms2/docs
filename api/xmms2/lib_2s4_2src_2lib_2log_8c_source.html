160 /**
161  * Estimates the size needed to write the entire oplist to the log.
162  *
163  * @param list The oplist to estimate the size of
164  * @param writing A pointer to an int that will be set to 1
165  * if the oplist contains a write.
166  * @return The estimated size.
167  */
168 static int _estimate_size (oplist_t *list, int *writing) {
169  int ret = 0, largest = 0;
170  _oplist_first (list);
171 
172  while (_oplist_next (list)) {
173  const char *key_a, *key_b, *src;
174  const s4_val_t *val_a, *val_b;
175  int size = sizeof (struct log_header);
176 
177  if (_oplist_get_add (list, &key_a, &val_a, &key_b, &val_b, &src)) {
178  size += sizeof (struct mod_header);
179  size += strlen (key_a) + strlen (key_a) + strlen (src);
180  size += _get_val_len (val_a) + _get_val_len (val_b);
181  } else if (_oplist_get_del (list, &key_a, &val_a, &key_b, &val_b, &src)) {
182  size += sizeof (struct mod_header);
183  size += strlen (key_a) + strlen (key_a) + strlen (src);
184  size += _get_val_len (val_a) + _get_val_len (val_b);
185  } else if (_oplist_get_writing (list)) {
186  /* A write is only the size of a log header */
187  *writing = 1;
188  }
189 
190  largest = MAX (size, largest);
191  ret += size;
192  }
193 
194  if (ret == 0) {
195  return 0;
196  }
197 
198  /* Add the size of begin, end and a warp-around header.
199  * We also add the size of the largest entry as this is the
200  * most extra space that could be needed on a wrap-around.
201  */
202  ret += 3 * sizeof (struct log_header) + largest;
203  return ret;
204 }

225 /**
226  * Writes a log header to the log.
227  *
228  * @param s4 The database handle.
229  * @param hdr The header to write.
230  * @param size The size of the data following the header.
231  */
232 static void _log_write_header (s4_t *s4, struct log_header hdr, int size)
233 {
234  log_number_t pos, round;
235 
236  if (s4->log_data->logfile == NULL)
237  return;
238 
239  pos = s4->log_data->next_logpoint % LOG_SIZE;
240  round = s4->log_data->next_logpoint / LOG_SIZE;
241 
242  /* Wrap around if we're at the end */
243  if ((pos + size) > (LOG_SIZE - sizeof (struct log_header) * 2)) {
244  struct log_header hdr;
245 
246  hdr.num = pos + round * LOG_SIZE;
247  hdr.type = LOG_ENTRY_WRAP;
248  fwrite (&hdr, sizeof (struct log_header), 1, s4->log_data->logfile);
249  pos = 0;
250  round++;
251  rewind (s4->log_data->logfile);
252  }
253 
254  hdr.num = pos + round * LOG_SIZE;
255  fwrite (&hdr, sizeof (struct log_header), 1, s4->log_data->logfile);
256 
257  s4->log_data->last_logpoint = s4->log_data->next_logpoint;
258  s4->log_data->next_logpoint = ftell (s4->log_data->logfile) + round * LOG_SIZE + size;
259 }

344 /**
345  * Writes all the operations in an oplist to disk.
346  *
347  * @param list The oplist to write.
348  * @return 0 on error, non-zero on success.
349  */
350 int _log_write (oplist_t *list)
351 {
352  s4_t *s4 = _oplist_get_db (list);
353  int writing = 0;
354  int size = _estimate_size (list, &writing);
355 
356  if (s4->log_data->logfile == NULL || size == 0)
357  return 1;
358 
359  _log_lock (s4);
360  if (writing) {
361  s4->log_data->last_synced = s4->log_data->last_logpoint;
362  }
363 
364  if ((s4->log_data->next_logpoint + size) > (s4->log_data->last_checkpoint + LOG_SIZE)) {
365  _log_unlock (s4);
366  return writing;
367  }
368 
369  _log_simple (s4, LOG_ENTRY_BEGIN);
370 
371  _oplist_first (list);
372  while (_oplist_next (list)) {
373  const char *key_a, *key_b, *src;
374  const s4_val_t *val_a, *val_b;
375 
376  if (_oplist_get_add (list, &key_a, &val_a, &key_b, &val_b, &src)) {
377  _log_mod (s4, LOG_ENTRY_ADD, key_a, val_a, key_b, val_b, src);
378  } else if (_oplist_get_del (list, &key_a, &val_a, &key_b, &val_b, &src)) {
379  _log_mod (s4, LOG_ENTRY_DEL, key_a, val_a, key_b, val_b, src);
380  } else if (_oplist_get_writing (list)) {
381  _log_simple (s4, LOG_ENTRY_WRITING);
382  }
383  }
384 
385  _log_simple (s4, LOG_ENTRY_END);
386 
387  if (s4->log_data->last_synced > (s4->log_data->last_checkpoint + LOG_SIZE / 2))
388  _start_sync (s4);
389 
390  _log_flush (s4);
391  _log_unlock (s4);
392  return 1;
393 }

447  * Reads a modification entry (add or del).
448  *
449  * @param s4 The database.
450  * @param list The oplist to insert the operation in.
451  * @param type The type of the log entry.
452  * @return 0 on error, non-zero on success.

454 static int _read_mod (s4_t *s4, oplist_t *list, log_type_t type)
455 {

457  const s4_val_t *val_a, *val_b;
458  struct mod_header mhdr;
459 
460  if (list == NULL)
461  return 0;
462 
463  fread (&mhdr, sizeof (struct mod_header), 1, s4->log_data->logfile);
464 
465  key_a = _read_str (s4, mhdr.ka_len);
466  val_a = _read_val (s4, mhdr.va_len);

474  return 0;
475  }

478  _oplist_insert_add (list, key_a, val_a, key_b, val_b, src);
479  } else if (type == LOG_ENTRY_DEL) {

481  }
482 
483  return 1;

486 /**
487  * Redoes everything that happened since the last checkpoint
488  *
489  * @param s4 The database to add changes to
490  * @return 0 on error, non-zero otherwise
491  */
492 static int _log_redo (s4_t *s4)
493 {
494  struct log_header hdr;
495  log_number_t pos, round, new_checkpoint = -1, new_synced = -1;
496  log_number_t last_valid_logpoint;
497  oplist_t *oplist = NULL;
498  int invalid_entry = 0;
499 
500  fflush (s4->log_data->logfile);
501 
502  /* Check if the log wrapped around since our last write */
503  pos = s4->log_data->last_logpoint % LOG_SIZE;
504  if (fseek (s4->log_data->logfile, pos, SEEK_SET) != 0 ||
505  fread (&hdr, sizeof (struct log_header), 1, s4->log_data->logfile) != 1) {
506  return 0;
507  }
508 
509  /* If it did, we have to read in everything */
510  if (hdr.num != s4->log_data->last_logpoint) {
511  _reread_file (s4);
512  }
513 
514  last_valid_logpoint = s4->log_data->last_logpoint;
515  s4->log_data->next_logpoint = s4->log_data->last_logpoint + sizeof (struct log_header);
516 
517  pos = s4->log_data->next_logpoint % LOG_SIZE;
518  round = s4->log_data->next_logpoint / LOG_SIZE;
519  if (fseek (s4->log_data->logfile, pos, SEEK_SET) != 0) {
520  return 0;
521  }
522 
523  /* Read log entries until fread fails, or the header num is different
524  * from the expected number.
525  */
526  while (!invalid_entry
527  && fread (&hdr, sizeof (struct log_header), 1, s4->log_data->logfile) == 1
528  && hdr.num == (pos + round * LOG_SIZE)) {
529 
530  s4->log_data->last_logpoint = s4->log_data->next_logpoint;
531 
532  switch (hdr.type) {
533  case LOG_ENTRY_WRAP:
534  round++;
535  rewind (s4->log_data->logfile);
536  break;
537 
538  case LOG_ENTRY_DEL:
539  case LOG_ENTRY_ADD:
540  if (!_read_mod (s4, oplist, hdr.type))
541  invalid_entry = 1;
542 
543  break;
544 
545  case LOG_ENTRY_CHECKPOINT:
546  fread (&new_checkpoint, sizeof (log_number_t), 1, s4->log_data->logfile);
547  break;
548 
549  case LOG_ENTRY_WRITING:
550  new_synced = s4->log_data->last_logpoint;
551  break;
552 
553  case LOG_ENTRY_BEGIN:
554  oplist = _oplist_new (_transaction_dummy_alloc (s4));
555  new_checkpoint = -1;
556  new_synced = -1;
557  break;
558 
559  case LOG_ENTRY_END:
560  if (oplist == NULL) {
561  break;
562  }
563 
564  _oplist_execute (oplist, 0);
565  _transaction_dummy_free (_oplist_get_trans (oplist));
566  _oplist_free (oplist);
567  oplist = NULL;
568 
569  if (new_checkpoint != -1) {
570  s4->log_data->last_synced = s4->log_data->last_checkpoint = new_checkpoint;
571  } else if (new_synced != -1) {
572  s4->log_data->last_synced = new_synced;
573  }
574  last_valid_logpoint = s4->log_data->last_logpoint;
575  break;
576 
577  case LOG_ENTRY_INIT:
578  /* Ignore */
579  break;
580 
581  default:
582  /* Unknown header type */
583  invalid_entry = 1;
584  break;
585  }
586 
587  pos = ftell (s4->log_data->logfile);
588  s4->log_data->next_logpoint = pos + round * LOG_SIZE;
589  }
590 
591  if (oplist != NULL) {
592  _transaction_dummy_free (_oplist_get_trans (oplist));
593  _oplist_free (oplist);
594  }
595 
596  s4->log_data->last_logpoint = last_valid_logpoint;
597  s4->log_data->next_logpoint = last_valid_logpoint + sizeof (struct log_header);
598  pos = s4->log_data->next_logpoint % LOG_SIZE;
599  fseek (s4->log_data->logfile, pos, SEEK_SET);
600 
601  return 1;
602 }
