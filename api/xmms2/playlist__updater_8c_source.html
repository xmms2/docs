158 /**
159  * Wait until a playlist needs update.
160  * @internal
161  */
162 static gpointer
163 xmms_playlist_updater_loop (xmms_playlist_updater_t *updater)
164 {
165  gchar *plname;
166 
167  g_mutex_lock (&updater->mutex);
168 
169  while (updater->keep_running) {
170  if (!updater->stack) {
171  g_cond_wait (&updater->cond, &updater->mutex);
172  } else {
173  plname = xmms_playlist_updater_pop (updater);
174  g_mutex_unlock (&updater->mutex);
175  xmms_playlist_update (updater->playlist, plname);
176  g_mutex_lock (&updater->mutex);
177  g_free (plname);
178  }
179  }
180 
181  g_mutex_unlock (&updater->mutex);
182 
183  return NULL;
184 }

222 /**
223  * Queue a playlist for update.
224  */
225 void
226 xmms_playlist_updater_push (xmms_playlist_updater_t *updater,
227  const gchar *plname)
228 {
229  g_return_if_fail (updater);
230  g_return_if_fail (plname);
231 
232  g_mutex_lock (&updater->mutex);
233 
234  /* don't schedule the playlist if it's already scheduled */
235  if (!g_list_find_custom (updater->stack, plname, (GCompareFunc) g_strcmp0)) {
236  updater->stack = g_list_prepend (updater->stack, g_strdup (plname));
237  g_cond_signal (&updater->cond);
238  }
239 
240  g_mutex_unlock (&updater->mutex);
241 }
