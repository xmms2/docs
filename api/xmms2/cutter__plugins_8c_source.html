76 static gboolean
77 xmms_segment_init (xmms_xform_t *xform)
78 {
79  const gchar *nptr;
80  char *endptr;
81  gint startms, stopms;
82  gint64 startbytes, stopbytes;
83  const gchar *metakey;
84  xmms_stream_type_t *intype;
85 
86  g_return_val_if_fail (xform, FALSE);
87 
88  xmms_xform_outdata_type_copy (xform);
89 
90  /* get startms */
91  metakey = XMMS_MEDIALIB_ENTRY_PROPERTY_STARTMS;
92  if (xmms_xform_metadata_get_str (xform, metakey, &nptr)) {
93  startms = (gint) strtol (nptr, &endptr, 10);
94  if (*endptr != '\0') {
95  XMMS_DBG ("\"startms\" has garbage, ignoring");
96  startms = 0;
97  }
98  } else {
99  startms = 0;
100  }
101 
102  /* get stopms */
103  metakey = XMMS_MEDIALIB_ENTRY_PROPERTY_STOPMS;
104  if (xmms_xform_metadata_get_str (xform, metakey, &nptr)) {
105  stopms = (gint) strtol (nptr, &endptr, 10);
106  if (*endptr != '\0') {
107  xmms_log_info ("\"stopms\" has garbage, ignoring");
108  stopms = INT_MAX;
109  }
110  } else {
111  stopms = INT_MAX;
112  }
113 
114  /* some calculation */
115  intype = xmms_xform_intype_get (xform);
116 
117  startbytes = xmms_sample_ms_to_bytes (intype, startms);
118  if (stopms == INT_MAX) {
119  stopbytes = G_MAXINT64;
120  } else {
121  stopbytes = xmms_sample_ms_to_bytes (intype, stopms);
122  }
123 
124  return xmms_cutter_init (xform, startbytes, stopbytes);
125 }

182 static gboolean
183 xmms_cutter_init (xmms_xform_t *xform, gint64 start_bytes, gint64 stop_bytes)
184 {
185  gint fmt, chans, srate;
186  xmms_cutter_data_t *data;
187  xmms_error_t error;
188  gint64 res;
189 
190  g_return_val_if_fail (xform, FALSE);
191 
192  /* Set outdata_type */
193  fmt = xmms_xform_indata_get_int (xform, XMMS_STREAM_TYPE_FMT_FORMAT);
194  chans = xmms_xform_indata_get_int (xform, XMMS_STREAM_TYPE_FMT_CHANNELS);
195  srate = xmms_xform_indata_get_int (xform, XMMS_STREAM_TYPE_FMT_SAMPLERATE);
196  xmms_xform_outdata_type_add (xform,
197  XMMS_STREAM_TYPE_MIMETYPE, "audio/pcm",
198  XMMS_STREAM_TYPE_FMT_FORMAT, fmt,
199  XMMS_STREAM_TYPE_FMT_CHANNELS, chans,
200  XMMS_STREAM_TYPE_FMT_SAMPLERATE, srate,
201  XMMS_STREAM_TYPE_END);
202 
203  /* Set duration */
204  if (stop_bytes != G_MAXINT64) {
205  const gchar *metakey;
206  gint64 dur;
207 
208  if (stop_bytes < start_bytes) {
209  xmms_log_error ("Negative duration requested, rejecting.");
210  return FALSE;
211  }
212  metakey = XMMS_MEDIALIB_ENTRY_PROPERTY_DURATION;
213  dur = xmms_sample_bytes_to_ms (xmms_xform_outtype_get (xform),
214  stop_bytes - start_bytes);
215  xmms_xform_metadata_set_int (xform, metakey, dur);
216  }
217 
218  /* allocate and set data */
219  data = g_new0 (xmms_cutter_data_t, 1);
220  data->start_bytes = start_bytes;
221  data->seek_bytes = 0;
222  data->current_bytes = 0;
223  data->stop_bytes = stop_bytes;
224 
225  xmms_xform_private_data_set (xform, data);
226 
227  /* Now seek to start of desired data */
228  if (data->start_bytes > 0) {
229  xmms_error_reset (&error);
230  res = xmms_cutter_seek (xform, 0, XMMS_XFORM_SEEK_SET, &error);
231  if (res == -1 || xmms_error_iserror (&error)) {
232  xmms_log_error ("Couldn't seek, assuming we're at 0. Error is '%s'",
233  xmms_error_message_get (&error));
234  data->current_bytes = 0;
235  data->seek_bytes = data->start_bytes;
236  }
237  }
238 
239  return TRUE;
240 }

251 static gint
252 xmms_cutter_read (xmms_xform_t *xform, xmms_sample_t *buf, gint len,
253  xmms_error_t *error)
254 {
255  xmms_cutter_data_t *data;
256  gint res;
257  gint64 hi, lo;
258 
259  g_return_val_if_fail(error, -1);
260  g_return_val_if_fail(xform, -1);
261  data = xmms_xform_private_data_get (xform);
262  g_return_val_if_fail(data, -1);
263 
264  do {
265  if (data->current_bytes >= data->stop_bytes) {
266  return 0; /* EOF */
267  }
268  res = xmms_xform_read (xform, buf, len, error);
269  if (res == 0 || xmms_error_iserror (error)) {
270  return 0; /* EOF */
271  }
272  lo = MAX(0, data->seek_bytes - data->current_bytes);
273  hi = MIN(res, data->stop_bytes - data->current_bytes);
274  data->current_bytes += res;
275  } while (lo >= hi);
276  memmove (buf, buf + lo, hi-lo);
277  return hi-lo;
278 }

280 static gint64
281 xmms_cutter_seek (xmms_xform_t *xform, gint64 samples,
282  xmms_xform_seek_mode_t whence, xmms_error_t *error)
283 {
284  xmms_cutter_data_t *data;
285  xmms_stream_type_t *ot;
286  gint64 bytes, res, outer_samples;
287 
288  g_return_val_if_fail(error, -1);
289  g_return_val_if_fail(xform, -1);
290  ot = xmms_xform_outtype_get (xform);
291  data = xmms_xform_private_data_get (xform);
292  g_return_val_if_fail(data, -1);
293 
294  bytes = xmms_sample_samples_to_bytes(ot, samples);
295  switch (whence) {
296  case XMMS_XFORM_SEEK_CUR:
297  bytes += data->current_bytes;
298  break;
299  case XMMS_XFORM_SEEK_SET:
300  bytes += data->start_bytes;
301  break;
302  case XMMS_XFORM_SEEK_END:
303  if (data->stop_bytes == G_MAXINT64) {
304  xmms_error_set (error, XMMS_ERROR_INVAL,
305  "Cannot seek relative to end without stop_bytes");
306  return -1;
307  }
308  bytes += data->stop_bytes;
309  }
310 
311  if (bytes < data->start_bytes || bytes > data->stop_bytes) {
312  xmms_error_set (error, XMMS_ERROR_INVAL, "Seeking out of range");
313  return -1;
314  }
315 
316  res = xmms_xform_seek (xform, xmms_sample_bytes_to_samples_inexact (ot, bytes),
317  XMMS_XFORM_SEEK_SET, error);
318  if (res == -1 || xmms_error_iserror (error)) {
319  return -1;
320  }
321 
322  data->current_bytes = xmms_sample_samples_to_bytes (ot, res);
323  data->seek_bytes = MAX(bytes, data->current_bytes);
324 
325  outer_samples = xmms_sample_bytes_to_samples (ot,
326  data->seek_bytes - data->start_bytes, error);
327  if (xmms_error_iserror (error)) {
328  /* The running assumption is that seek_bytes and start_bytes make sense
329  in samples. If that is not the case something is very wrong. */
330  xmms_log_error ("seek_bytes - start_bytes impossible! Please report this!");
331  return -1;
332  }
333  return outer_samples;
334 }
