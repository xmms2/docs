62 static void
63 xmms_xform_effect_callbacks_init (void)
64 {
65  xmms_config_property_t *cfg;
66  xmms_xform_plugin_t *plugin;
67  gchar key[64];
68  const gchar *name;
69  gint effect_no;
70 
71  for (effect_no = 0; ; effect_no++) {
72  g_snprintf (key, sizeof (key), "effect.order.%i", effect_no);
73  cfg = xmms_config_lookup (key);
74  if (!cfg) {
75  break;
76  }
77 
78  xmms_config_property_callback_set (cfg, xmms_xform_effect_properties_update,
79  GINT_TO_POINTER (effect_no));
80 
81  name = xmms_config_property_get_string (cfg);
82  if (!name[0]) {
83  continue;
84  }
85 
86  plugin = xmms_xform_find_plugin (name);
87  if (!plugin) {
88  xmms_log_error ("Couldn't find any effect named '%s'", name);
89  continue;
90  }
91 
92  xmms_xform_plugin_config_property_register (plugin, "enabled",
93  "1", NULL, NULL);
94 
95  xmms_object_unref (plugin);
96  }
97 
98  /* the name stored in the last present property was not "" or there was no
99  last present property */
100  if ((!effect_no) || name[0]) {
101  xmms_config_property_register (key, "", xmms_xform_effect_properties_update,
102  GINT_TO_POINTER (effect_no));
103  }
104 }

106 static void
107 xmms_xform_effect_properties_update (xmms_object_t *object, xmmsv_t *data, gpointer udata)
108 {
109  xmms_config_property_t *cfg = (xmms_config_property_t *) object;
110  xmms_xform_plugin_t *plugin;
111  const gchar *name;
112  gchar key[64];
113  gint effect_no = GPOINTER_TO_INT (udata);
114 
115  name = xmms_config_property_get_string (cfg);
116  if (!name[0]) {
117  return;
118  }
119 
120  plugin = xmms_xform_find_plugin (name);
121  if (!plugin) {
122  xmms_log_error ("Couldn't find any effect named '%s'", name);
123  } else {
124  xmms_xform_plugin_config_property_register (plugin, "enabled",
125  "1", NULL, NULL);
126  xmms_object_unref (plugin);
127  }
128 
129  /* setup new effect.order.n */
130  g_snprintf (key, sizeof (key), "effect.order.%i", effect_no + 1);
131  if (!xmms_config_lookup (key)) {
132  xmms_config_property_register (key, "", xmms_xform_effect_properties_update,
133  GINT_TO_POINTER (effect_no + 1));
134  }
135 }
