46 /**
47  * Allocates a new collection #xmmsv_t of the given type.
48  *
49  * @param type the #xmmsv_coll_type_t specifying the type of collection to create.
50  * @return The new #xmmsv_t. Must be unreferenced with #xmmsv_unref.

52 xmmsv_t *
53 xmmsv_new_coll (xmmsv_coll_type_t type)
54 {
55  xmmsv_coll_internal_t *coll;
56  xmmsv_t *val;
57 
58  coll = _xmmsv_coll_new (type);
59  if (coll == NULL) {
60  return NULL;
61  }
62 
63  val = _xmmsv_new (XMMSV_TYPE_COLL);
64  if (val == NULL) {
65  _xmmsv_coll_free (coll);
66  return NULL;
67  }
68 
69  val->value.coll = coll;
70 
71  return val;
72 }

109 /**
110  * Allocate a new collection of the given type.
111  * The pointer will have to be deallocated using #xmmsv_coll_unref.
112  *
113  * @param type the #xmmsv_coll_type_t specifying the type of collection to create.
114  * @return a pointer to the newly created collection, or NULL if the type is invalid.
115  */
116 static xmmsv_coll_internal_t*
117 _xmmsv_coll_new (xmmsv_coll_type_t type)
118 {
119  xmmsv_coll_internal_t *coll;
120 
121  x_return_val_if_fail (type <= XMMS_COLLECTION_TYPE_LAST, NULL);
122 
123  coll = x_new0 (xmmsv_coll_internal_t, 1);
124  if (!coll) {
125  x_oom ();
126  return NULL;
127  }
128 
129  coll->type = type;
130 
131  coll->idlist = xmmsv_new_list ();
132  xmmsv_list_restrict_type (coll->idlist, XMMSV_TYPE_INT64);
133 
134  coll->operands = xmmsv_new_list ();
135  xmmsv_list_restrict_type (coll->operands, XMMSV_TYPE_COLL);
136 
137  coll->attributes = xmmsv_new_dict ();
138 
139  return coll;
140 }

181 static int
182 _xmmsv_coll_operand_find (xmmsv_list_iter_t *it, xmmsv_t *op)
183 {
184  xmmsv_t *v;
185 
186  while (xmmsv_list_iter_entry (it, &v)) {
187  if (v == op) {
188  return 1;
189  }
190  xmmsv_list_iter_next (it);
191  }
192  return 0;
193 }

195 /**
196  * Add the operand to the given collection.
197  * @param coll The collection to add the operand to.
198  * @param op The operand to add.
199  */
200 void
201 xmmsv_coll_add_operand (xmmsv_t *coll, xmmsv_t *op)
202 {
203  xmmsv_list_iter_t *it;
204  x_return_if_fail (coll);
205  x_return_if_fail (op);
206 
207  /* we used to check if it already existed here before */
208  if (!xmmsv_get_list_iter (coll->value.coll->operands, &it))
209  return;
210 
211  if (_xmmsv_coll_operand_find (it, op)) {
212  x_api_warning ("with an operand already in operand list");
213  xmmsv_list_iter_explicit_destroy (it);
214  return;
215  }
216 
217  xmmsv_list_iter_explicit_destroy (it);
218 
219  xmmsv_list_append (coll->value.coll->operands, op);
220 }

222 /**
223  * Remove all the occurences of the operand in the given collection.
224  * @param coll The collection to remove the operand from.
225  * @param op The operand to remove.
226  */
227 void
228 xmmsv_coll_remove_operand (xmmsv_t *coll, xmmsv_t *op)
229 {
230  xmmsv_list_iter_t *it;
231 
232  x_return_if_fail (coll);
233  x_return_if_fail (op);
234 
235  if (!xmmsv_get_list_iter (coll->value.coll->operands, &it))
236  return;
237 
238  if (_xmmsv_coll_operand_find (it, op)) {
239  xmmsv_list_iter_remove (it);
240  } else {
241  x_api_warning ("with an operand not in operand list");
242  }
243  xmmsv_list_iter_explicit_destroy (it);
244 }

664 static xmmsv_t *
665 xmmsv_coll_normalize_order_arguments (xmmsv_t *value)
666 {
667  xmmsv_t *order;
668  const char *key;
669 
670  if (value == NULL) {
671  return NULL;
672  }
673 
674  if (xmmsv_is_type (value, XMMSV_TYPE_DICT)) {
675  return xmmsv_ref (value);
676  }
677 
678  x_api_error_if (!xmmsv_get_string (value, &key),
679  "order entry must be string or dict", NULL);
680 
681  order = xmmsv_new_dict ();
682 
683  if (key[0] == '-') {
684  xmmsv_dict_set_string (order, "direction", "DESC");
685  key++;
686  }
687 
688  if (strcmp (key, "random") == 0) {
689  xmmsv_dict_set_string (order, "type", "random");
690  } else if (strcmp (key, "id") == 0) {
691  xmmsv_dict_set_string (order, "type", "id");
692  } else {
693  xmmsv_dict_set_string (order, "type", "value");
694  xmmsv_dict_set_string (order, "field", key);
695  }
696 
697  return order;
698 }

701 /**
702  * Return a collection with an order-operator added.
703  *
704  * @param coll the original collection
705  * @param value an ordering string, optionally starting with "-" (for
706  * descending ordering), followed by a string "id", "random"
707  * or a key identifying a property, such as "artist" or "album".
708  * Or it can be a dict containing the attributes to set.
709  * @return coll with order-operators added
710  */
711 xmmsv_t *
712 xmmsv_coll_add_order_operator (xmmsv_t *coll, xmmsv_t *value)
713 {
714  value = xmmsv_coll_normalize_order_arguments (value);
715  if (value != NULL) {
716  xmmsv_t *ordered;
717 
718  ordered = xmmsv_new_coll (XMMS_COLLECTION_TYPE_ORDER);
719  xmmsv_coll_add_operand (ordered, coll);
720  xmmsv_coll_attributes_set (ordered, value);
721 
722  xmmsv_unref (value);
723 
724  return ordered;
725  }
726 
727  return xmmsv_ref (coll);
728 }

730 /**
731  * Return a collection with several order-operators added.
732  *
733  * @param coll the original collection
734  * @param order list of ordering strings or dicts.
735  *
736  * @return coll with order-operators added
737  */
738 xmmsv_t *
739 xmmsv_coll_add_order_operators (xmmsv_t *coll, xmmsv_t *order)
740 {
741  xmmsv_list_iter_t *it;
742  xmmsv_t *current, *value;
743 
744  x_api_error_if (coll == NULL, "with a NULL coll", NULL);
745 
746  xmmsv_ref (coll);
747 
748  if (!order) {
749  return coll;
750  }
751 
752  x_api_error_if (!xmmsv_is_type (order, XMMSV_TYPE_LIST),
753  "with a non list order", coll);
754 
755  current = coll;
756 
757  xmmsv_get_list_iter (order, &it);
758  xmmsv_list_iter_last (it);
759 
760  while (xmmsv_list_iter_entry (it, &value)) {
761  xmmsv_t *ordered;
762 
763  ordered = xmmsv_coll_add_order_operator (current, value);
764  xmmsv_unref (current);
765 
766  current = ordered;
767  xmmsv_list_iter_prev (it);
768  }
769 
770  return current;
771 }

794  xmmsv_coll_add_operand (ret, coll);

799  xmmsv_coll_attribute_set_string (ret, "start", str);

802  }
803  }

809  } else {
810  x_api_warning ("could not set collection limit operator length");
811  }
812  }
