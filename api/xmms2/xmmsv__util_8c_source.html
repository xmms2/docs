25 static int
26 source_match_pattern (const char *source, const char *pattern)
27 {
28  int match = 0;
29  int lpos = strlen (pattern) - 1;
30 
31  if (strcasecmp (pattern, source) == 0) {
32  match = 1;
33  } else if (lpos >= 0 && pattern[lpos] == '*' &&
34  (lpos == 0 || strncasecmp (source, pattern, lpos) == 0)) {
35  match = 1;
36  }
37 
38  return match;
39 }

59 /**
60  * Helper function to transform a key-source-value dict-of-dict
61  * #xmmsv_t (formerly a propdict) to a regular key-value dict, given a
62  * list of source preference.
63  *
64  * @param propdict A key-source-value dict-of-dict #xmmsv_t.
65  * @param src_prefs A list of source names or patterns. Must be
66  * NULL-terminated. If this argument is NULL, the
67  * default source preferences is used.
68  * @return An #xmmsv_t containing a simple key-value dict. Must be
69  * unreffed manually when done.
70  */
71 xmmsv_t *
72 xmmsv_propdict_to_dict (xmmsv_t *propdict, const char **src_prefs)
73 {
74  xmmsv_t *dict, *source_dict, *value, *best_value;
75  xmmsv_dict_iter_t *key_it, *source_it;
76  const char *key, *source;
77  const char **local_prefs;
78  int match_index, best_index;
79 
80  dict = xmmsv_new_dict ();
81 
82  local_prefs = src_prefs ? src_prefs : xmmsv_default_source_pref;
83 
84  xmmsv_get_dict_iter (propdict, &key_it);
85  while (xmmsv_dict_iter_pair (key_it, &key, &source_dict)) {
86  best_value = NULL;
87  best_index = -1;
88  xmmsv_get_dict_iter (source_dict, &source_it);
89  while (xmmsv_dict_iter_pair (source_it, &source, &value)) {
90  match_index = find_match_index (source, local_prefs);
91  /* keep first match or better match */
92  if (match_index >= 0 && (best_index < 0 ||
93  match_index < best_index)) {
94  best_value = value;
95  best_index = match_index;
96  }
97  xmmsv_dict_iter_next (source_it);
98  }
99 
100  /* Note: we do not insert a key-value pair if no source matches */
101  if (best_value) {
102  xmmsv_dict_set (dict, key, best_value);
103  }
104 
105  xmmsv_dict_iter_next (key_it);
106  }
107 
108  return dict;
109 }

137 /**
138  * Encodes an url with arguments stored in dict args.
139  *
140  * The encoded url is allocated using malloc and has to be freed by the user.
141  *
142  * @param url The url to encode.
143  * @param args The dict with arguments, or NULL.
144  * @return The encoded url
145  */
146 char *
147 xmmsv_encode_url_full (const char *url, xmmsv_t *args)
148 {
149  static const char hex[16] = "0123456789abcdef";
150  int i = 0, j = 0, extra = 0, l;
151  char *res;
152 
153  x_api_error_if (!url, "with a NULL url", NULL);
154 
155  if (args) {
156  if (!xmmsv_dict_foreach (args, _sum_len_string_dict, (void *) &extra)) {
157  return NULL;
158  }
159  }
160 
161  /* Provide enough room for the worst-case scenario (all characters of the
162  URL must be encoded), the args, and a \0. */
163  res = malloc (strlen (url) * 3 + 1 + extra);
164  if (!res) {
165  return NULL;
166  }
167 
168  for (i = 0; url[i]; i++) {
169  unsigned char chr = url[i];
170  if (GOODCHAR (chr)) {
171  res[j++] = chr;
172  } else if (chr == ' ') {
173  res[j++] = '+';
174  } else {
175  res[j++] = '%';
176  res[j++] = hex[((chr & 0xf0) >> 4)];
177  res[j++] = hex[(chr & 0x0f)];
178  }
179  }
180 
181  if (args) {
182  xmmsv_dict_iter_t *it;
183  const char *arg, *key;
184  xmmsv_t *val;
185 
186  for (xmmsv_get_dict_iter (args, &it), i = 0;
187  xmmsv_dict_iter_pair (it, &key, &val);
188  xmmsv_dict_iter_next (it), i++) {
189 
190  l = strlen (key);
191  res[j] = (i == 0) ? '?' : '&';
192  j++;
193  memcpy (&res[j], key, l);
194  j += l;
195  if (xmmsv_get_string (val, &arg)) {
196  l = strlen (arg);
197  res[j] = '=';
198  j++;
199  memcpy (&res[j], arg, l);
200  j += l;
201  }
202  }
203  }
204 
205  res[j] = '\0';
206 
207  return res;
208 }

224 /**
225  * Decode an URL-encoded string.
226  *
227  * Some strings (currently only the url of media) has no known
228  * encoding, and must be encoded in an UTF-8 clean way. This is done
229  * similar to the url encoding web browsers do. This functions decodes
230  * a string encoded in that way. OBSERVE that the decoded string HAS
231  * NO KNOWN ENCODING and you cannot display it on screen in a 100%
232  * guaranteed correct way (a good heuristic is to try to validate the
233  * decoded string as UTF-8, and if it validates assume that it is an
234  * UTF-8 encoded string, and otherwise fall back to some other
235  * encoding).
236  *
237  * Do not use this function if you don't understand the
238  * implications. The best thing is not to try to display the url at
239  * all.
240  *
241  * Note that the fact that the string has NO KNOWN ENCODING and CAN
242  * NOT BE DISPLAYED does not stop you from open the file if it is a
243  * local file (if it starts with "file://").
244  *
245  * @param url the #xmmsv_t containing a url-encoded string
246  * @return a new #xmmsv_t containing the decoded string as a XMMSV_BIN or NULL on failure
247  *
248  */
249 xmmsv_t *
250 xmmsv_decode_url (const xmmsv_t *inv)
251 {
252  int i = 0, j = 0;
253  const char *ins;
254  unsigned char *url;
255  xmmsv_t *ret;
256 
257  if (!xmmsv_get_string (inv, &ins)) {
258  return NULL;
259  }
260 
261  url = x_malloc (strlen (ins));
262  if (!url) {
263  x_oom ();
264  return NULL;
265  }
266 
267  while (ins[i]) {
268  unsigned char chr = ins[i++];
269 
270  if (chr == '+') {
271  chr = ' ';
272  } else if (chr == '%') {
273  char ts[3];
274  char *t;
275 
276  ts[0] = ins[i++];
277  if (!ts[0])
278  goto err;
279  ts[1] = ins[i++];
280  if (!ts[1])
281  goto err;
282  ts[2] = '\0';
283 
284  chr = strtoul (ts, &t, 16);
285 
286  if (t != &ts[2])
287  goto err;
288  }
289 
290  url[j++] = chr;
291  }
292 
293  ret = xmmsv_new_bin (url, j);
294  free (url);
295 
296  return ret;
297 
298 err:
299  free (url);
300  return NULL;
301 }

304  * This function will make a pretty string about the information in
305  * xmmsv dict.
306  *
307  * @param target A allocated char *
308  * @param len Length of target
309  * @param fmt A format string to use. You can insert items from the dict by

315 int
316 xmmsv_dict_format (char *target, int len, const char *fmt, xmmsv_t *val)
317 {
318  const char *pos;
319 
320  if (!target) {
321  return 0;
322  }
323 
324  if (!fmt) {
325  return 0;
326  }
327 
328  memset (target, 0, len);
329 
330  pos = fmt;
331  while (strlen (target) + 1 < len) {
332  char *next_key, *key, *end;
333  int keylen;
334  xmmsv_dict_iter_t *it;
335  xmmsv_t *v;
336 
337  next_key = strstr (pos, "${");
338  if (!next_key) {
339  strncat (target, pos, len - strlen (target) - 1);
340  break;
341  }
342 
343  strncat (target, pos, MIN (next_key - pos, len - strlen (target) - 1));
344  keylen = strcspn (next_key + 2, "}");
345  key = malloc (keylen + 1);
346 
347  if (!key) {
348  fprintf (stderr, "Unable to allocate %u bytes of memory, OOM?", keylen);
349  break;
350  }
351 
352  memset (key, 0, keylen + 1);
353  strncpy (key, next_key + 2, keylen);
354 
355  xmmsv_get_dict_iter (val, &it);
356 
357  if (strcmp (key, "seconds") == 0) {
358  int64_t duration;
359 
360  if (xmmsv_dict_iter_find (it, "duration")) {
361  xmmsv_dict_iter_pair (it, NULL, &v);
362  xmmsv_get_int (v, &duration);
363  } else {
364  duration = 0;
365  }
366 
367  if (!duration) {
368  strncat (target, "00", len - strlen (target) - 1);
369  } else {
370  char seconds[21];
371  /* rounding */
372  duration += 500;
373  snprintf (seconds, sizeof (seconds), "%02" PRId64, (duration/1000)%60);
374  strncat (target, seconds, len - strlen (target) - 1);
375  }
376  } else if (strcmp (key, "minutes") == 0) {
377  int64_t duration;
378 
379  if (xmmsv_dict_iter_find (it, "duration")) {
380  xmmsv_dict_iter_pair (it, NULL, &v);
381  xmmsv_get_int (v, &duration);
382  } else {
383  duration = 0;
384  }
385 
386  if (!duration) {
387  strncat (target, "00", len - strlen (target) - 1);
388  } else {
389  char minutes[21];
390  /* rounding */
391  duration += 500;
392  snprintf (minutes, sizeof (minutes), "%02" PRId64, duration/60000);
393  strncat (target, minutes, len - strlen (target) - 1);
394  }
395  } else {
396  const char *result = NULL;
397  char tmp[21];
398 
399  if (xmmsv_dict_iter_find (it, key)) {
400  xmmsv_dict_iter_pair (it, NULL, &v);
401 
402  xmmsv_type_t type = xmmsv_get_type (v);
403  if (type == XMMSV_TYPE_STRING) {
404  xmmsv_get_string (v, &result);
405  } else if (type == XMMSV_TYPE_INT64) {
406  int64_t i;
407  xmmsv_get_int (v, &i);
408  snprintf (tmp, 21, "%" PRId64, i);
409  result = tmp;
410  } else if (type == XMMSV_TYPE_FLOAT) {
411  float f;
412  xmmsv_get_float (v, &f);
413  snprintf (tmp, 12, "%.6f", f);
414  result = tmp;
415  }
416  }
417 
418  if (result)
419  strncat (target, result, len - strlen (target) - 1);
420  }
421 
422  free (key);
423  end = strchr (next_key, '}');
424 
425  if (!end) {
426  break;
427  }
428 
429  pos = end + 1;
430  }
431 
432  return strlen (target);
433 }

436 _xmmsv_utf8_charlen (unsigned char c)
437 {
438  if ((c & 0x80) == 0) {
439  return 1;
440  } else if ((c & 0x60) == 0x40) {
441  return 2;
442  } else if ((c & 0x70) == 0x60) {
443  return 3;
444  } else if ((c & 0x78) == 0x70) {
445  return 4;
446  }
447  return 0;

472  return 0;
473  }
