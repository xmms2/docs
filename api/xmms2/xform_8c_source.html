168 void
169 xmms_xform_browse_add_entry (xmms_xform_t *xform, const gchar *filename,
170  guint32 flags)
171 {
172  const gchar *url;
173  gchar *efile, *eurl, *t;
174  gint l, isdir;
175 
176  g_return_if_fail (filename);
177 
178  t = strchr (filename, '/');
179  g_return_if_fail (!t); /* filenames can't contain '/', can they? */
180 
181  url = xmms_xform_get_url (xform);
182  g_return_if_fail (url);
183 
184  xform->browse_dict = xmmsv_new_dict ();
185 
186  eurl = xmms_medialib_url_encode (url);
187  efile = xmms_medialib_url_encode (filename);
188 
189  /* can't use g_build_filename as we need to preserve
190  slashes stuff like file:/// */
191  l = strlen (url);
192  if (l && url[l - 1] == '/') {
193  t = g_strdup_printf ("%s%s", eurl, efile);
194  } else {
195  t = g_strdup_printf ("%s/%s", eurl, efile);
196  }
197 
198  isdir = !!(flags & XMMS_XFORM_BROWSE_FLAG_DIR);
199  xmms_xform_browse_add_entry_property_str (xform, "path", t);
200  xmms_xform_browse_add_entry_property_int (xform, "isdir", isdir);
201 
202  if (xform->browse_list == NULL) {
203  xform->browse_list = xmmsv_new_list ();
204  }
205  xmmsv_list_append (xform->browse_list, xform->browse_dict);
206  xmmsv_unref (xform->browse_dict);
207 
208  g_free (t);
209  g_free (efile);
210  g_free (eurl);
211 }

213 static gint
214 xmms_browse_list_sortfunc (xmmsv_t **a, xmmsv_t **b)
215 {
216  xmmsv_t *val1, *val2;
217  const gchar *s1, *s2;
218  int r1, r2;
219 
220  g_return_val_if_fail (xmmsv_is_type (*a, XMMSV_TYPE_DICT), 0);
221  g_return_val_if_fail (xmmsv_is_type (*b, XMMSV_TYPE_DICT), 0);
222 
223  r1 = xmmsv_dict_get (*a, "intsort", &val1);
224  r2 = xmmsv_dict_get (*b, "intsort", &val2);
225 
226  if (r1 && r2) {
227  gint i1, i2;
228 
229  if (!xmmsv_get_int (val1, &i1))
230  return 0;
231  if (!xmmsv_get_int (val2, &i2))
232  return 0;
233  return i1 > i2;
234  }
235 
236  if (!xmmsv_dict_get (*a, "path", &val1))
237  return 0;
238  if (!xmmsv_dict_get (*b, "path", &val2))
239  return 0;
240 
241  if (!xmmsv_get_string (val1, &s1))
242  return 0;
243  if (!xmmsv_get_string (val2, &s2))
244  return 0;
245 
246  return xmms_natcmp (s1, s2);
247 }

842 {

909 gint
910 xmms_xform_this_read (xmms_xform_t *xform, gpointer buf, gint siz,
911  xmms_error_t *err)
912 {
913  gint read = 0;
914  gint nexths;
915 
916  if (xform->error) {
917  xmms_error_set (err, XMMS_ERROR_GENERIC, "Read on errored xform");
918  return -1;
919  }
920 
921  /* update hotspots */
922  nexths = xmms_xform_hotspots_update (xform);
923  if (nexths >= 0) {
924  siz = MIN (siz, nexths);
925  }
926 
927  if (xform->buffered) {
928  read = MIN (siz, xform->buffered);
929  memcpy (buf, xform->buffer, read);
930  xform->buffered -= read;
931 
932  /* buffer edited, update hotspot positions */
933  g_queue_foreach (xform->hotspots, &xmms_xform_hotspot_callback, &read);
934 
935  if (xform->buffered) {
936  /* unless we are _peek:ing often
937  this should be fine */
938  memmove (xform->buffer, &xform->buffer[read], xform->buffered);
939  }
940  }
941 
942  if (xform->eos) {
943  return read;
944  }
945 
946  while (read < siz) {
947  gint res;
948 
949  res = xmms_xform_plugin_read (xform->plugin, xform, buf + read, siz - read, err);
950  if (xform->metadata_collected && xform->metadata_changed)
951  xmms_xform_metadata_update (xform);
952 
953  if (res < -1) {
954  XMMS_DBG ("Read method of %s returned bad value (%d) - BUG IN PLUGIN", xmms_xform_shortname (xform), res);
955  res = -1;
956  }
957 
958  if (res == 0) {
959  xform->eos = TRUE;
960  break;
961  } else if (res == -1) {
962  xform->error = TRUE;
963  return -1;
964  } else {
965  if (read == 0)
966  xmms_xform_hotspots_update (xform);
967 
968  if (!g_queue_is_empty (xform->hotspots)) {
969  if (xform->buffered + res > xform->buffersize) {
970  xform->buffersize = MAX (xform->buffersize * 2,
971  xform->buffersize + res);
972  xform->buffer = g_realloc (xform->buffer,
973  xform->buffersize);
974  }
975 
976  g_memmove (xform->buffer + xform->buffered, buf + read, res);
977  xform->buffered += res;
978  break;
979  }
980  read += res;
981  }
982  }
983 
984  return read;
985 }
