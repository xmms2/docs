73 gboolean
74 xmms_medialib_session_commit (xmms_medialib_session_t *session)
75 {
76  GHashTableIter iter;
77  gpointer key;
78 
79  if (!s4_commit (session->trans)) {
80  xmms_medialib_session_free_full (session);
81  return FALSE;
82  }
83 
84  if (session->added != NULL) {
85  g_hash_table_iter_init (&iter, session->added);
86 
87  while (g_hash_table_iter_next (&iter, &key, NULL)) {
88  xmms_medialib_entry_send_added (session->medialib,
89  GPOINTER_TO_INT (key));
90  if (session->updated != NULL)
91  g_hash_table_remove (session->updated, key);
92  }
93  }
94 
95  if (session->removed != NULL) {
96  g_hash_table_iter_init (&iter, session->removed);
97 
98  while (g_hash_table_iter_next (&iter, &key, NULL)) {
99  xmms_medialib_entry_send_removed (session->medialib,
100  GPOINTER_TO_INT (key));
101  if (session->updated != NULL)
102  g_hash_table_remove (session->updated, key);
103  }
104  }
105 
106 
107  if (session->updated != NULL) {
108  g_hash_table_iter_init (&iter, session->updated);
109 
110  while (g_hash_table_iter_next (&iter, &key, NULL)) {
111  xmms_medialib_entry_send_update (session->medialib,
112  GPOINTER_TO_INT (key));
113  }
114  }
115 
116  xmms_medialib_session_free (session);
117 
118  return TRUE;
119 }

135 gint
136 xmms_medialib_session_property_set (xmms_medialib_session_t *session,
137  xmms_medialib_entry_t entry,
138  const gchar *key,
139  const s4_val_t *value,
140  const gchar *source)
141 {
142  const gchar *sources[2] = { source, NULL };
143  const s4_result_t *res;
144  s4_condition_t *cond;
145  s4_fetchspec_t *spec;
146  s4_resultset_t *set;
147  s4_sourcepref_t *sp;
148  s4_val_t *song_id;
149  gint result;
150  GHashTable *events;
151 
152  song_id = s4_val_new_int (entry);
153 
154  sp = s4_sourcepref_create (sources);
155 
156  cond = s4_cond_new_filter (S4_FILTER_EQUAL, "song_id", song_id,
157  sp, S4_CMP_CASELESS, S4_COND_PARENT);
158 
159  spec = s4_fetchspec_create ();
160  s4_fetchspec_add (spec, key, sp, S4_FETCH_DATA);
161 
162  set = s4_query (session->trans, spec, cond);
163  s4_cond_free (cond);
164  s4_fetchspec_free (spec);
165 
166  res = s4_resultset_get_result (set, 0, 0);
167  if (res != NULL) {
168  const s4_val_t *old_value = s4_result_get_val (res);
169  s4_del (session->trans, "song_id", song_id,
170  key, old_value, source);
171  }
172 
173  s4_resultset_free (set);
174  s4_sourcepref_unref (sp);
175 
176  result = s4_add (session->trans, "song_id", song_id,
177  key, value, source);
178 
179  s4_val_free (song_id);
180 
181  if (strcmp (key, XMMS_MEDIALIB_ENTRY_PROPERTY_URL) == 0) {
182  events = xmms_medialib_session_get_table (&session->added);
183  } else {
184  events = xmms_medialib_session_get_table (&session->updated);
185  }
186 
187  g_hash_table_insert (events,
188  GINT_TO_POINTER (entry),
189  GINT_TO_POINTER (entry));
190 
191  return result;
192 }
