101 /**
102  * Gets a pointer to a constant string value with a string equal to str.
103  * _string_lookup_val will always return the same value for the same string.
104  *
105  * @param s4 The database to look for the string in
106  * @param str The string to find the constant string of
107  * @return A pointer to a string value
108  */
109 const s4_val_t *_string_lookup_val (s4_t *s4, const char *str)
110 {
111  s4_val_t *ret;
112 
113  g_mutex_lock (&s4->const_data->strings_lock);
114 
115  ret = g_hash_table_lookup (s4->const_data->strings_table, str);
116  if (ret == NULL) {
117  str = g_string_chunk_insert (s4->const_data->strings, str);
118  ret = s4_val_new_internal_string (str, s4);
119  g_hash_table_insert (s4->const_data->strings_table, (void*)str, ret);
120  }
121 
122  g_mutex_unlock (&s4->const_data->strings_lock);
123 
124  return ret;
125 }

149 /**
150  * Gets the casefolded string corresponding to str. str must have
151  * been obtained by calling _string_lookup.

154  * @param str The string to find the casefold string of
155  * @return The casefolded string of str

157 const char *_string_lookup_casefolded (s4_t *s4, const char *str)
158 {
159  const char *ret;
160 
161  g_mutex_lock (&s4->const_data->case_lock);
162  ret = g_hash_table_lookup (s4->const_data->case_table, str);
163 
164  if (ret == NULL) {

206 {

209  g_mutex_lock (&s4->const_data->int_lock);
