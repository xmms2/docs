83 {

87  gint priority;
88 
89  va_start (ap, plugin);
90  t = xmms_stream_type_parse (ap);
91  va_end (ap);
92 
93  config_key = g_strconcat ("priority.",
94  xmms_stream_type_get_str (t, XMMS_STREAM_TYPE_NAME),

96  priority = xmms_stream_type_get_int (t, XMMS_STREAM_TYPE_PRIORITY);
97  g_snprintf (config_value, sizeof (config_value), "%d", priority);
98  xmms_xform_plugin_config_property_register (plugin, config_key,
99  config_value, NULL, NULL);
100  g_free (config_key);
101 
102  plugin->in_types = g_list_prepend (plugin->in_types, t);
103 }

122 gboolean
123 xmms_xform_plugin_supports (const xmms_xform_plugin_t *plugin, const xmms_stream_type_t *st,
124  gint *priority)
125 {
126  GList *t;
127 
128  g_return_val_if_fail (st, FALSE);
129  g_return_val_if_fail (plugin, FALSE);
130  g_return_val_if_fail (priority, FALSE);
131 
132  for (t = plugin->in_types; t; t = g_list_next (t)) {
133  xmms_config_property_t *config_priority;
134  const gchar *type_name;
135  gchar *config_key;
136 
137  if (!xmms_stream_type_match (t->data, st)) {
138  continue;
139  }
140 
141  type_name = xmms_stream_type_get_str (t->data, XMMS_STREAM_TYPE_NAME);
142 
143  config_key = g_strconcat ("priority.", type_name, NULL);
144  config_priority = xmms_plugin_config_lookup ((xmms_plugin_t *) plugin,
145  config_key);
146  g_free (config_key);
147 
148  if (config_priority) {
149  *priority = xmms_config_property_get_int (config_priority);
150  } else {
151  *priority = XMMS_STREAM_TYPE_PRIORITY_DEFAULT;
152  }
153 
154  return TRUE;
155  }
156 
157  return FALSE;
158 }
