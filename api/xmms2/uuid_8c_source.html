32 /**
33  * Creates a new random UUID and saves it in the array passed
34  *
35  * @param uuid The array to save the new UUID to
36  */
37 void s4_create_uuid (unsigned char uuid[16])
38 {
39  int i;
40  uint32_t *rnd = (uint32_t *)uuid;
41 
42  for (i = 0; i < 4; i++) {
43  rnd[i] = g_random_int ();
44  }
45 
46  /* Set type to random */
47  uuid[6] &= 0xf;
48  uuid[6] |= 0x40;
49  /* Set variant */
50  uuid[8] &= 0x3f;
51  uuid[8] |= 0x80;
52 }

70 /**
71  * Gets the UUID-string of a S4 database
72  *
73  * @param s4 The database to find the UUID of
74  * @return A NUL-terminated string with the UUID. Must be freed
75  */
76 char *s4_get_uuid_string (s4_t* s4)
77 {
78  int i, j;
79  char *uuid_str = malloc (37);
80 
81  for (i = 0, j = 0; i < 16; i++) {
82  uuid_str[j++] = HEX_TO_CHAR (s4->uuid[i] >> 4);
83  uuid_str[j++] = HEX_TO_CHAR (s4->uuid[i] & 0xf);
84 
85  if (i == 3 || i == 5 || i == 7 || i == 9) {
86  uuid_str[j++] = '-';
87  }
88  }
89  uuid_str[j] = '\0';
90 
91  return uuid_str;
92 }
