75 int
76 xmmsv_bitbuffer_get_bits (xmmsv_t *v, int bits, int64_t *res)
77 {
78  int64_t r, t;
79  int i;
80 
81  x_api_error_if (bits < 1, "less than one bit requested", 0);
82 
83  if (bits == 1) {
84  int pos = v->value.bit.pos;
85 
86  if (pos >= v->value.bit.len)
87  return 0;
88  r = (v->value.bit.buf[pos / 8] >> (7-(pos % 8)) & 1);
89  v->value.bit.pos += 1;
90  *res = r;
91  return 1;
92  }
93 
94  r = 0;
95  for (i = 0; i < bits; i++) {
96  t = 0;
97  if (!xmmsv_bitbuffer_get_bits (v, 1, &t))
98  return 0;
99  r = (r << 1) | t;
100  }
101  *res = r;
102  return 1;
103 }

119 int
120 xmmsv_bitbuffer_put_bits (xmmsv_t *v, int bits, int64_t d)
121 {
122  unsigned char t;
123  int pos;
124  int i;
125 
126  x_api_error_if (v->value.bit.ro, "write to readonly bitbuffer", 0);
127  x_api_error_if (bits < 1, "less than one bit requested", 0);
128 
129  if (bits == 1) {
130  pos = v->value.bit.pos;
131 
132  if (pos >= v->value.bit.alloclen) {
133  int ol, nl;
134  nl = v->value.bit.alloclen * 2;
135  ol = v->value.bit.alloclen;
136  nl = nl < 128 ? 128 : nl;
137  nl = (nl + 7) & ~7;
138  v->value.bit.buf = realloc (v->value.bit.buf, nl / 8);
139  memset (v->value.bit.buf + ol / 8, 0, (nl - ol) / 8);
140  v->value.bit.alloclen = nl;
141  }
142  t = v->value.bit.buf[pos / 8];
143 
144  t = (t & (~(1<<(7-(pos % 8))))) | (d << (7-(pos % 8)));
145 
146  v->value.bit.buf[pos / 8] = t;
147 
148  v->value.bit.pos += 1;
149  if (v->value.bit.pos > v->value.bit.len)
150  v->value.bit.len = v->value.bit.pos;
151  return 1;
152  }
153 
154  for (i = 0; i < bits; i++) {
155  if (!xmmsv_bitbuffer_put_bits (v, 1, !!(d & (1LL << (bits-i-1)))))
156  return 0;
157  }
158 
159  return 1;
160 }
