179 /**
180  * Try to write message to transport. If full message isn't written
181  * the message will keep track of the amount of data written and not
182  * write already written data next time.
183  *
184  * @returns TRUE if full message was written, FALSE otherwise.
185  * disconnected is set if transport was disconnected
186  */
187 bool
188 xmms_ipc_msg_write_transport (xmms_ipc_msg_t *msg,
189  xmms_ipc_transport_t *transport,
190  bool *disconnected)
191 {
192  char *buf;
193  unsigned int ret, len;
194 
195  x_return_val_if_fail (msg, false);
196  x_return_val_if_fail (transport, false);
197 
198  xmmsv_bitbuffer_align (msg->bb);
199 
200  len = xmmsv_bitbuffer_len (msg->bb) / 8;
201 
202  x_return_val_if_fail (len > msg->xfered, true);
203 
204  buf = (char *) (xmmsv_bitbuffer_buffer (msg->bb) + msg->xfered);
205  ret = xmms_ipc_transport_write (transport, buf, len - msg->xfered);
206 
207  if (ret == SOCKET_ERROR) {
208  if (xmms_socket_error_recoverable ()) {
209  return false;
210  }
211 
212  if (disconnected) {
213  *disconnected = true;
214  }
215 
216  return false;
217  } else if (!ret) {
218  if (disconnected) {
219  *disconnected = true;
220  }
221  } else {
222  msg->xfered += ret;
223  }
224 
225  return (len == msg->xfered);
226 }

228 /**
229  * Try to read message from transport into msg.
230  *
231  * @returns TRUE if message is fully read.
232  */
233 bool
234 xmms_ipc_msg_read_transport (xmms_ipc_msg_t *msg,
235  xmms_ipc_transport_t *transport,
236  bool *disconnected)
237 {
238  char buf[512];
239  unsigned int ret, len, rlen;
240 
241  x_return_val_if_fail (msg, false);
242  x_return_val_if_fail (transport, false);
243 
244  while (true) {
245  len = XMMS_IPC_MSG_HEAD_LEN;
246 
247  if (msg->xfered >= XMMS_IPC_MSG_HEAD_LEN) {
248  len += xmms_ipc_msg_get_length (msg);
249 
250  if (msg->xfered == len) {
251  return true;
252  }
253  }
254 
255  x_return_val_if_fail (msg->xfered < len, false);
256 
257  rlen = len - msg->xfered;
258  if (rlen > sizeof (buf))
259  rlen = sizeof (buf);
260 
261  ret = xmms_ipc_transport_read (transport, buf, rlen);
262 
263  if (ret == SOCKET_ERROR) {
264  if (xmms_socket_error_recoverable ()) {
265  return false;
266  }
267 
268  if (disconnected) {
269  *disconnected = true;
270  }
271 
272  return false;
273  } else if (ret == 0) {
274  if (disconnected) {
275  *disconnected = true;
276  }
277 
278  return false;
279  } else {
280  xmmsv_bitbuffer_goto (msg->bb, msg->xfered * 8);
281  xmmsv_bitbuffer_put_data (msg->bb, (unsigned char *) buf, ret);
282  msg->xfered += ret;
283  xmmsv_bitbuffer_goto (msg->bb, XMMS_IPC_MSG_HEAD_LEN * 8);
284  }
285  }
286 }
