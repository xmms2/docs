118 /**
119  * Commits a transaction. On success the operations in the transactions
120  * will be applied in one atomic step, on error none of the operations
121  * in the transaction will be applied.
122  *
123  * @param trans The transaction to commit.
124  * @return 0 on error (and sets s4_errno), non-zero on success.
125  */
126 int s4_commit (s4_transaction_t *trans)
127 {
128  int ret = 0;
129  int need_sync = 0;
130  s4_t *s4 = _transaction_get_db (trans);
131 
132  if (trans->failed) {
133  s4_set_errno (trans->error_code);
134  } else {
135  ret = _log_write (trans->ops);
136 
137  if (ret == 0) {
138  need_sync = 1;
139  s4_set_errno (S4E_LOGFULL);
140  }
141  }
142 
143  if (ret == 0) {
144  _oplist_last (trans->ops);
145  _oplist_rollback (trans->ops);
146  }
147 
148  _log_unlock_file (s4);
149  _transaction_free (trans);
150 
151  if (need_sync) {
152  _sync (s4);
153  }
154 
155  return ret;
156 }

179 /**
180  * Adds a relationship to the database.
181  * It takes both a database handle and a transaction handle,
182  * but only one if used. If the transaction handle is NULL it
183  * will create a local transaction, otherwise it will use the
184  * transaction passed, and thus s4 is not needed.
185  *
186  * @param s4 The database to add to.
187  * @param trans The transaction to use.
188  * @param key_a Key A.
189  * @param val_a Value A.
190  * @param key_b Key B.
191  * @param val_b Value B.
192  * @param src Source.
193  * @return 0 on error, non-zero on success.
194  */
195 int s4_add (s4_transaction_t *trans,
196  const char *key_a, const s4_val_t *val_a,
197  const char *key_b, const s4_val_t *val_b,
198  const char *src)
199 {
200  int ret;
201  s4_t *db = _transaction_get_db (trans);
202 
203  if (trans->flags & S4_TRANS_READONLY) {
204  trans->failed = 1;
205  trans->error_code = S4E_READONLY;
206  return 0;
207  }
208 
209  key_a = _string_lookup (db, key_a);
210  key_b = _string_lookup (db, key_b);
211  src = _string_lookup (db, src);
212  val_a = _const_lookup (db, val_a);
213  val_b = _const_lookup (db, val_b);
214 
215  if (trans->failed) {
216  ret = 0;
217  } else {
218  _oplist_insert_add (trans->ops, key_a, val_a, key_b, val_b, src);
219  ret = _s4_add (trans, key_a, val_a, key_b, val_b, src);
220 
221  if (!ret) {
222  trans->failed = 1;
223  trans->error_code = S4E_EXECUTE;
224  }
225  }
226 
227  return ret;
228 }
