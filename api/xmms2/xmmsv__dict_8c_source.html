50 /* MurmurHash2, by Austin Appleby */
51 static uint32_t
52 _xmmsv_dict_hash (const void *key, int len)
53 {
54  /* 'm' and 'r' are mixing constants generated offline.
55  * They're not really 'magic', they just happen to work well.
56  */
57  const uint32_t seed = 0x12345678;
58  const uint32_t m = 0x5bd1e995;
59  const int r = 24;
60 
61  /* Initialize the hash to a 'random' value */
62  uint32_t h = seed ^ len;
63 
64  /* Mix 4 bytes at a time into the hash */
65  const unsigned char * data = (const unsigned char *)key;
66 
67  while (len >= 4)
68  {
69  uint32_t k;
70  memcpy (&k, data, sizeof (k));
71 
72  k *= m;
73  k ^= k >> r;
74  k *= m;
75 
76  h *= m;
77  h ^= k;
78 
79  data += 4;
80  len -= 4;
81  }
82 
83  /* Handle the last few bytes of the input array */
84  switch (len)
85  {
86  case 3: h ^= data[2] << 16;
87  case 2: h ^= data[1] << 8;
88  case 1: h ^= data[0];
89  h *= m;
90  };
91 
92  /* Do a few final mixes of the hash to ensure the last few
93  * bytes are well-incorporated.
94  */
95  h ^= h >> 13;
96  h *= m;
97  h ^= h >> 15;
98 
99  return h;
100 }

102 /* Searches the hash table for an entry matching the hash and string in data.
103  * It will save the found position in pos.
104  * If a deleted position was found before the key, it will be saved in deleted
105  * Returns 1 if the entry was found, 0 otherwise
106  */
107 static int
108 _xmmsv_dict_search (xmmsv_dict_internal_t *dict, xmmsv_dict_data_t data,
109  int *pos, int *deleted)
110 {
111  int bucket = data.hash & HASH_MASK (dict);
112  int stop = bucket;
113  int size = 1 << dict->size;
114 
115  *deleted = -1;
116 
117  while (dict->data[bucket].str != NULL) {
118  /* If this is a free entry we save it in the free pointer */
119  if (dict->data[bucket].str == DELETED_STR) {
120  if (*deleted == -1) {
121  *deleted = bucket;
122  }
123  /* If we found the entry we save it in the pos pointer */
124  } else if (dict->data[bucket].hash == data.hash
125  && strcmp (dict->data[bucket].str, data.str) == 0) {
126  *pos = bucket;
127  return 1;
128  }
129 
130  /* If we hit the end we roll around */
131  if (++bucket >= size)
132  bucket = 0;
133  /* If we have checked the whole table we exit */
134  if (bucket == stop)
135  break;
136  }
137 
138  /* Save the position to the first free entry
139  * (or the start bucket if we made it the whole way around
140  * without finding the entry and without finding any empty
141  * entries, but in that case free is set, since there have to
142  * be atleast 1 deleted or empty entry in the table)
143  */
144  *pos = bucket;
145  return 0;
146 }

148 /* Inserts data into the hash table */
149 static void
150 _xmmsv_dict_insert (xmmsv_dict_internal_t *dict, xmmsv_dict_data_t data, int alloc)
151 {
152  int pos, deleted;
153 
154  if (_xmmsv_dict_search (dict, data, &pos, &deleted)) {
155  /* If the key already exists we change the data*/
156  xmmsv_unref (dict->data[pos].value);
157  dict->data[pos].value = data.value;
158  } else {
159  /* Otherwise we insert a new entry */
160  if (alloc)
161  data.str = strdup (data.str);
162  dict->elems++;
163  /* If we found a deleted entry before an empty one we use the free entry */
164  if (deleted != -1) {
165  dict->data[deleted] = data;
166  } else {
167  dict->data[pos] = data;
168  }
169  }
170 }

184 /* Resizes the hash table by creating a new data table
185  * twice the size of the old one
186  */
187 static void
188 _xmmsv_dict_resize (xmmsv_dict_internal_t *dict)
189 {
190  int i;
191  xmmsv_dict_data_t *old_data;
192 
193  /* Double the table size */
194  dict->size++;
195  dict->elems = 0;
196  old_data = dict->data;
197  dict->data = x_new0 (xmmsv_dict_data_t, 1 << dict->size);
198 
199  /* Insert all the entries in the old table into the new one */
200  for (i = 0; i < (1 << (dict->size - 1)); i++) {
201  if (old_data[i].str != NULL) {
202  _xmmsv_dict_insert (dict, old_data[i], 0);
203  }
204  }
205 
206  free (old_data);
207 }

234 {

238  /* free iterators */

354 /**
355  * Get the element corresponding to the given key in the dict #xmmsv_t
356  * (if it exists). This function does not increase the refcount of
357  * the element, the reference is still owned by the dict.
358  *
359  * @param dictv A #xmmsv_t containing a dict.
360  * @param key The key in the dict.
361  * @param val Pointer set to a borrowed reference to the element
362  * corresponding to the given key in the dict.
363  * @return 1 upon success otherwise 0
364  */
365 int
366 xmmsv_dict_get (xmmsv_t *dictv, const char *key, xmmsv_t **val)
367 {
368  xmmsv_dict_internal_t *dict;
369  int ret = 0;
370  int pos, deleted;
371 
372  x_return_val_if_fail (key, 0);
373  x_return_val_if_fail (dictv, 0);
374  x_return_val_if_fail (xmmsv_is_type (dictv, XMMSV_TYPE_DICT), 0);
375 
376  xmmsv_dict_data_t data = DICT_INIT_DATA (key);
377  dict = dictv->value.dict;
378 
379  if (_xmmsv_dict_search (dict, data, &pos, &deleted)) {
380  /* If there was a deleted entry before the one we found
381  * we can optimize a little by moving the entry to the
382  * deleted slot (and thus closer to the actual bucket it
383  * belongs to)
384  */
385  if (deleted != -1) {
386  dict->data[deleted] = dict->data[pos];
387  dict->data[pos].str = DELETED_STR;
388  }
389  if (val != NULL) {
390  *val = dict->data[pos].value;
391  }
392  ret = 1;
393  }
394 
395  return ret;
396 }
