73 /**
74  * Creates a new resultset where the order is the same as in the idlist
75  *
76  * @param set The resultset to sort. It will be freed by this function
77  * @param idlist The idlist to order by
78  * @return A new set with the same order as the idlist
79  */
80 static s4_resultset_t *
81 xmms_medialib_result_sort_idlist (s4_resultset_t *set, xmmsv_t *idlist)
82 {
83  const s4_resultrow_t *row;
84  const s4_result_t *result;
85  GHashTable *row_table;
86  s4_resultset_t *ret;
87  gint32 ival, i;
88 
89  row_table = g_hash_table_new (NULL, NULL);
90 
91  for (i = 0; s4_resultset_get_row (set, i, &row); i++) {
92  if (s4_resultrow_get_col (row, 0, &result)
93  && s4_val_get_int (s4_result_get_val (result), &ival)) {
94  g_hash_table_insert (row_table, GINT_TO_POINTER (ival), (void *) row);
95  }
96  }
97 
98  ret = s4_resultset_create (s4_resultset_get_colcount (set));
99 
100  for (i = 0; xmmsv_list_get_int (idlist, i, &ival); i++) {
101  row = g_hash_table_lookup (row_table, GINT_TO_POINTER (ival));
102  if (row != NULL) {
103  s4_resultset_add_row (ret, row);
104  }
105  }
106 
107  g_hash_table_destroy (row_table);
108  s4_resultset_free (set);

113 /**
114  * Sorts a resultset
115  *
116  * @param set The resultset to sort
117  * @param fetch The fetch-list used when set was created
118  * @param order A list with orderings. An ordering can be a string
119  * telling which column to sort by (prefixed by '-' to sort ascending)
120  * or a list of integers (an idlist).
121  * @return The set (or a new set) with the correct ordering
122  */
123 static s4_resultset_t *
124 xmms_medialib_result_sort (s4_resultset_t *set, xmms_fetch_info_t *fetch_info, xmmsv_t *order)
125 {
126  gint i, stop, type;
127  s4_order_t *s4_order;
128  xmmsv_t *val;
129 
130  /* Find the first idlist-order operand */
131  for (i = 0; xmmsv_list_get (order, i, &val); i++) {
132  xmmsv_dict_entry_get_int (val, "type", &type);
133  if (type == SORT_TYPE_LIST) {
134  xmmsv_t *idlist;
135  xmmsv_dict_get (val, "list", &idlist);
136  set = xmms_medialib_result_sort_idlist (set, idlist);
137  break;
138  }
139  }
140 
141  /* We willorder by the operands before the idlist */
142  stop = i;
143 
144  s4_order = s4_order_create ();
145 
146  for (i = 0; i < stop && xmmsv_list_get (order, i, &val); i++) {
147  xmmsv_dict_entry_get_int (val, "type", &type);
148 
149  if (type == SORT_TYPE_COLUMN) {
150  gint id, j, direction, collation;
151  s4_order_entry_t *entry;
152  xmmsv_t *ids;
153 
154  if (!xmmsv_dict_entry_get_int (val, "direction", &direction))
155  direction = S4_ORDER_ASCENDING;
156  if (!xmmsv_dict_entry_get_int (val, "collation", &collation))
157  collation = S4_CMP_COLLATE;
158 
159  xmmsv_dict_get (val, "field", &ids);
160 
161  entry = s4_order_add_column (s4_order, collation, direction);
162  for (j = 0; xmmsv_list_get_int (ids, j, &id); j++) {
163  s4_order_entry_add_choice (entry, id);
164  }
165  } else if (type == SORT_TYPE_RANDOM) {
166  gint seed;
167  if (!xmmsv_dict_entry_get_int (val, "seed", &seed))
168  seed = g_random_int_range (G_MININT32, G_MAXINT32);
169  s4_order_add_random (s4_order, seed);
170  break;
171  }
172  }
173 
174  s4_resultset_sort (set, s4_order);
175  s4_order_free (s4_order);
176 
177  return set;
178 }

205 /* Returns non-zero if the collection has an ordering, 0 otherwise */
206 static gboolean
207 has_order (xmmsv_t *coll)
208 {
209  xmmsv_t *operands, *operand;
210  gint i;
211 
212  operands = xmmsv_coll_operands_get (coll);
213 
214  switch (xmmsv_coll_get_type (coll)) {
215  /* Filter keeps the ordering of the operand */
216  case XMMS_COLLECTION_TYPE_HAS:
217  case XMMS_COLLECTION_TYPE_MATCH:
218  case XMMS_COLLECTION_TYPE_TOKEN:
219  case XMMS_COLLECTION_TYPE_EQUALS:
220  case XMMS_COLLECTION_TYPE_NOTEQUAL:
221  case XMMS_COLLECTION_TYPE_SMALLER:
222  case XMMS_COLLECTION_TYPE_SMALLEREQ:
223  case XMMS_COLLECTION_TYPE_GREATER:
224  case XMMS_COLLECTION_TYPE_GREATEREQ:
225  /* Intersection is orderded if the first operand is ordeed */
226  case XMMS_COLLECTION_TYPE_INTERSECTION:
227  xmmsv_list_get (operands, 0, &operand);
228  return has_order (operand);
229 
230  /* Union is ordered if all operands are ordered (concat) */
231  case XMMS_COLLECTION_TYPE_UNION:
232  for (i = 0; xmmsv_list_get (operands, i, &operand); i++) {
233  if (!has_order (operand))
234  return FALSE;
235  }
236 
237  /* These are always ordered */
238  case XMMS_COLLECTION_TYPE_IDLIST:
239  case XMMS_COLLECTION_TYPE_ORDER:
240  case XMMS_COLLECTION_TYPE_LIMIT:
241  return TRUE;
242 
243  case XMMS_COLLECTION_TYPE_REFERENCE:
244  if (!is_universe (coll)) {
245  xmmsv_list_get (operands, 0, &operand);
246  return has_order (operand);
247  }
248  case XMMS_COLLECTION_TYPE_COMPLEMENT:
249  case XMMS_COLLECTION_TYPE_UNIVERSE:
250  case XMMS_COLLECTION_TYPE_MEDIASET:
251  break;
252  }
253 
254  return FALSE;
255 }

325 static void
326 get_filter_type_and_compare_mode (xmmsv_t *coll,
327  s4_filter_type_t *type,
328  s4_cmp_mode_t *cmp_mode)
329 {
330  const gchar *value;
331 
332  *type = filter_type_from_collection (coll);
333 
334  if (!xmmsv_coll_attribute_get_string (coll, "collation", &value)) {
335  /* For <, <=, >= and > we default to natcoll,
336  * so that strings will order correctly
337  */
338  switch (*type) {
339  case S4_FILTER_SMALLER:
340  case S4_FILTER_GREATER:
341  case S4_FILTER_SMALLEREQ:
342  case S4_FILTER_GREATEREQ:
343  *cmp_mode = S4_CMP_COLLATE;
344  break;
345  default:
346  *cmp_mode = S4_CMP_CASELESS;
347  }
348  } else if (strcmp (value, "NOCASE") == 0) {
349  *cmp_mode = S4_CMP_CASELESS;
350  } else if (strcmp (value, "BINARY") == 0) {
351  *cmp_mode = S4_CMP_BINARY;
352  } else if (strcmp (value, "NATCOLL") == 0) {
353  *cmp_mode = S4_CMP_COLLATE;
354  } else {
355  /* Programming error, too weak validation. */
356  g_assert_not_reached ();
357  }
358 }

360 static s4_condition_t *
361 filter_condition (xmms_medialib_session_t *session,
362  xmmsv_t *coll, xmms_fetch_info_t *fetch,
363  xmmsv_t *order)
364 {
365  s4_sourcepref_t *sp;
366  s4_filter_type_t type;
367  s4_cmp_mode_t cmp_mode;
368  gint32 ival, flags = 0;
369  const gchar *filter_type, *key, *val;
370  xmmsv_t *operands, *operand;
371  s4_condition_t *cond;
372  s4_val_t *value = NULL;
373 
374  if (!xmmsv_coll_attribute_get_string (coll, "type", &filter_type) || strcmp (filter_type, "value") == 0) {
375  /* If 'field' is not set, match against every key */
376  if (!xmmsv_coll_attribute_get_string (coll, "field", &key)) {
377  key = NULL;
378  }
379  } else {
380  key = (gchar *) "song_id";
381  flags = S4_COND_PARENT;
382  }
383 
384  if (xmmsv_coll_attribute_get_string (coll, "value", &val)) {
385  gchar *endptr;
386 
387  ival = strtol (val, &endptr, 10);
388  if (endptr > val && *endptr == '\0') {
389  value = s4_val_new_int (ival);
390  } else {
391  value = s4_val_new_string (val);
392  }
393  }
394 
395  if (xmmsv_coll_attribute_get_string (coll, "source-preference", &val)) {
396  gchar **prefs;
397  prefs = g_strsplit (val, ":", -1);
398  sp = s4_sourcepref_create ((const gchar **) prefs);
399  g_strfreev (prefs);
400  } else {
401  sp = xmms_medialib_session_get_source_preferences (session);
402  }
403 
404  get_filter_type_and_compare_mode (coll, &type, &cmp_mode);
405 
406  cond = s4_cond_new_filter (type, key, value, sp, cmp_mode, flags);
407 
408  s4_val_free (value);
409  s4_sourcepref_unref (sp);
410 
411  operands = xmmsv_coll_operands_get (coll);
412  xmmsv_list_get (operands, 0, &operand);
413 
414  if (!is_universe (operand)) {
415  s4_condition_t *op_cond = cond;
416  cond = s4_cond_new_combiner (S4_COMBINE_AND);
417  s4_cond_add_operand (cond, op_cond);
418  s4_cond_unref (op_cond);
419  op_cond = collection_to_condition (session, operand, fetch, order);
420  s4_cond_add_operand (cond, op_cond);
421  s4_cond_unref (op_cond);
422  }
423 
424  return cond;
425 }

480 /**
481  * Calculate a hash code for the combination of multiple column values.
482  */
483 static gint
484 limit_condition_calculate_hash (const s4_resultrow_t *row, gint *idx)
485 {
486  gint hash = 1;
487  gint i;
488 
489  for (i = 0; idx[i] != '\0'; i++) {
490  const s4_result_t *result;
491  const s4_val_t *value;
492  const gchar *string;
493  gint32 number;
494 
495  if (!s4_resultrow_get_col (row, idx[i], &result))
496  continue;
497 
498  value = s4_result_get_val (result);
499  if (s4_val_get_str (value, &string)) {
500  hash = hash * 31 + g_str_hash (string);
501  } else if (s4_val_get_int (value, &number)) {
502  hash = hash * 31 + (number ^ (number >> 16));
503  }
504  };
505 
506  return hash;
507 }

572 /**
573  * Limit search result based on the combined key of multiple fields.
574  * This is useful when grouping is applied on the result and you want
575  * for example to query the second and third albums by an artist.
576  */
577 static void
578 limit_condition_by_value (s4_resultset_t *set, xmmsv_t *id_list, GHashTable *id_table,
579  gint start, guint length, gint *indices)
580 {
581  GHashTable *group_table, *skip_table;
582  const s4_resultrow_t *row;
583  gint i;
584 
585  skip_table = g_hash_table_new (g_direct_hash, g_direct_equal);
586  group_table = g_hash_table_new (g_direct_hash, g_direct_equal);
587 
588  for (i = 0; s4_resultset_get_row (set, i, &row); i++) {
589  const s4_result_t *result;
590  const s4_val_t *value;
591  gint hash, mid;
592 
593  hash = limit_condition_calculate_hash (row, indices);
594 
595  if (g_hash_table_size (skip_table) < start) {
596  g_hash_table_insert (skip_table,
597  GINT_TO_POINTER (hash),
598  GINT_TO_POINTER (TRUE));
599  continue;
600  }
601 
602  if (g_hash_table_lookup (group_table, GINT_TO_POINTER (hash)) == NULL) {
603  if (g_hash_table_lookup (skip_table, GINT_TO_POINTER (hash)) != NULL)
604  continue;
605 
606  if (g_hash_table_size (group_table) >= length)
607  continue;
608 
609  g_hash_table_insert (group_table,
610  GINT_TO_POINTER (hash),
611  GINT_TO_POINTER (TRUE));
612  }
613 
614  if (!s4_resultrow_get_col (row, 0, &result))
615  continue;
616 
617  value = s4_result_get_val (result);
618  if (!s4_val_get_int (value, &mid))
619  continue;

802 static s4_condition_t *
803 union_ordered_condition (xmms_medialib_session_t *session, xmmsv_t *coll,
804  xmms_fetch_info_t *fetch, xmmsv_t *order)

928  case XMMS_COLLECTION_TYPE_NOTEQUAL:
929  case XMMS_COLLECTION_TYPE_SMALLER:

931  case XMMS_COLLECTION_TYPE_GREATER:

942  case XMMS_COLLECTION_TYPE_ORDER:
