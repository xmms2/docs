22 /**
23  * Create a new argument for a method.
24  *
25  * @param name The name of the argument. Must not be NULL.
26  * @param docstring The docstring of the argument.
27  * @param type The expected type of the argument. Use XMMSV_TYPE_NONE to
28  * accept any type. XMMSV_TYPE_ERROR is reserved and should not be used.
29  * @param default_value Value to set this argument to if it's missing in the
30  * method call. Implies that the argument is optional. If NULL, the argument
31  * is not optional.
32  */
33 xmmsv_t *
34 xmmsv_sc_argument_new (const char *name, const char *docstring,
35  xmmsv_type_t type, xmmsv_t *default_value)
36 {
37  xmmsv_t *arg;
38 
39  x_api_error_if (!name, "with NULL name.", NULL);
40  x_api_error_if (type == XMMSV_TYPE_ERROR, "with ERROR type.", NULL);
41  x_api_error_if (default_value && type != XMMSV_TYPE_NONE &&
42  xmmsv_get_type (default_value) != type,
43  "with wrong type for default value.", NULL);
44 
45  arg = xmmsv_new_dict ();
46  if (!arg) {
47  x_oom ();
48  return NULL;
49  }
50 
51  xmmsv_dict_set_string (arg, "name", name);
52  xmmsv_dict_set_int (arg, "type", type);
53 
54  if (docstring) {
55  xmmsv_dict_set_string (arg, "docstring", docstring);
56  }
57 
58  if (default_value) {
59  xmmsv_dict_set (arg, "default_value", default_value);
60  }
61 
62  return arg;
63 }
