183 static s4_t *
184 xmms_medialib_database_convert (const gchar *database_name,
185  const gchar *indices[])
186 {
187  const gchar *coll_conf, *conv_conf;
188  gchar *cmdline, *new_name, *obsolete_name;
189  xmms_config_property_t *cfg;
190  gint exit_status;
191  s4_t *s4;
192 
193  cfg = xmms_config_lookup ("collection.directory");
194  coll_conf = xmms_config_property_get_string (cfg);
195 
196  cfg = xmms_config_lookup ("sqlite2s4.path");
197  conv_conf = xmms_config_property_get_string (cfg);
198 
199  new_name = xmms_medialib_database_converted_name (database_name);
200 
201  cmdline = g_strjoin (" ", conv_conf, database_name,
202  new_name, coll_conf, NULL);
203 
204  xmms_log_info ("Attempting to migrate database to new format.");
205 
206  if (!g_spawn_command_line_sync (cmdline, NULL, NULL, &exit_status, NULL) || exit_status) {
207  xmms_log_fatal ("Could not run \"%s\", try to run it manually", cmdline);
208  }
209 
210  g_free (cmdline);
211 
212  s4 = s4_open (new_name, indices, 0);
213  /* Now we give up */
214  if (s4 == NULL) {
215  xmms_log_fatal ("Could not open the S4 database");
216  }
217 
218  xmms_log_info ("Migration successful.");
219 
220  /* Move the sqlite database */
221  obsolete_name = g_strconcat (database_name, ".obsolete", NULL);
222  g_rename (database_name, obsolete_name);
223  g_free (obsolete_name);
224 
225  /* Update the config path */
226  cfg = xmms_config_lookup ("medialib.path");
227  xmms_config_property_set_data (cfg, new_name);
228 
229  g_free (new_name);
230 
231  return s4;
232 }

237 {
238  gint flags = 0;
239  s4_t *s4;
240 
241  g_return_val_if_fail (database_name, NULL);
242 
243  if (strcmp (database_name, "memory://") == 0) {
244  flags = S4_MEMORY;
245  }
246 
247  s4 = s4_open (database_name, indices, flags);
248  if (s4 != NULL) {
249  return s4;
250  }
251

254  xmms_log_fatal ("Could not open the S4 database");

610 static gboolean

631  return TRUE;
