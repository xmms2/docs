62 /**
63  * Free a #xmmsv_t along with its internal data.
64  * @internal
65  */
66 static void
67 _xmmsv_free (xmmsv_t *val)
68 {
69  x_return_if_fail (val);
70 
71  switch (val->type) {
72  case XMMSV_TYPE_NONE :
73  case XMMSV_TYPE_END :
74  case XMMSV_TYPE_INT64 :
75  case XMMSV_TYPE_FLOAT :
76  break;
77  case XMMSV_TYPE_ERROR :
78  free (val->value.error);
79  val->value.error = NULL;
80  break;
81  case XMMSV_TYPE_STRING :
82  free (val->value.string);
83  val->value.string = NULL;
84  break;
85  case XMMSV_TYPE_COLL:
86  _xmmsv_coll_free (val->value.coll);
87  val->value.coll = NULL;
88  break;
89  case XMMSV_TYPE_BIN :
90  free (val->value.bin.data);
91  val->value.bin.len = 0;
92  break;
93  case XMMSV_TYPE_LIST:
94  _xmmsv_list_free (val->value.list);
95  val->value.list = NULL;
96  break;
97  case XMMSV_TYPE_DICT:
98  _xmmsv_dict_free (val->value.dict);
99  val->value.dict = NULL;
100  break;
101  case XMMSV_TYPE_BITBUFFER:
102  if (!val->value.bit.ro && val->value.bit.buf) {
103  free (val->value.bit.buf);
104  }
105  val->value.bit.buf = NULL;
106  break;
107  }
108 
109  free (val);
110 }

203 /**
204  * Allocates a new binary data #xmmsv_t.
205  * @param data The data to store in the #xmmsv_t.
206  * @param len The size of the data.
207  * @return The new #xmmsv_t. Must be unreferenced with
208  * #xmmsv_unref.
209  */
210 xmmsv_t *
211 xmmsv_new_bin (const unsigned char *data, unsigned int len)
212 {
213  xmmsv_t *val = _xmmsv_new (XMMSV_TYPE_BIN);
214 
215  if (val) {
216  /* copy the data! */
217  val->value.bin.data = x_malloc (len);
218  if (!val->value.bin.data) {
219  free (val);
220  x_oom ();
221  return NULL;
222  }
223  memcpy (val->value.bin.data, data, len);
224  val->value.bin.len = len;
225  }
226 
227  return val;
228 }

245 /**
246  * Decreases the references for the #xmmsv_t
247  * When the number of references reaches 0 it will
248  * be freed. And thus all data you extracted from it
249  * will be deallocated.
250  */
251 void
252 xmmsv_unref (xmmsv_t *val)
253 {
254  x_return_if_fail (val);
255  x_api_error_if (val->ref < 1, "with a freed value",);
256 
257  val->ref--;
258  if (val->ref == 0) {
259  _xmmsv_free (val);
260  }
261 }
