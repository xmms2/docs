74 xmms_mediainfo_reader_start (xmms_medialib_t *medialib)
75 {
76  xmms_mediainfo_reader_t *mrt;
77 
78  mrt = xmms_object_new (xmms_mediainfo_reader_t,
79  xmms_mediainfo_reader_stop);
80 
81  xmms_mediainfo_reader_register_ipc_commands (XMMS_OBJECT (mrt));
82 
83  g_mutex_init (&mrt->mutex);
84  g_cond_init (&mrt->cond);
85  mrt->running = TRUE;
86  mrt->thread = g_thread_new ("x2 media info", xmms_mediainfo_reader_thread, mrt);
87 
88  xmms_object_ref (medialib);
89  mrt->medialib = medialib;
90 
91 
92  xmms_object_connect (XMMS_OBJECT (mrt->medialib),
93  XMMS_IPC_SIGNAL_MEDIALIB_ENTRY_ADDED,
94  on_medialib_entry_added, mrt);
95 
96  xmms_object_connect (XMMS_OBJECT (mrt->medialib),
97  XMMS_IPC_SIGNAL_MEDIALIB_ENTRY_CHANGED,
98  on_medialib_entry_added, mrt);
99 
100  return mrt;
101 }

152 static gpointer
153 xmms_mediainfo_reader_thread (gpointer data)
154 {
155  GList *goal_format;
156  GTimeVal timeval;
157  xmms_stream_type_t *f;
158  guint num = 0;
159 
160  xmms_mediainfo_reader_t *mrt = (xmms_mediainfo_reader_t *) data;
161 
162  xmms_object_emit (XMMS_OBJECT (mrt),
163  XMMS_IPC_SIGNAL_MEDIAINFO_READER_STATUS,
164  xmmsv_new_int (XMMS_MEDIAINFO_READER_STATUS_RUNNING));
165 
166  f = _xmms_stream_type_new (XMMS_STREAM_TYPE_BEGIN,
167  XMMS_STREAM_TYPE_MIMETYPE,
168  "audio/pcm",
169  XMMS_STREAM_TYPE_END);
170  goal_format = g_list_prepend (NULL, f);
171 
172  while (mrt->running) {
173  xmmsc_medialib_entry_status_t prev_status;
174  xmms_medialib_entry_t entry;
175  xmms_medialib_session_t *session;
176  xmms_xform_t *xform;
177 
178  session = xmms_medialib_session_begin (mrt->medialib);
179  entry = xmms_medialib_entry_not_resolved_get (session);
180  XMMS_DBG ("got %d as not resolved", entry);
181 
182  if (!entry) {
183  xmms_medialib_session_abort (session);
184  xmms_object_emit (XMMS_OBJECT (mrt),
185  XMMS_IPC_SIGNAL_MEDIAINFO_READER_STATUS,
186  xmmsv_new_int (XMMS_MEDIAINFO_READER_STATUS_IDLE));
187 
188  g_mutex_lock (&mrt->mutex);
189  g_cond_wait (&mrt->cond, &mrt->mutex);
190  g_mutex_unlock (&mrt->mutex);
191 
192  num = 0;
193 
194  xmms_object_emit (XMMS_OBJECT (mrt),
195  XMMS_IPC_SIGNAL_MEDIAINFO_READER_STATUS,
196  xmmsv_new_int (XMMS_MEDIAINFO_READER_STATUS_RUNNING));
197  continue;
198  }
199 
200  prev_status = xmms_medialib_entry_property_get_int (session, entry,
201  XMMS_MEDIALIB_ENTRY_PROPERTY_STATUS);
202 
203  if (num == 0) {
204  xmms_object_emit (XMMS_OBJECT (mrt),
205  XMMS_IPC_SIGNAL_MEDIAINFO_READER_UNINDEXED,
206  xmmsv_new_int (xmms_medialib_num_not_resolved (session)));
207  num = 10;
208  } else {
209  num--;
210  }
211 
212  xform = xmms_xform_chain_setup_session (mrt->medialib, session, entry,
213  goal_format, TRUE);
214 
215  if (!xform) {
216  if (prev_status == XMMS_MEDIALIB_ENTRY_STATUS_NEW) {
217  xmms_medialib_entry_remove (session, entry);
218  } else {
219  xmms_medialib_entry_status_set (session, entry,
220  XMMS_MEDIALIB_ENTRY_STATUS_NOT_AVAILABLE);
221  }
222  } else {
223  xmms_object_unref (xform);
224  g_get_current_time (&timeval);
225 
226  xmms_medialib_entry_property_set_int (session, entry,
227  XMMS_MEDIALIB_ENTRY_PROPERTY_ADDED,
228  timeval.tv_sec);
229  }
230  xmms_medialib_session_commit (session);
231  }
232 
233  g_list_free (goal_format);
234  xmms_object_unref (f);
235 
236  return NULL;
237 }
