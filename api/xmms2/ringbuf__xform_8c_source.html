195 static gpointer
196 xmms_ringbuf_xform_thread (gpointer data)
197 {
198  xmms_xform_t *xform = (xmms_xform_t *)data;
199  xmms_ringbuf_priv_t *priv;
200 
201  priv = xmms_xform_private_data_get (xform);
202 
203  g_mutex_lock (&priv->state_lock);
204  while (priv->state != STATE_WANT_STOP) {
205  if (priv->state == STATE_WANT_BUFFER) {
206  priv->state = STATE_BUFFERING;
207  g_cond_signal (&priv->state_cond);
208  while (priv->state == STATE_BUFFERING) {
209  g_mutex_unlock (&priv->state_lock);
210  fill (xform, priv);
211  g_mutex_lock (&priv->state_lock);
212  }
213  } else if (priv->state == STATE_WANT_SEEK) {
214  seek (xform, priv);
215  priv->state = STATE_SEEK_DONE;
216  priv->seek_whence = XMMS_XFORM_SEEK_CUR;
217  priv->seek_offset = 0;
218  g_cond_signal (&priv->state_cond);
219  while (priv->state == STATE_SEEK_DONE) {
220  g_cond_wait (&priv->state_cond, &priv->state_lock);
221  }
222  }
223  XMMS_DBG ("thread: state: %d", priv->state);
224  }
225  priv->state = STATE_IS_STOPPED;
226  g_cond_signal (&priv->state_cond);
227  g_mutex_unlock (&priv->state_lock);
228 
229  return NULL;
