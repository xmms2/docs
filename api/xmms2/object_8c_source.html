53 /**
54  * Cleanup all the resources for the object
55  */
56 void
57 xmms_object_cleanup (xmms_object_t *object)
58 {
59  g_return_if_fail (object);
60  g_return_if_fail (XMMS_IS_OBJECT (object));
61 
62  if (object->signals) {
63  /* destroy the tree manually (ie not via a value_destroy_func
64  * callback since we're often "replacing" values when we're
65  * adding new elements to the signal lists. and we don't want
66  * the value to be destroyed in those cases :)
67  */
68  g_tree_foreach (object->signals, cleanup_signal_list, NULL);
69  g_tree_destroy (object->signals);
70  }
71 
72  if (object->cmds) {
73  /* We don't need to free the commands themselves -- they are
74  * stored in read-only memory.
75  */
76  g_tree_destroy (object->cmds);
77  }
78 
79  g_mutex_clear (&object->mutex);
80 }

109 void
110 xmms_object_connect (xmms_object_t *object, guint32 signalid,
111  xmms_object_handler_t handler, gpointer userdata)
112 {
113  GList *list = NULL;
114  xmms_object_handler_entry_t *entry;
115 
116  g_return_if_fail (object);
117  g_return_if_fail (XMMS_IS_OBJECT (object));
118  g_return_if_fail (handler);
119 
120  entry = g_new0 (xmms_object_handler_entry_t, 1);
121  entry->handler = handler;
122  entry->userdata = userdata;
123 
124  if (!object->signals)
125  object->signals = g_tree_new (compare_signal_key);
126  else
127  list = g_tree_lookup (object->signals,
128  GINT_TO_POINTER (signalid));
129 
130  list = g_list_prepend (list, entry);
131 
132  /* store the list's new head in the tree */
133  g_tree_insert (object->signals, GINT_TO_POINTER (signalid), list);
134 }

140 void
141 xmms_object_disconnect (xmms_object_t *object, guint32 signalid,
142  xmms_object_handler_t handler, gpointer userdata)
143 {
144  GList *list, *node = NULL;
145  xmms_object_handler_entry_t *entry;
146 
147  g_return_if_fail (object);
148  g_return_if_fail (XMMS_IS_OBJECT (object));
149  g_return_if_fail (handler);
150 
151  g_mutex_lock (&object->mutex);
152 
153  if (object->signals) {
154  list = g_tree_lookup (object->signals,
155  GINT_TO_POINTER (signalid));
156 
157  for (node = list; node; node = g_list_next (node)) {
158  entry = node->data;
159 
160  if (entry->handler == handler && entry->userdata == userdata)
161  break;
162  }
163 
164  if (node) {
165  list = g_list_remove_link (list, node);
166 
167  /* store the list's new head in the tree */
168  g_tree_insert (object->signals,
169  GINT_TO_POINTER (signalid), list);
170  }
171  }
172 
173  g_mutex_unlock (&object->mutex);
174 
175  g_return_if_fail (node);
176 
177  g_free (node->data);
178  g_list_free_1 (node);
179 }

182  * Emit a signal and thus call all the handlers that are connected.

186  * @param data the data that should be sent to the handler.

189 void
190 xmms_object_emit (xmms_object_t *object, guint32 signalid, xmmsv_t *data)
191 {
192  GList *list, *node, *list2 = NULL;
193  xmms_object_handler_entry_t *entry;
194 
195  g_return_if_fail (object);
196  g_return_if_fail (XMMS_IS_OBJECT (object));
197 
198  g_mutex_lock (&object->mutex);
199 
200  if (object->signals) {
201  list = g_tree_lookup (object->signals,
202  GINT_TO_POINTER (signalid));
203 
204  for (node = list; node; node = g_list_next (node)) {
205  entry = node->data;
206 
207  list2 = g_list_prepend (list2, entry);
208  }
209  }
210 
211  g_mutex_unlock (&object->mutex);
212 
213  while (list2) {
214  entry = list2->data;
215 
216  /* NULL entries may never be added to the trees. */
217  g_assert (entry);
218  g_assert (entry->handler);
219 
220  entry->handler (object, data, entry->userdata);
221 
222  list2 = g_list_delete_link (list2, list2);
223  }
224 
225  xmmsv_unref (data);
226 }
