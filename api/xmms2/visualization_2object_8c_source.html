46 /* create an uninitialised vis client. don't use this method without mutex! */
47 static int32_t
48 create_client (void)
49 {
50  int32_t id;
51 
52  for (id = 0; id < vis->clientc; ++id) {
53  if (!vis->clientv[id]) {
54  break;
55  }
56  }
57 
58  if (id == vis->clientc) {
59  vis->clientc++;
60  }
61 
62  vis->clientv = g_renew (xmms_vis_client_t*, vis->clientv, vis->clientc);
63  if (!vis->clientv || (!(vis->clientv[id] = g_new (xmms_vis_client_t, 1)))) {
64  vis->clientc = 0;
65  id = -1;
66  }
67 
68  xmms_log_info ("Attached visualization client %d", id);
69  return id;
70 }

83 void
84 delete_client (int32_t id)
85 {
86  xmms_vis_client_t *c;
87 
88  if (id < 0 || id >= vis->clientc) {
89  return;
90  }
91 
92  c = vis->clientv[id];
93  if (c == NULL) {
94  return;
95  }
96 
97  if (c->type == VIS_UNIXSHM) {
98  cleanup_shm (&c->transport.shm);
99  } else if (c->type == VIS_UDP) {
100  cleanup_udp (&c->transport.udp, vis->socket);
101  }
102 
103  g_free (c);
104  vis->clientv[id] = NULL;
105 
106  xmms_log_info ("Removed visualization client %d", id);
107 }

174 static gboolean
175 property_set (xmmsc_vis_properties_t *p, const gchar* key, const gchar* data)
176 {
177  if (!g_ascii_strcasecmp (key, "type")) {
178  if (!g_ascii_strcasecmp (data, "pcm")) {
179  p->type = VIS_PCM;
180  } else if (!g_ascii_strcasecmp (data, "spectrum")) {
181  p->type = VIS_SPECTRUM;
182  } else if (!g_ascii_strcasecmp (data, "peak")) {
183  p->type = VIS_PEAK;
184  } else {
185  return FALSE;
186  }
187  } else if (!g_ascii_strcasecmp (key, "stereo")) {
188  p->stereo = (atoi (data) > 0);
189  } else if (!g_ascii_strcasecmp (key, "pcm.hardwire")) {
190  p->pcm_hardwire = (atoi (data) > 0);
191  /* TODO: all the stuff following */
192  } else if (!g_ascii_strcasecmp (key, "timeframe")) {
193  p->timeframe = g_strtod (data, NULL);
194  if (p->timeframe == 0.0) {
195  return FALSE;
196  }
197  } else {
198  return FALSE;
199  }
200  return TRUE;
201 }

203 static int32_t
204 xmms_visualization_client_register (xmms_visualization_t *vis, xmms_error_t *err)
205 {
206  int32_t id;
207  xmms_vis_client_t *c;
208 
209  g_mutex_lock (&vis->clientlock);
210  id = create_client ();
211  if (id < 0) {
212  xmms_error_set (err, XMMS_ERROR_OOM, "could not allocate dataset");
213  } else {
214  /* do necessary initialisations here */
215  c = get_client (id);
216  c->type = VIS_NONE;
217  c->format = 0;
218  properties_init (&c->prop);
219  }
220  g_mutex_unlock (&vis->clientlock);
221  return id;
222 }

225 static int32_t
226 xmms_visualization_client_set_property (xmms_visualization_t *vis, int32_t id, const gchar* key, const gchar* value, xmms_error_t *err)
227 {
228  xmms_vis_client_t *c;
229 
230  x_fetch_client (id);
231 
232  if (!property_set (&c->prop, key, value)) {
233  xmms_error_set (err, XMMS_ERROR_INVAL, "property could not be set!");
234  }
235 
236  x_release_client ();
237 
238  /* the format identifier (between client and server) changes. so the client can recognize the first packet
239  which is built using the new format according to the newly set property */
240  return (++c->format);
241 }

243 static int32_t
244 xmms_visualization_client_set_properties (xmms_visualization_t *vis, int32_t id, xmmsv_t* prop, xmms_error_t *err)
245 {
246  xmms_vis_client_t *c;
247  xmmsv_dict_iter_t *it;
248  const gchar *key, *valstr;
249  xmmsv_t *value;
250 
251  x_fetch_client (id);
252 
253  if (!xmmsv_is_type (prop, XMMSV_TYPE_DICT)) {
254  xmms_error_set (err, XMMS_ERROR_INVAL, "properties must be sent as a dict!");
255  } else {
256  /* record every pair */
257  xmmsv_get_dict_iter (prop, &it);
258  while (xmmsv_dict_iter_valid (it)) {
259  if (!xmmsv_dict_iter_pair (it, &key, &value)) {
260  xmms_error_set (err, XMMS_ERROR_INVAL, "key-value property pair could not be read!");
261  } else if (!xmmsv_get_string (value, &valstr)) {
262  xmms_error_set (err, XMMS_ERROR_INVAL, "property value could not be read!");
263  } else if (!property_set (&c->prop, key, valstr)) {
264  xmms_error_set (err, XMMS_ERROR_INVAL, "property could not be set!");
265  }
266  xmmsv_dict_iter_next (it);
267  }
268  /* TODO: propagate new format to xform! */
269  }
270 
271  x_release_client ();
272 
273  return (++c->format);
274 }

316 void
317 send_data (int channels, int size, short *buf)
318 {
319  int i;
320  struct timeval time;
321  guint32 latency;
322 
323  if (!vis) {
324  return;
325  }
326 
327  latency = xmms_output_latency (vis->output);
328 
329  fft_init ();
330 
331  gettimeofday (&time, NULL);
332  time.tv_sec += (latency / 1000);
333  time.tv_usec += (latency % 1000) * 1000;
334  if (time.tv_usec > 1000000) {
335  time.tv_sec++;
336  time.tv_usec -= 1000000;
337  }
338 
339  g_mutex_lock (&vis->clientlock);
340  for (i = 0; i < vis->clientc; ++i) {
341  if (vis->clientv[i]) {
342  package_write (vis->clientv[i], i, &time, channels, size, buf);
343  }
344  }
345  g_mutex_unlock (&vis->clientlock);
346 }
