62 path_get_body (const gchar *path)
63 {
64  gchar *beg, *end;
65 
66  g_return_val_if_fail (path, NULL);
67 
68  beg = strstr (path, "://");
69 
70  if (!beg) {

72  }
73

102 gchar *
103 xmms_build_playlist_url (const gchar *plspath, const gchar *file)
104 {
105  gchar *url;
106  gchar *path;
107 
108  g_return_val_if_fail (plspath, NULL);
109  g_return_val_if_fail (file, NULL);
110 
111  if (strstr (file, "://") != NULL) {
112  return g_strdup (file);
113  }
114 
115  if (file[0] == '/') {
116  path = path_get_body (plspath);
117  url = g_strconcat (path, file, NULL);
118  } else {
119  path = path_get_dirname (plspath);
120  url = g_strconcat (path, "/", file, NULL);
121  }
122 
123  g_free (path);
124  return url;
125 }
