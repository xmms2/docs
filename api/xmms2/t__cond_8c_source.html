203 static int xor_combiner (s4_condition_t *cond, check_function_t func, void *check_data)
204 {
205  int i, ret = 1;
206  s4_condition_t *op;
207 
208  for (i = 0; (op = s4_cond_get_operand (cond, i)) != NULL; i++) {
209  int tmp = func (op, check_data);
210 
211  ret = !!ret ^ !!tmp;
212  }
213 
214  return ret;
215 }
