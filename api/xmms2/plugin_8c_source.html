238 /**
239  * @internal Shut down the plugin system. This function unrefs all the plugins
240  * loaded.
241  */
242 void
243 xmms_plugin_shutdown ()
244 {
245 #ifdef HAVE_VALGRIND
246  /* print out a leak summary at this point, because the final leak
247  * summary won't include proper backtraces of leaks found in
248  * plugins, since we close the so's here.
249  *
250  * note: the following call doesn't do anything if we're not run
251  * in valgrind
252  */
253  VALGRIND_DO_LEAK_CHECK
254  ;
255 #endif
256 
257  while (xmms_plugin_list) {
258  xmms_plugin_t *p = xmms_plugin_list->data;
259 
260  /* if this plugin's refcount is > 1, then there's a bug
261  * in one of the other subsystems
262  */
263  if (p->object.ref > 1) {
264  XMMS_DBG ("%s's refcount is %i",
265  p->name, p->object.ref);
266  }
267 
268  xmms_object_unref (p);
269 
270  xmms_plugin_list = g_list_delete_link (xmms_plugin_list,
271  xmms_plugin_list);
272  }
273 }

275 /**
276  * @internal Load a plugin.
277  * @param[in] desc The plugin description.
278  * @param[in] module The model reference if dynamically loaded, otherwise NULL.
279  * @return TRUE if the plugin was loaded, otherwise FALSE.
280  */
281 gboolean
282 xmms_plugin_load (const xmms_plugin_desc_t *desc, GModule *module)
283 {
284  xmms_plugin_t *plugin;
285  xmms_plugin_t *(*allocer) (void);
286  gboolean (*verifier) (xmms_plugin_t *);
287  gint expected_ver;
288 
289  XMMS_DBG ("Loading plugin '%s'", desc->name);
290 
291  switch (desc->type) {
292  case XMMS_PLUGIN_TYPE_OUTPUT:
293  expected_ver = XMMS_OUTPUT_API_VERSION;
294  allocer = xmms_output_plugin_new;
295  verifier = xmms_output_plugin_verify;
296  break;
297  case XMMS_PLUGIN_TYPE_XFORM:
298  expected_ver = XMMS_XFORM_API_VERSION;
299  allocer = xmms_xform_plugin_new;
300  verifier = xmms_xform_plugin_verify;
301  break;
302  default:
303  XMMS_DBG ("Unknown plugin type!");
304  return FALSE;
305  }
306 
307  if (desc->api_version != expected_ver) {
308  XMMS_DBG ("Bad api version!");
309  return FALSE;
310  }
311 
312  plugin = allocer ();
313  if (!plugin) {
314  XMMS_DBG ("Alloc failed!");
315  return FALSE;
316  }
317 
318  if (!xmms_plugin_setup (plugin, desc)) {
319  xmms_log_error ("Setup failed for plugin '%s'!", desc->name);
320  xmms_object_unref (plugin);
321  return FALSE;
322  }
323 
324  if (!desc->setup_func (plugin)) {
325  xmms_log_error ("Setup function failed for plugin '%s'!",
326  desc->name);
327  xmms_object_unref (plugin);
328  return FALSE;
329  }
330 
331  if (!verifier (plugin)) {
332  xmms_log_error ("Verify failed for plugin '%s'!", desc->name);
333  xmms_object_unref (plugin);
334  return FALSE;
335  }
336 
337  plugin->module = module;
338 
339  xmms_plugin_list = g_list_prepend (xmms_plugin_list, plugin);
340  return TRUE;
341 }

343 /**
344  * @internal Scan a particular directory for plugins to load
345  * @param[in] dir Absolute path to plugins directory
346  * @return TRUE if directory successfully scanned for plugins
347  */
348 static gboolean
349 xmms_plugin_scan_directory (const gchar *dir)
350 {
351  GDir *d;
352  const char *name;
353  gchar *path;
354  gchar *temp;
355  gchar *pattern;
356  GModule *module;
357  gpointer sym;
358 
359  temp = get_module_ext (dir);
360 
361  XMMS_DBG ("Scanning directory for plugins (%s)", temp);
362 
363  pattern = g_path_get_basename (temp);
364 
365  g_free (temp);
366 
367  d = g_dir_open (dir, 0, NULL);
368  if (!d) {
369  xmms_log_error ("Failed to open plugin directory (%s)", dir);
370  return FALSE;
371  }
372 
373  while ((name = g_dir_read_name (d))) {
374 
375  if (!g_pattern_match_simple (pattern, name))
376  continue;
377 
378  path = g_build_filename (dir, name, NULL);
379  if (!g_file_test (path, G_FILE_TEST_IS_REGULAR)) {
380  g_free (path);
381  continue;
382  }
383 
384  XMMS_DBG ("Trying to load file: %s", path);
385  module = g_module_open (path, G_MODULE_BIND_LOCAL);
386  if (!module) {
387  xmms_log_error ("Failed to open plugin %s: %s",
388  path, g_module_error ());
389  g_free (path);
390  continue;
391  }
392 
393  if (!g_module_symbol (module, "XMMS_PLUGIN_DESC", &sym)) {
394  xmms_log_error ("Failed to find plugin header in %s", path);
395  g_module_close (module);
396  g_free (path);
397  continue;
398  }
399  g_free (path);
400 
401  if (!xmms_plugin_load ((const xmms_plugin_desc_t *) sym, module)) {
402  g_module_close (module);
403  }
404  }
405 
406  g_dir_close (d);
407  g_free (pattern);
408 
409  return TRUE;
410 }
