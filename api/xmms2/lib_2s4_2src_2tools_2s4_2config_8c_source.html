72 static void encode_string (const char *val, char *buf)
73 {
74  for (; *val; val++) {
75  if (isalnum (*val)) {
76  *buf++ = *val;
77  } else {
78  *buf++ = '%';
79  *buf++ = HEX_TO_CHAR ((*val >> 4) & 0xf);
80  *buf++ = HEX_TO_CHAR (*val & 0xf);
81  }
82  }
83  *buf = '\0';
84 }

86 static void decode_string (char *buf)
87 {
88  char *w, *r;
89  for (w = r = buf; *r; w++, r++) {
90  if (*r == '%') {
91  *w = CHAR_TO_HEX (*(r+1)) << 4 | CHAR_TO_HEX (*(r+2));
92  r += 2;
93  } else {
94  *w = *r;
95  }
96  }
97  *w = '\0';
98 }

150 void set_var (const char *key, char *val)
151 {
152  int i, j;
153 
154  for (i = 0; user_vars[i].key != NULL; i++) {
155  if (strcmp (key, user_vars[i].key) == 0) {
156  if (user_vars[i].possible_values[0] != NULL) {
157  for (j = 0; user_vars[i].possible_values[j] != NULL; j++) {
158  if (strcmp (user_vars[i].possible_values[j], val) == 0) {
159  break;
160  }
161  }
162  if (user_vars[i].possible_values[j] == NULL) {
163  fprintf (stderr, "%s is not a valid value for %s\n", val, key);
164  free (val);
165  return;
166  }
167  }
168 
169  free (user_vars[i].value);
170  user_vars[i].value = val;
171  return;
172  }
173  }
174 
175  fprintf (stderr, "%s is not a configuration variable\n", key);
176  free (val);
177 }
