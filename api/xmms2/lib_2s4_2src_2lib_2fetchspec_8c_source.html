55 /**
56  * Adds something to fetch to the fetch specification. If key is NULL, it will fetch
57  * everything in an entry
58  *
59  * @param spec The specification to add to
60  * @param key The key to fetch
61  * @param sourcepref The sourcepref to use when deciding which key-value pair to fetch
62  * @param flags Flags to decide what to fetch
63  */
64 void s4_fetchspec_add (s4_fetchspec_t *spec, const char *key, s4_sourcepref_t *sourcepref, int flags)
65 {
66  fetch_data_t data;
67  data.flags = flags;
68  data.const_key = 0;
69  if (key == NULL) {
70  data.key = NULL;
71  } else {
72  data.key = strdup (key);
73  }
74  if (sourcepref != NULL) {
75  data.pref = s4_sourcepref_ref (sourcepref);
76  } else {
77  data.pref = NULL;
78  }
79 
80  g_array_append_val (spec->array, data);
81 }

83 void s4_fetchspec_update_key (s4_t *s4, s4_fetchspec_t *spec)
84 {
85  int i;
86  for (i = 0; i < spec->array->len; i++) {
87  fetch_data_t *data = &g_array_index (spec->array, fetch_data_t, i);
88  const char *new_key = _string_lookup (s4, data->key);
89 
90  if (!data->const_key)
91  free ((void*)data->key);
92 
93  data->key = new_key;
94  data->const_key = 1;
95  }
96 }

98 /**
99  * Frees a fetchspec
100  * @param spec The fetchspec to free
101  */
102 void s4_fetchspec_free (s4_fetchspec_t *spec)
103 {
104  int i;
105  for (i = 0; i < spec->array->len; i++) {
106  fetch_data_t *data = &g_array_index (spec->array, fetch_data_t, i);
107  if (data->pref != NULL) {
108  s4_sourcepref_unref (data->pref);
109  }
110  if (data->key != NULL && !data->const_key) {
111  free ((void*)data->key);
112  }
113  }
114  g_array_free (spec->array, TRUE);
115  free (spec);
116 }

130 /*
131  * Decrements the reference count of the spec. If the count reaches zero
132  * the spec is freed.
133  * @param spec The spec to decrement the count of
134  */
135 void s4_fetchspec_unref (s4_fetchspec_t *spec)
136 {
137  if (spec->ref_count <= 0) {
138  S4_ERROR ("s4_fetchspec_unref: refcount <= 0");
139  }
140  spec->ref_count--;
141  if (spec->ref_count == 0) {
142  s4_fetchspec_free (spec);
143  }
144 }
