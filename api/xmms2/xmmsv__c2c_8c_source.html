75 /**
76  * Format a client-to-client message.
77  *
78  * Messages are dictionaries of the form:\n
79  * "sender" : id of the sender client\n
80  * "destination" : id of the destination client\n
81  * "id" : 0 if the message doesn't expect reply, else a message id\n
82  * "payload" : the contents of the message
83  *
84  * @param sender the id of the sender client
85  * @param dest the id of the destination client
86  * @param id the id of the message
87  * @param payload the contents of the message
88  *
89  * @return the formatted message
90  *
91  * @note Increases the refcount of payload.
92  */
93 xmmsv_t *
94 xmmsv_c2c_message_format (int sender, int dest, int id,
95  xmmsv_t *payload)
96 {
97  xmmsv_ref (payload);
98  return xmmsv_build_dict (XMMSV_DICT_ENTRY_INT ("sender", sender),
99  XMMSV_DICT_ENTRY_INT ("destination", dest),
100  XMMSV_DICT_ENTRY_INT ("id", id),
101  XMMSV_DICT_ENTRY ("payload", payload),
102  XMMSV_DICT_END);
103 }
