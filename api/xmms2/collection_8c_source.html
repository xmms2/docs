222 /** Create a idlist from a playlist file
223  * @param dag The collection DAG.
224  * @param path URL to the playlist file
225  * @param err If error occurs, a message is stored in this variable.
226  * @returns A idlist
227  */
228 static xmmsv_t *
229 xmms_collection_client_idlist_from_playlist (xmms_coll_dag_t *dag,
230  const gchar *path,
231  xmms_error_t *err)
232 {
233  xmms_stream_type_t *stream_type;
234  xmms_xform_t *xform;
235  GList *stream_types;
236  xmmsv_t *dict, *list, *coll;
237  xmmsv_list_iter_t *it;
238  const gchar* src;
239 
240  stream_type = _xmms_stream_type_new (XMMS_STREAM_TYPE_BEGIN,
241  XMMS_STREAM_TYPE_MIMETYPE,
242  "application/x-xmms2-playlist-entries",
243  XMMS_STREAM_TYPE_END);
244  stream_types = g_list_prepend (NULL, stream_type);
245 
246  /* we don't want any effects for playlist, so just report we're rehashing */
247  xform = xmms_xform_chain_setup_url (dag->medialib, 0, path, stream_types, TRUE);
248 
249  if (!xform) {
250  xmms_error_set (err, XMMS_ERROR_NO_SAUSAGE, "We can't handle this type of playlist or URL");
251  return NULL;
252  }
253 
254  list = xmms_xform_browse_method (xform, "/", err);
255  if (xmms_error_iserror (err)) {
256  xmms_object_unref (xform);
257  return NULL;
258  }
259 
260  coll = xmmsv_new_coll (XMMS_COLLECTION_TYPE_IDLIST);
261  src = "plugin/playlist";
262 
263  xmmsv_get_list_iter (list, &it);
264 
265  while (xmmsv_list_iter_entry (it, &dict)) {
266  xmms_medialib_session_t *session;
267  xmms_medialib_entry_t entry;
268  xmmsv_t *value;
269  const gchar *realpath;
270 
271  xmmsv_list_iter_next (it);
272 
273  if (!xmmsv_dict_get (dict, "realpath", &value)) {
274  xmms_log_error ("Playlist plugin did not set realpath; probably a bug in plugin");
275  continue;
276  }
277 
278  xmmsv_get_string (value, &realpath);
279  xmmsv_ref (value);
280 
281  xmmsv_dict_remove (dict, "realpath");
282  xmmsv_dict_remove (dict, "path");
283 
284  do {
285  session = xmms_medialib_session_begin (dag->medialib);
286  entry = xmms_medialib_entry_new_encoded (session, realpath, err);
287 
288  if (entry) {
289  add_metadata_from_tree_user_data_t udata;
290  udata.entry = entry;
291  udata.src = src;
292  udata.session = session;
293 
294 
295  xmmsv_dict_foreach (dict, add_metadata_from_tree, &udata);
296 
297  xmmsv_coll_idlist_append (coll, entry);
298  } else {
299  xmms_log_error ("couldn't add %s to collection!", realpath);
300  }
301  } while (!xmms_medialib_session_commit (session));
302 
303  xmmsv_unref (value);
304  }
305 
306  xmmsv_unref (list);
307 
308  xmms_object_unref (xform);
309  xmms_object_unref (stream_type);
310  g_list_free (stream_types);
311 
312  return coll;
313 }

358 /** Save the given collection in the DAG under the given name in the given namespace.
359  *
360  * @param dag The collection DAG in which to save the collection.
361  * @param name The name under which to save the collection.
362  * @param namespace The namespace in which to save th collection.
363  * @param coll The collection structure to save.
364  * @param err If an error occurs, a message is stored in it.
365  * @returns True on success, false otherwise.
366  */
367 void
368 xmms_collection_client_save (xmms_coll_dag_t *dag, const gchar *name, const gchar *namespace,
369  xmmsv_t *coll, xmms_error_t *err)
370 {
371  const gchar *valerr = "Invalid collection: unknown reason. This is "
372  "probably a bug in xmms2d.";
373  xmms_collection_namespace_id_t nsid;
374  xmmsv_t *existing;
375  gchar *alias;
376  GList *list, *item;
377 
378  nsid = xmms_collection_get_namespace_id (namespace);
379  if (nsid == XMMS_COLLECTION_NSID_INVALID) {
380  xmms_error_set (err, XMMS_ERROR_INVAL, "invalid collection namespace");
381  return;
382  } else if (nsid == XMMS_COLLECTION_NSID_ALL) {
383  xmms_error_set (err, XMMS_ERROR_GENERIC, "cannot save collection in all namespaces");
384  return;
385  }
386 
387  /* Validate collection structure */
388  if (!xmms_collection_validate (dag, coll, name, namespace, &valerr)) {
389  xmms_error_set (err, XMMS_ERROR_INVAL, valerr);
390  return;
391  }
392 
393  g_mutex_lock (&dag->mutex);
394 
395  /* Unreference previously saved collection */
396  existing = xmms_collection_get_pointer (dag, name, nsid);
397  if (existing != NULL) {
398  /* Rebind reference pointers to the new collection */
399  coll_rebind_infos_t infos = { name, namespace, existing, coll };
400  xmms_collection_apply_to_all_collections (dag, rebind_references, &infos);
401  }
402 
403  /* Link references in newly saved collection to actual operators */
404  xmms_collection_apply_to_collection (dag, coll, bind_all_references, NULL);
405 
406  /* Update existing collection in the table */
407  if (existing != NULL) {
408  list = NULL;
409  while ((alias = xmms_collection_find_alias (dag, nsid, existing, NULL)) != NULL) {
410  /* update all pairs pointing to the old coll */
411  xmms_collection_update_pointer (dag, alias, nsid, coll);
412 
413  list = g_list_prepend (list, alias);
414  }
415 
416  for (item = g_list_first (list); item; item = g_list_next (item)) {
417  alias = item->data;
418 
419  XMMS_COLLECTION_CHANGED_MSG (XMMS_COLLECTION_CHANGED_UPDATE,
420  alias, namespace);
421  g_free (alias);
422  }
423  g_list_free (list);
424 
425  /* Save new collection in the table */
426  } else {
427  xmms_collection_update_pointer (dag, name, nsid, coll);
428 
429  XMMS_COLLECTION_CHANGED_MSG (XMMS_COLLECTION_CHANGED_ADD,
430  name, namespace);
431  }
432 
433  g_mutex_unlock (&dag->mutex);
434 }

511 /** Find all collections in the given namespace that contain a given media.
512  *
513  * @param dag The collection DAG.
514  * @param mid The id of the media.
515  * @param namespace The namespace in which to look for collections.
516  * @param err If an error occurs, a message is stored in it.
517  * @returns A newly allocated xmmsv_t list with the names of the matching collections.
518  */
519 xmmsv_t *
520 xmms_collection_client_find (xmms_coll_dag_t *dag, gint32 mid, const gchar *namespace,
521  xmms_error_t *err)
522 {
523  xmms_collection_namespace_id_t nsid;
524  xmmsv_t *result;
525  gchar *open_name;
526  GHashTable *match_table;
527  xmmsv_t *coll, *filter_coll;
528  xmmsv_t *idlist;
529 
530  /* Verify namespace */
531  nsid = xmms_collection_get_namespace_id (namespace);
532  if (nsid == XMMS_COLLECTION_NSID_INVALID) {
533  xmms_error_set (err, XMMS_ERROR_INVAL, "invalid collection namespace");
534  return NULL;
535  }
536  if (nsid == XMMS_COLLECTION_NSID_ALL) {
537  xmms_error_set (err, XMMS_ERROR_INVAL, "cannot search in all namespaces");
538  return NULL;
539  }
540 
541  /* Prepare the match table of all collections for the given namespace */
542  match_table = g_hash_table_new_full (g_str_hash, g_str_equal, g_free, g_free);
543  xmms_collection_foreach_in_namespace (dag, nsid, build_match_table, match_table);
544 
545  filter_coll = xmmsv_new_coll (XMMS_COLLECTION_TYPE_EQUALS);
546  xmmsv_coll_attribute_set_string (filter_coll, "type", "id");
547  xmms_collection_set_int_attr (filter_coll, "value", mid);
548 
549  /* While not all collections have been checked, check next */
550  while (g_hash_table_find (match_table, find_unchecked, &open_name) != NULL) {
551  coll_find_state_t *match = g_new (coll_find_state_t, 1);
552  coll = xmms_collection_get_pointer (dag, open_name, nsid);
553 
554  xmmsv_coll_add_operand (filter_coll, coll);
555  idlist = xmms_collection_query_ids (dag, coll, err);
556 
557  if (xmmsv_list_get_size (idlist) > 0) {
558  *match = XMMS_COLLECTION_FIND_STATE_MATCH;
559  } else {
560  *match = XMMS_COLLECTION_FIND_STATE_NOMATCH;
561  }
562

1052 /**
1053  * Internal recursive validation function used to validate the whole
1054  * graph of a collection.
1055  */
1056 static gboolean
1057 xmms_collection_validate_recurs (xmms_coll_dag_t *dag, xmmsv_t *coll,
1058  const gchar *save_name,
1059  const gchar *save_namespace,
1060  const gchar **err)
1061 {
1062  guint num_operands = 0;
1063  xmmsv_t *op, *ref;
1064  const gchar *attr, *attr2;
1065  gboolean valid = TRUE;
1066  xmmsv_coll_type_t type;
1067  xmms_collection_namespace_id_t nsid;
1068 
1069  /* count operands */
1070  num_operands = xmmsv_list_get_size (xmmsv_coll_operands_get (coll));
1071 
1072  /* analyse by type */
1073  type = xmmsv_coll_get_type (coll);
1074  switch (type) {
1075  case XMMS_COLLECTION_TYPE_REFERENCE:
1076  /* zero or one (bound in DAG) operand */
1077  if (num_operands > 1) {
1078  *err = "Invalid collection: REFERENCE with more than one operand.";
1079  return FALSE;
1080  }
1081 
1082  /* check if referenced collection exists */
1083  xmmsv_coll_attribute_get_string (coll, "reference", &attr);
1084  if (attr == NULL) {
1085  *err = "Invalid collection: REFERENCE without \"reference\"-"
1086  "attribute.";
1087  return FALSE;
1088  } else if (strcmp (attr, "All Media") != 0) {
1089  xmmsv_coll_attribute_get_string (coll, "namespace", &attr2);
1090 
1091  if (attr2 == NULL) {
1092  *err = "Invalid collection: REFERENCE without \"namespace\"-"
1093  "attribute.";
1094  return FALSE;
1095  }
1096 
1097  nsid = xmms_collection_get_namespace_id (attr2);
1098  if (nsid == XMMS_COLLECTION_NSID_INVALID) {
1099  *err = "Invalid collection: REFERENCE with invalid "
1100  "\"namespace\"-attribute.";
1101  return FALSE;
1102  }
1103 
1104  g_mutex_lock (&dag->mutex);
1105  ref = xmms_collection_get_pointer (dag, attr, nsid);
1106  if (ref == NULL) {
1107  g_mutex_unlock (&dag->mutex);
1108  *err = "Invalid collection: REFERENCE refers to inexistent "
1109  "collection.";
1110  return FALSE;
1111  }
1112 
1113  if (save_name && save_namespace) {
1114  /* self-reference is of course forbidden */
1115  if (strcmp (attr, save_name) == 0 &&
1116  strcmp (attr2, save_namespace) == 0) {
1117 
1118  g_mutex_unlock (&dag->mutex);
1119  *err = "Invalid collection: cycle detected: REFERENCE "
1120  "refers to intended name and namespace.";
1121  return FALSE;
1122 
1123  /* check if the referenced coll references this one (loop!) */
1124  } else if (xmms_collection_has_reference_to (dag, ref, save_name,
1125  save_namespace)) {
1126  g_mutex_unlock (&dag->mutex);
1127  *err = "Invalid collection: cycle detected: REFERENCE "
1128  "refers to collection that (possibly indirectly) "
1129  "refers to intended name and namespace.";
1130  return FALSE;
1131  }
1132  }
1133 
1134  g_mutex_unlock (&dag->mutex);
1135  } else {
1136  /* "All Media" reference, so no referenced coll pointer */
1137  ref = NULL;
1138  }
1139 
1140  /* ensure that the operand is consistent with the reference infos */
1141  if (num_operands == 1) {
1142  xmmsv_list_get (xmmsv_coll_operands_get (coll), 0, &op);
1143  if (op != ref) {
1144  *err = "Invalid collection: REFERENCE refers to other "
1145  "collection than its operand. This is probably a bug in "
1146  "xmms2d.";
1147  xmms_log_error ("%s", *err);
1148  return FALSE;
1149  }
1150  }
1151  break;
1152 
1153  case XMMS_COLLECTION_TYPE_UNION:
1154  case XMMS_COLLECTION_TYPE_INTERSECTION:
1155  /* need operand(s) */
1156  if (num_operands == 0) {
1157  *err = "Invalid collection: UNION or INTERSECTION with zero "
1158  "operands.";
1159  return FALSE;
1160  }
1161  break;
1162 
1163  case XMMS_COLLECTION_TYPE_COMPLEMENT:
1164  /* one operand */
1165  if (num_operands != 1) {
1166  *err = "Invalid collection: COMPLEMENT with fewer or more than one "
1167  "operand.";
1168  return FALSE;
1169  }
1170  break;
1171 
1172  case XMMS_COLLECTION_TYPE_HAS:
1173  case XMMS_COLLECTION_TYPE_MATCH:
1174  case XMMS_COLLECTION_TYPE_TOKEN:
1175  case XMMS_COLLECTION_TYPE_EQUALS:
1176  case XMMS_COLLECTION_TYPE_NOTEQUAL:
1177  case XMMS_COLLECTION_TYPE_SMALLER:
1178  case XMMS_COLLECTION_TYPE_SMALLEREQ:
1179  case XMMS_COLLECTION_TYPE_GREATER:
