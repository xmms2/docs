252 /**
253  * Compares an int and a string
254  *
255  * @param i The int
256  * @param s The string
257  * @param coll_str A collated version of the string. If this is NULL
258  * strings will always be placed before integers.
259  * @return <0 if i<s, 0 if i==s and >0 if i>s
260  */
261 static int _int_str_cmp (int32_t i, const char *s, const char *coll_str)
262 {
263  static char *coll_num = NULL;
264 
265  if (coll_num == NULL) {
266  coll_num = s4_string_collate ("1");
267  }
268 
269  /* If we're comparing binary or caselessly we place integers after strings
270  * This is because with binary and caseless string-matching "123" < "23"
271  * but integers still use integer matching and 23 < 123, so we have to
272  * keep them separate (or we would break antisymmetry)
273  */
274  if (coll_str == NULL) {
275  return 1;
276  /* But for collated matching we have "23" < "123" and we can mix string
277  * and integer matching without breaking anything
278  */
279  } else {
280  char *end;
281  int32_t j = strtol (s, &end, 10);
282 
283  /* Strings with space in front of the number are treated like normal strings */
284  if (end != s && !isspace (*s)) {
285  return (i > j)?1:((i < j)?-1:-(*end != '\0'));
286  } else {
287  /* See if the given string is smaller or greater than a number */
288  return strcmp (coll_num, coll_str);
289  }
290  }
291 }

293 /**
294  * Compares two values
295  *
296  * @param v1 The first value
297  * @param v2 The second value
298  * @param mode How to compare the values
299  * @return <0 if v1<v2, 0 if v1==v2 and >0 if v1>v2
300  */
301 int s4_val_cmp (const s4_val_t *v1, const s4_val_t *v2, s4_cmp_mode_t mode)
302 {
303  int32_t i1,i2;
304  const char *s1,*s2;
305 
306  if (s4_val_get_int (v1, &i1) && s4_val_get_int (v2, &i2))
307  return (i1 > i2)?1:((i1 < i2)?-1:0);
308  else if (mode == S4_CMP_BINARY && s4_val_get_str (v1, &s1) && s4_val_get_str (v2, &s2))
309  return strcmp (s1, s2);
310  else if (mode == S4_CMP_CASELESS && s4_val_get_casefolded_str (v1, &s1) && s4_val_get_casefolded_str (v2, &s2))
311  return strcmp (s1, s2);
312  else if (mode == S4_CMP_COLLATE && s4_val_get_collated_str (v1, &s1) && s4_val_get_collated_str (v2, &s2))
313  return strcmp (s1, s2);
314  else if (s4_val_get_int (v1, &i1) && s4_val_get_str (v2, &s2))
315  if (mode == S4_CMP_COLLATE) {
316  s4_val_get_collated_str (v2, &s1);
317  return _int_str_cmp (i1, s2, s1);
318  } else {
319  return _int_str_cmp (i1, s2, NULL);
320  }
321  else if (s4_val_get_int (v2, &i2) && s4_val_get_str (v1, &s1))
322  if (mode == S4_CMP_COLLATE) {
323  s4_val_get_collated_str (v1, &s2);
324  return -_int_str_cmp (i2, s1, s2);
325  } else {
326  return -_int_str_cmp (i2, s1, NULL);
327  }
328  else /* This should never be hit */
329  return 0;
330 }
