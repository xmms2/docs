18 /**
19  * @defgroup Lock Lock
20  * @ingroup S4
21  * @brief Locks entries so only one transaction can use tham at a time.
22  *
23  * Locks can be locked in three ways, read-only, upgradable and exclusive.
24  * - Read-only is for read-only transactions that know they will not want
25  * an exclusive version of the lock later one
26  * - Upgradable is for transactions that might want to upgrade it to an
27  * exclusive lock later on.
28  * - Exclusive locks can only be held by one transaction.
29  *
30  * @{
31  */

83 /* Checks if making trans wait for lock would deadlock.
84  * Deadlock happens when we have to wait on a lock held by
85  * a transaction waiting for a lock this transaction holds.
86  * It can be thought of as searching for a cycle in a graph, where vertices
87  * are locks and edges go from the lock to the locks the transactions holding
88  * this lock are waiting for. If there is a cycle we have a deadlock.
89  *
90  * Returns 1 if it will deadlock, 0 otherwise
91  */
92 static int _lock_will_deadlock_helper (s4_lock_t *lock, s4_transaction_t *trans, GHashTable *visited, int first)
93 {
94  GHashTableIter iter;
95  GList *waiting_for = NULL;
96  s4_transaction_t *t;
97  int ret = 0;
98 
99  /* Bail if the lock is NULL or has already been visited */
100  if (lock == NULL || g_hash_table_lookup (visited, lock) != NULL)
101  return 0;
102 
103  /* Add this lock to the table of visited locks */
104  g_hash_table_insert (visited, lock, GINT_TO_POINTER (1));
105 
106  g_mutex_lock (&lock->lock);
107  g_hash_table_iter_init (&iter, lock->transactions);
108 
109  /* Get a list of all transactions holding this lock */
110  while (g_hash_table_iter_next (&iter, (void**)&t, NULL)) {
111  if (t != trans) {
112  waiting_for = g_list_prepend (waiting_for, _transaction_get_waiting_for (t));
113  } else {
114  /* If we found a lock that this transaction holds
115  * and it's not the first lock, we have a deadlock.
116  */
117  ret = !first;
118  }
119  }
120 
121  g_mutex_unlock (&lock->lock);
122 
123  /* Check all the locks the transactions in the list are waiting for */
124  while (!ret && waiting_for != NULL) {
125  ret = _lock_will_deadlock_helper (waiting_for->data, trans, visited, 0);
126  waiting_for = g_list_delete_link (waiting_for, waiting_for);
127  }
128 
129  g_list_free (waiting_for);
130 
131  return ret;
132 }

134 static int _lock_will_deadlock (s4_lock_t *lock, s4_transaction_t *trans)
135 {

146 /* Aquires an exclusive lock. */
147 int _lock_exclusive (s4_lock_t *lock, s4_transaction_t *trans)
148 {
149  _transaction_set_waiting_for (trans, lock);
150 
151  if (_lock_will_deadlock (lock, trans)) {
152  _transaction_set_waiting_for (trans, NULL);
153  s4_set_errno (S4E_DEADLOCK);
154  return 0;
155  }
156 
157  g_mutex_lock (&lock->lock);
158 
159  if (_lock_has_trans (lock, trans)) {
160  /* If we already hold this lock, but not exclusively,
161  * we have to upgrade it
162  */
163  if (!lock->exclusive) {
164  lock->want_upgrade = 1;
165  lock->readers--;
166  while (lock->readers) {
167  g_cond_wait (&lock->upgrade_signal, &lock->lock);
168  }
169  lock->want_upgrade = 0;
170  }
171  } else {
172  lock->writers_waiting++;
173  while (lock->readers || lock->exclusive || lock->upgrade) {
174  g_cond_wait (&lock->signal, &lock->lock);
175  }
176  lock->writers_waiting--;
177 
178  _lock_add_trans (lock, trans);
179  _transaction_add_lock (trans, lock);
180  }
181 
182  _transaction_set_waiting_for (trans, NULL);
183  lock->exclusive = 1;
184 
185  g_mutex_unlock (&lock->lock);
186  return 1;
187 }

189 /* Aquires a shared (upgradable if this is not a read-only transcation) lock */
190 int _lock_shared (s4_lock_t *lock, s4_transaction_t *trans)
191 {
192  /* If this is not a read-only transaction, we might want to
193  * aquire this lock exclusively later on, therefore it must be
194  * upgradable
195  */
196  int upgrade = !(_transaction_get_flags (trans) & S4_TRANS_READONLY);
197 
198  _transaction_set_waiting_for (trans, lock);
199 
200  if (_lock_will_deadlock (lock, trans)) {
201  _transaction_set_waiting_for (trans, NULL);
202  s4_set_errno (S4E_DEADLOCK);
203  return 0;
204  }
205 
206  g_mutex_lock (&lock->lock);
207 
208  /* If we do not already hold this lock we have to aquire it */
209  if (!_lock_has_trans (lock, trans)) {
210  while (lock->exclusive || lock->writers_waiting || (lock->upgrade && upgrade)) {
211  g_cond_wait (&lock->signal, &lock->lock);
212  }
213 
214  lock->readers++;
215  if (upgrade) {
216  lock->upgrade = 1;
217  }
218  _lock_add_trans (lock, trans);
219  _transaction_add_lock (trans, lock);
220  }
221 
222  _transaction_set_waiting_for (trans, NULL);
223  g_mutex_unlock (&lock->lock);
224  return 1;
225 }

227 /* Unlocks a single lock held by trans */
228 static void _lock_unlock (s4_lock_t *lock, s4_transaction_t *trans)
229 {
230  int upgrade = !(_transaction_get_flags (trans) & S4_TRANS_READONLY);
231 
232  g_mutex_lock (&lock->lock);
233  if (lock->exclusive) {
234  lock->exclusive = 0;
235  g_cond_signal (&lock->signal);
236  } else if (lock->readers) {
237  lock->readers--;
238 
239  if (lock->readers == 0) {
240  if (lock->want_upgrade) {
241  g_cond_signal (&lock->upgrade_signal);
242  } else {
243  g_cond_signal (&lock->signal);
244  }
245  }
246 
247  }
248 
249  _lock_del_trans (lock, trans);
250  if (upgrade) {
251  lock->upgrade = 0;
252  }
253 
254  g_mutex_unlock (&lock->lock);
255 }
