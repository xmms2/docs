70 static void create_db (struct db_struct *db)
71 {
72  s4_val_t *name_val, *arg_val;
73  int i, j;
74 
75  for (i = 0; db[i].name != NULL; i++) {
76  name_val = s4_val_new_string (db[i].name);
77 
78  for (j = 0; db[i].args[j] != NULL; j++) {
79  s4_transaction_t *trans;
80 
81  arg_val = s4_val_new_string (db[i].args[j]);
82 
83  trans = s4_begin (s4, 0);
84  CU_ASSERT (s4_add (trans, "entry", name_val, "property", arg_val, db[i].src));
85  s4_commit (trans);
86 
87  s4_val_free (arg_val);
88  }
89 
90  s4_val_free (name_val);
91  }
92 }

94 static void check_db (struct db_struct *db)
95 {
96  int i, j;
97  s4_fetchspec_t *fs = s4_fetchspec_create ();
98  s4_fetchspec_add (fs, NULL, NULL, S4_FETCH_DATA);
99 
100  for (i = 0; db[i].name != NULL; i++) {
101  s4_val_t *name_val = s4_val_new_string (db[i].name);
102  s4_condition_t *cond = s4_cond_new_filter (S4_FILTER_EQUAL,
103  "entry", name_val, NULL, S4_CMP_CASELESS, S4_COND_PARENT);
104  s4_transaction_t *trans = s4_begin (s4, 0);
105  s4_resultset_t *set = s4_query (trans, fs, cond);
106  s4_commit (trans);
107  const s4_result_t *res = s4_resultset_get_result (set, 0, 0);
108  int found[ARG_SIZE] = {0};
109 
110  for (; res != NULL; res = s4_result_next (res)) {
111  for (j = 0; db[i].args[j] != NULL; j++) {
112  const char *str;
113 
114  if (s4_val_get_str (s4_result_get_val (res), &str) && !strcmp (str, db[i].args[j]) &&
115  !strcmp (s4_result_get_key (res), "property") &&
116  !strcmp (s4_result_get_src (res), db[i].src)) {
117  found[j] = 1;
118  break;
119  }
120  }
121  }
122  for (j = 0; db[i].args[j] != NULL; j++) {
123  CU_ASSERT_EQUAL (found[j], 1);
124  }
125 
126  s4_resultset_free (set);
127  s4_cond_free (cond);
128  s4_val_free (name_val);
129  }
130 
131  s4_fetchspec_free (fs);
132 }
