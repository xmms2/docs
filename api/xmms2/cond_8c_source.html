75 /*
76  * A shortcutting OR combiner. It returns as soon as one of the operands is met
77  */
78 static int or_combiner (s4_condition_t *cond, check_function_t func, void *check_data)
79 {
80  int i, ret = 1;
81 
82  for (i = 0; ret && i < cond->u.combine.operands->len; i++) {
83  ret = func (g_ptr_array_index (cond->u.combine.operands, i), check_data);
84  }
85 
86  return ret;
87 }

89 /*
90  * A shortcutting AND combiner. It returns as soon as one of the operands is not met
91  */
92 static int and_combiner (s4_condition_t *cond, check_function_t func, void *check_data)
93 {
94  int i, ret = 0;
95 
96  for (i = 0; !ret && i < cond->u.combine.operands->len; i++) {
97  ret = func (g_ptr_array_index (cond->u.combine.operands, i), check_data);
98  }
99 
100  return ret;
101 }

186 /* A token filter. Checks if the passed value contains a token */
187 static int token_filter (const s4_val_t *value, s4_condition_t *cond)
188 {
189  const char *token = cond->u.filter.funcdata;
190  const char *s;
191  int32_t i;
192  s4_cmp_mode_t mode = cond->u.filter.cmp_mode;
193 
194  if ((mode == S4_CMP_CASELESS && s4_val_get_casefolded_str (value, &s)) ||
195  s4_val_get_str (value, &s)) {
196  while (*s) {
197  /* Skip whitespaces */
198  for (; isspace (*s); s++);
199 
200  /* Compare token */
201  for (i = 0; *s && *s == token[i] && token[i] != '*'; i++, s++);
202 
203  /* Check if it matched */
204  if (token[i] == '*' || (!token[i] && (isspace (*s) || !*s))) {
205  return 0;
206  }
207 
208  /* Eat the rest of the token */
209  for (; *s && !isspace (*s); s++);
210  }
211  } else if (s4_val_get_int (value, &i)) {
212  char *end;
213  int32_t j = strtol (token, &end, 10);
214 
215  if (end != token) {
216  /* If the token is just a number we have to have an exact match */
217  if (*end == '\0' && j == i) {
218  return 0;
219  /* If the character after the last digit is a star we have to
220  * shift the number until it has the right number of digits
221  */
222  } else if (*end == '*') {
223  for (; i > j; i /= 10);
224  if (i == j) {
225  return 0;
226  }
227  }
228  }
229  }
230 
231  return 1;
232 }

234 /*
235  * Sets the correct function and function data based on filter type
236  */
237 static void _set_filter_function (s4_condition_t *cond, s4_filter_type_t type, const s4_val_t *val)
238 {
239  switch (type) {
240  case S4_FILTER_EQUAL:
241  cond->u.filter.func = equal_filter;
242  cond->u.filter.funcdata = s4_val_copy (val);
243  cond->u.filter.free_func = (free_func_t)s4_val_free;
244  cond->u.filter.monotonic = 1;
245  break;
246  case S4_FILTER_NOTEQUAL:
247  cond->u.filter.func = notequal_filter;
248  cond->u.filter.funcdata = s4_val_copy (val);
249  cond->u.filter.free_func = (free_func_t)s4_val_free;
250  cond->u.filter.monotonic = 0;
251  break;
252  case S4_FILTER_GREATER:
253  cond->u.filter.func = greater_filter;
254  cond->u.filter.funcdata = s4_val_copy (val);
255  cond->u.filter.free_func = (free_func_t)s4_val_free;
256  cond->u.filter.monotonic = 1;
257  break;
258  case S4_FILTER_SMALLER:
259  cond->u.filter.func = smaller_filter;
260  cond->u.filter.funcdata = s4_val_copy (val);
261  cond->u.filter.free_func = (free_func_t)s4_val_free;
262  cond->u.filter.monotonic = 1;
263  break;
264  case S4_FILTER_GREATEREQ:
265  cond->u.filter.func = greatereq_filter;
266  cond->u.filter.funcdata = s4_val_copy (val);
267  cond->u.filter.free_func = (free_func_t)s4_val_free;
268  cond->u.filter.monotonic = 1;
269  break;
270  case S4_FILTER_SMALLEREQ:
271  cond->u.filter.func = smallereq_filter;
272  cond->u.filter.funcdata = s4_val_copy (val);
273  cond->u.filter.free_func = (free_func_t)s4_val_free;
274  cond->u.filter.monotonic = 1;
275  break;
276  case S4_FILTER_MATCH:
277  {
278  const char *s;
279  int32_t i;
280  char c[12];
281 
282  if (!s4_val_get_str (val, &s)) {
283  s4_val_get_int (val, &i);
284  sprintf (c, "%i", i);
285  s = c;
286  }
287 
288  cond->u.filter.func = match_filter;
289  cond->u.filter.funcdata = s4_pattern_create (s, cond->u.filter.cmp_mode == S4_CMP_CASELESS);
290  cond->u.filter.free_func = (free_func_t)s4_pattern_free;
291  cond->u.filter.monotonic = 0;
292  }
293  break;
294  case S4_FILTER_EXISTS:
295  cond->u.filter.func = (filter_function_t)everything;
296  cond->u.filter.funcdata = NULL;
297  cond->u.filter.free_func = NULL;
298  cond->u.filter.monotonic = 1;
299  break;
300  case S4_FILTER_TOKEN:
301  {
302  const char *str;
303  char *s;
304  int32_t i;
305 
306  if (s4_val_get_int (val, &i)) {
307  s = g_strdup_printf ("%i", i);
308  } else {
309  switch (cond->u.filter.cmp_mode) {
310  case S4_CMP_COLLATE:
311  /* Collated token matching makes no sense,
312  * we use binary matching instead
313  */
314  case S4_CMP_BINARY:
315  s4_val_get_str (val, &str);
316  break;
317  case S4_CMP_CASELESS:
318  s4_val_get_casefolded_str (val, &str);
319  break;
320  }
321  s = strdup (str);
322  }
323 
324  cond->u.filter.func = token_filter;
325  cond->u.filter.funcdata = s;
326  cond->u.filter.free_func = (free_func_t)g_free;
327  cond->u.filter.monotonic = 0;
328  }
329  break;
330  default: /* Unknown filter type */
331  cond->u.filter.func = (filter_function_t)never;
332  cond->u.filter.funcdata = NULL;
333  cond->u.filter.free_func = NULL;
334  cond->u.filter.monotonic = 0;
335  break;
336  }
337 }

568 /**
569  * Frees a condition and operands recursively
570  *
571  * @param cond The condition to free
572  */
573 void s4_cond_free (s4_condition_t *cond)
574 {
575  if (cond->type == S4_COND_COMBINER) {
576  g_ptr_array_free (cond->u.combine.operands, TRUE);
577  free (cond);
578  } else if (cond->type == S4_COND_FILTER) {
579  if (cond->u.filter.free_func != NULL)
580  cond->u.filter.free_func (cond->u.filter.funcdata);
581  if (cond->u.filter.sp != NULL)
582  s4_sourcepref_unref (cond->u.filter.sp);
583  if (!cond->u.filter.const_key && cond->u.filter.key != NULL)
584  free ((char*)cond->u.filter.key);
585  free (cond);
586  }
587 }
