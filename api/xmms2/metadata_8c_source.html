115 /* Parse a string like '1/10' and return how many integers were found */
116 static gint
117 xmms_xform_metadata_parse_number_slash_number (const gchar *value, gint *first, gint *second)
118 {
119  gchar *endptr = NULL;
120  gint ivalue;
121 
122  *first = *second = -1;
123 
124  ivalue = strtol (value, &endptr, 10);
125  if (!(endptr > value)) {
126  return 0;
127  }
128 
129  *first = ivalue;
130 
131  if (*endptr != '/') {
132  return 1;
133  }
134 
135  value = (const gchar *) endptr + 1;
136 
137  ivalue = strtol (value, &endptr, 10);
138  if (!(endptr > value)) {
139  return 1;
140  }
141 
142  *second = ivalue;
143 
144  return 2;
145 }

147 static gboolean
148 xmms_xform_metadata_parse_tracknumber (xmms_xform_t *xform, const gchar *key,
149  const gchar *value, gsize length)
150 {
151  gint tracknr, tracktotal;
152 
153  xmms_xform_metadata_parse_number_slash_number (value, &tracknr, &tracktotal);
154  if (tracknr > 0) {
155  const gchar *metakey = XMMS_MEDIALIB_ENTRY_PROPERTY_TRACKNR;
156  xmms_xform_metadata_set_int (xform, metakey, tracknr);
157  }
158  if (tracktotal > 0) {
159  const gchar *metakey = XMMS_MEDIALIB_ENTRY_PROPERTY_TOTALTRACKS;
160  xmms_xform_metadata_set_int (xform, metakey, tracktotal);
161  }
162 
163  return tracknr > 0;
164 }

185 gboolean
186 xmms_xform_metadata_parse_replay_gain (xmms_xform_t *xform, const gchar *key,
187  const gchar *value, gsize length)
188 {
189  gchar *endptr = NULL;
190  gdouble number;
191 
192  number = strtod (value, &endptr);
193  if (endptr > value) {
194  gchar buffer[8];
195  g_snprintf (buffer, sizeof (buffer), "%f",
196  pow (10.0, number / 20));
197  xmms_xform_metadata_set_str (xform, key, buffer);
198  return TRUE;
199  }
200 
201  return FALSE;
202 }

239 GHashTable *
240 xmms_metadata_mapper_init (const xmms_xform_metadata_basic_mapping_t *basic_mappings, gint basic_count,
241  const xmms_xform_metadata_mapping_t *mappings, gint count)
242 {
243  GHashTable *default_mapper, *result;
244  gint i;
245 
246  result = g_hash_table_new (xmms_strcase_hash, xmms_strcase_equal);
247 
248  default_mapper = xmms_metadata_mapper_get_default ();
249 
250  /* Initialize default mappers */
251  for (i = 0; i < basic_count; i++) {
252  const xmms_xform_metadata_mapping_t *mapping;
253 
254  mapping = g_hash_table_lookup (default_mapper, basic_mappings[i].to);
255  if (mapping == NULL) {
256  xmms_log_error ("No default metadata mapper for '%s'", basic_mappings[i].to);
257  continue;
258  }
259 
260  g_hash_table_insert (result, (gpointer) basic_mappings[i].from, (gpointer) mapping);
261  }
262 
263  for (i = 0; i < count; i++) {
264  g_hash_table_insert (result, (gpointer) mappings[i].key, (gpointer) &mappings[i]);
265  }
266 
267  g_hash_table_unref (default_mapper);
268 
269  return result;
270 }
