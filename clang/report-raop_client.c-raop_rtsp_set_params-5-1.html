88/* write upto 8 bits at-a-time into a buffer */
89static void
90write_bits (guint8 *buf, guint8 val, gint nbits, guint32 *offset)
91{
92 int bit_offset = *offset % 8;
93 int byte_offset = *offset / 8;
94 int left = 8 - bit_offset;
95
96 *offset += nbits;
97 if (nbits >= left) {
98 buf[byte_offset] |= (val >> (nbits - left));
99 val = (val << left) >> left;
100 nbits -= left;
101 left = 8;
102 byte_offset++;
103 }
104 if (nbits && nbits < left) {
105 buf[byte_offset] |= (val << (left - nbits));
106 }
107}

525 rc->stream_port);
526 if (ret == -1 && errno(*__errno_location ()) != EINPROGRESS115)
527 return RAOP_ESYS-4;
528 rc->io_state |= RAOP_IO_STREAM_WRITE0x08;
529 rc->io_state |= RAOP_IO_STREAM_READ0x04;
530 rc->rtsp_state = RAOP_RTSP_CONNECTED0x40;
531 } else if (rc->rtsp_state != RAOP_RTSP_CONNECTED0x40) {
532 rc->io_state |= RAOP_IO_RTSP_WRITE0x02;
533 }
534 } else if (fd == rc->stream_fd) {
535 char buf[56];
536 /* read data sent by ApEx we don't know what
537 * it is, just read it for now, and doesn't
538 * even care about returnval */
539 read (rc->stream_fd, buf, 56);
540 }
541 } else if (cond == G_IO_ERR) {
542 /* XXX */
543 }
544
545 return RAOP_EOK0;
546}

548static gint
549raop_client_stream_sample (raop_client_t *rc, guint16 *buf, guint32 len)
550{
551 guint8 hdr[] = {0x24, 0x00, 0x00, 0x00,
552 0xF0, 0xFF, 0x00, 0x00,
553 0x00, 0x00, 0x00, 0x00,
554 0x00, 0x00, 0x00, 0x00};
555 int i;
556 short cnt;
557 guint8 *pbuf;
558 guint8 iv[16];
559 guint32 offset = 0;
560
561 cnt = len + 3 + 12;
562 cnt = GUINT16_TO_BE (cnt)((((guint16) ( (guint16) ((guint16) (cnt) >> 8) | (guint16
) ((guint16) (cnt) << 8)))));
563 memcpy (hdr + 2, (void *) &cnt, sizeof (cnt));
564
565 memset (rc->sbuf, 0, sizeof (rc->sbuf));
566 memcpy (rc->sbuf, hdr, sizeof (hdr));
567 pbuf = rc->sbuf + sizeof (hdr);
568
569 /* ALAC frame header */
570 write_bits (pbuf, 1, 3, &offset); /* # of channels */
571 write_bits (pbuf, 0, 4, &offset); /* output waiting? */
572 write_bits (pbuf, 0, 4, &offset); /* unknown */
573 write_bits (pbuf, 0, 8, &offset); /* unknown */
574 write_bits (pbuf, 0, 1, &offset); /* sample size follows the header */
575 write_bits (pbuf, 0, 2, &offset); /* unknown */
576 write_bits (pbuf, 1, 1, &offset); /* uncompressed */
577
578 for (i = 0; i < len/2; i++) {
579 write_bits (pbuf, (buf[i]) >> 8, 8, &offset);
580 write_bits (pbuf, (buf[i]) & 0xff, 8, &offset);
581 }
582 memcpy (iv, rc->aes_iv, sizeof (iv));
583 AES_cbc_encrypt (pbuf, pbuf,
584 (len + 3) / 16 * 16,
585 rc->aes_key, iv, TRUE(!(0)));
586
587 rc->sbuf_size = len + 3 + sizeof (hdr);
588 rc->sbuf_offset = 0;
589
590 return RAOP_EOK0;
591}

680 rc->io_state = RAOP_IO_UNDEFINED0x00;
